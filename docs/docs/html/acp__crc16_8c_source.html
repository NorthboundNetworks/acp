42static uint16_t crc16_table[256];
43static int table_initialized = 0;
44
48void acp_crc16_init_table(void)
49{
50 if (table_initialized)
51 {
52 return; /* Already initialized */
53 }
54
55 for (int i = 0; i < 256; i++)
56 {
57 uint16_t crc = (uint16_t)(i << 8);
58
59 for (int j = 0; j < 8; j++)
60 {
61 if (crc & 0x8000)
62 {
63 crc = (crc << 1) ^ ACP_CRC16_POLY;
64 }
65 else
66 {
67 crc = crc << 1;
68 }
69 }
70
71 crc16_table[i] = crc;
72 }
73
74 table_initialized = 1;
75}
76
80const uint16_t *acp_crc16_get_table(void)
81{
82 acp_crc16_init_table();
83 return crc16_table;
84}

101uint16_t acp_crc16_update_byte(uint16_t crc, uint8_t byte)
102{
103 acp_crc16_init_table(); /* Ensure table is initialized */
104
105 uint8_t table_index = (uint8_t)((crc >> 8) ^ byte);
106 return (crc << 8) ^ crc16_table[table_index];
107}
108
112uint16_t acp_crc16_update(uint16_t crc, const uint8_t *data, size_t length)
113{
114 if (data == NULL || length == 0)
115 {
116 return crc;
117 }
118
119 acp_crc16_init_table(); /* Ensure table is initialized */
120
121 for (size_t i = 0; i < length; i++)
122 {
123 crc = acp_crc16_update_byte(crc, data[i]);
124 }
125
126 return crc;
127}
128
132uint16_t acp_crc16_finalize(uint16_t crc)
133{
134 /* For CRC16-CCITT, no final XOR is applied */
135 return crc ^ ACP_CRC16_FINAL_XOR;
136}
137
141uint16_t acp_crc16_calculate(const uint8_t *data, size_t length)
142{
143 if (data == NULL)
144 {
145 return 0;
146 }
147
148 uint16_t crc = acp_crc16_init();
149 crc = acp_crc16_update(crc, data, length);
150 return acp_crc16_finalize(crc);
151}
152
156int acp_crc16_verify(const uint8_t *data, size_t length, uint16_t expected_crc)
157{
158 uint16_t calculated_crc = acp_crc16_calculate(data, length);
159 return (calculated_crc == expected_crc) ? 1 : 0;
160}

221int acp_crc16_self_test(void)
222{
223 size_t test_count;
224 const acp_crc16_test_vector_t *vectors = acp_crc16_get_test_vectors(&test_count);
225 int failures = 0;
226
227 for (size_t i = 0; i < test_count; i++)
228 {
229 uint16_t calculated = acp_crc16_calculate(vectors[i].data, vectors[i].length);
230
231 if (calculated != vectors[i].expected_crc)
232 {
233 failures++;
234 }
235 }
236
237 return failures;
238}

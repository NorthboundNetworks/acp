46int acp_frame_encode(const acp_frame_t *frame, uint8_t *output, size_t output_size, size_t *bytes_written)
47{
48 if (!frame || !output || !bytes_written)
49 {
50 return ACP_ERR_INVALID_PARAM;
51 }
52
53 *bytes_written = 0;
54
55 /* Calculate variable header size based on flags */
56 size_t header_size = acp_wire_header_size(frame->flags);
57 size_t wire_frame_size = header_size + frame->length + 2; /* +2 for CRC */
58
59 /* Check if we have space for worst-case COBS encoding */
60 size_t max_encoded_size = acp_cobs_max_encoded_size(wire_frame_size) + 2; /* +2 for delimiters */
61 if (output_size < max_encoded_size)
62 {
63 ACP_LOG_ERROR("Output buffer too small: need %zu, have %zu", max_encoded_size, output_size);
64 return ACP_ERR_BUFFER_TOO_SMALL;
65 }
66
67 /* Build wire frame in temporary buffer */
68 uint8_t wire_frame[ACP_MAX_FRAME_SIZE];
69 if (wire_frame_size > sizeof(wire_frame))
70 {
71 ACP_LOG_ERROR("Frame too large: %zu bytes", wire_frame_size);
72 return ACP_ERR_PAYLOAD_TOO_LARGE;
73 }
74
75 /* Create base wire header */
76 acp_wire_header_base_t *base_header = (acp_wire_header_base_t *)wire_frame;
77 base_header->version = frame->version;
78 base_header->type = frame->type;
79 base_header->flags = frame->flags;
80 base_header->reserved = 0;
81 base_header->length = frame->length; /* Host byte order for now */
82
83 /* Convert length to network byte order */
84 base_header->length = ((base_header->length & 0xFF) << 8) | ((base_header->length >> 8) & 0xFF);
85
86 /* Add conditional sequence field if authenticated */
87 uint8_t *payload_start = wire_frame + sizeof(acp_wire_header_base_t);
88 if (frame->flags & ACP_FLAG_AUTHENTICATED)
89 {
90 /* Add sequence number in network byte order */
91 uint32_t seq_be = ((frame->sequence & 0xFF) << 24) |
92 (((frame->sequence >> 8) & 0xFF) << 16) |
93 (((frame->sequence >> 16) & 0xFF) << 8) |
94 ((frame->sequence >> 24) & 0xFF);
95 memcpy(payload_start, &seq_be, sizeof(uint32_t));
96 payload_start += sizeof(uint32_t);
97 }
98
99 /* Copy payload */
100 if (frame->length > 0)
101 {
102 memcpy(payload_start, frame->payload, frame->length);
103 }
104
105 /* Calculate and append CRC16 */
106 uint16_t crc = acp_crc16_calculate(wire_frame, wire_frame_size - 2);
107 wire_frame[wire_frame_size - 2] = (uint8_t)(crc & 0xFF);
108 wire_frame[wire_frame_size - 1] = (uint8_t)((crc >> 8) & 0xFF);
109
110 /* COBS encode the frame */
111 size_t encoded_len;
112 int result = acp_cobs_encode(wire_frame, wire_frame_size, output + 1, output_size - 2, &encoded_len);
113 if (result != ACP_OK)
114 {
115 ACP_LOG_ERROR("COBS encoding failed: %d", result);
116 return result;
117 }
118
119 /* Add frame delimiters */
120 output[0] = ACP_COBS_DELIMITER;
121 output[encoded_len + 1] = ACP_COBS_DELIMITER;
122
123 *bytes_written = encoded_len + 2;
124
125 ACP_LOG_DEBUG("Encoded frame: type=0x%02X, payload=%zu bytes, total=%zu bytes",
126 frame->type, (size_t)frame->length, *bytes_written);
127
128 return ACP_OK;
129}

131int acp_frame_decode(const uint8_t *input, size_t input_size, acp_frame_t *frame, size_t *bytes_consumed)
132{
133 if (!input || !frame || !bytes_consumed)
134 {
135 return ACP_ERR_INVALID_PARAM;
136 }
137
138 *bytes_consumed = 0;
139
140 /* Need at least minimum frame size (base header + CRC + delimiters) */
141 if (input_size < sizeof(acp_wire_header_base_t) + 2 + 2)
142 { /* base header + CRC + delimiters */
143 return ACP_ERR_NEED_MORE_DATA;
144 }
145
146 /* Find frame boundaries */
147 if (input[0] != ACP_COBS_DELIMITER)
148 {
149 ACP_LOG_WARN("Missing frame start delimiter");
150 return ACP_ERR_MALFORMED_FRAME;
151 }
152
153 /* Find end delimiter */
154 size_t frame_end = 0;
155 for (size_t i = 1; i < input_size; i++)
156 {
157 if (input[i] == ACP_COBS_DELIMITER)
158 {
159 frame_end = i;
160 break;
161 }
162 }
163
164 if (frame_end == 0)
165 {
166 return ACP_ERR_NEED_MORE_DATA;
167 }
168
169 /* COBS decode the frame content */
170 uint8_t decoded_frame[ACP_MAX_FRAME_SIZE];
171 size_t decoded_len;
172 int result = acp_cobs_decode(input + 1, frame_end - 1, decoded_frame, sizeof(decoded_frame), &decoded_len);
173 if (result != ACP_OK)
174 {
175 ACP_LOG_ERROR("COBS decoding failed: %d", result);
176 return result;
177 }
178
179 /* Need at least base header + CRC */
180 if (decoded_len < sizeof(acp_wire_header_base_t) + 2)
181 {
182 ACP_LOG_WARN("Decoded frame too short: %zu bytes", decoded_len);
183 return ACP_ERR_MALFORMED_FRAME;
184 }
185
186 /* Parse base wire header */
187 const acp_wire_header_base_t *base_header = (const acp_wire_header_base_t *)decoded_frame;
188
189 /* Calculate expected header size based on flags */
190 size_t expected_header_size = acp_wire_header_size(base_header->flags);
191
192 /* Verify we have enough data for the full header */
193 if (decoded_len < expected_header_size + 2)
194 {
195 ACP_LOG_WARN("Decoded frame too short for header: need %zu+2, have %zu",
196 expected_header_size, decoded_len);
197 return ACP_ERR_MALFORMED_FRAME;
198 }
199
200 /* Verify CRC */
201 uint16_t calculated_crc = acp_crc16_calculate(decoded_frame, decoded_len - 2);
202 uint16_t received_crc = ((uint16_t)decoded_frame[decoded_len - 1] << 8) | decoded_frame[decoded_len - 2];
203
204 if (calculated_crc != received_crc)
205 {
206 ACP_LOG_ERROR("CRC mismatch: calculated=0x%04X, received=0x%04X", calculated_crc, received_crc);
207 return ACP_ERR_CRC_MISMATCH;
208 }
209
210 /* Convert length from network byte order */
211 uint16_t payload_len = ((base_header->length & 0xFF) << 8) | ((base_header->length >> 8) & 0xFF);
212
213 /* Verify frame size consistency */
214 if (expected_header_size + payload_len + 2 != decoded_len)
215 {
216 ACP_LOG_ERROR("Frame size mismatch: header says %zu+%u+2, got %zu",
217 expected_header_size, payload_len, decoded_len);
218 return ACP_ERR_MALFORMED_FRAME;
219 }
220
221 /* Fill in frame structure */
222 frame->version = base_header->version;
223 frame->type = base_header->type;
224 frame->flags = base_header->flags;
225 frame->length = payload_len;
226
227 /* Parse conditional sequence field */
228 uint32_t sequence = 0;
229 const uint8_t *payload_start = decoded_frame + sizeof(acp_wire_header_base_t);
230 if (base_header->flags & ACP_FLAG_AUTHENTICATED)
231 {
232 /* Extract sequence number from network byte order */
233 uint32_t seq_be;
234 memcpy(&seq_be, payload_start, sizeof(uint32_t));
235 sequence = ((seq_be & 0xFF) << 24) |
236 (((seq_be >> 8) & 0xFF) << 16) |
237 (((seq_be >> 16) & 0xFF) << 8) |
238 ((seq_be >> 24) & 0xFF);
239 payload_start += sizeof(uint32_t);
240 }
241 frame->sequence = sequence;
242
243 /* Copy payload if present */
244 if (payload_len > 0)
245 {
246 if (payload_len > ACP_MAX_PAYLOAD_SIZE)
247 {
248 ACP_LOG_ERROR("Payload too large: %u bytes", payload_len);
249 return ACP_ERR_PAYLOAD_TOO_LARGE;
250 }
251 memcpy(frame->payload, payload_start, payload_len);
252 }
253
254 *bytes_consumed = frame_end + 1;
255
256 ACP_LOG_DEBUG("Decoded frame: type=0x%02X, payload=%u bytes, consumed=%zu bytes",
257 frame->type, payload_len, *bytes_consumed);
258
259 return ACP_OK;
260}

262size_t acp_frame_encoded_size(const acp_frame_t *frame)
263{
264 if (!frame)
265 {
266 return 0;
267 }
268
269 size_t header_size = acp_wire_header_size(frame->flags);
270 size_t wire_size = header_size + frame->length + 2; /* +2 for CRC */
271 return acp_cobs_max_encoded_size(wire_size) + 2; /* +2 for delimiters */
272}

278int acp_frame_create_telemetry(acp_frame_t *frame, uint8_t msg_type, uint32_t sequence,
279 const void *payload, size_t payload_len)
280{
281 if (!frame)
282 {
283 return ACP_ERR_INVALID_PARAM;
284 }
285
286 if (payload_len > ACP_MAX_PAYLOAD_SIZE)
287 {
288 return ACP_ERR_PAYLOAD_TOO_LARGE;
289 }
290
291 frame->version = ACP_PROTOCOL_VERSION;
292 frame->type = msg_type;
293 frame->flags = 0; /* No special flags for basic telemetry */
294 frame->sequence = sequence;
295 frame->length = (uint16_t)payload_len;
296
297 if (payload_len > 0 && payload)
298 {
299 memcpy(frame->payload, payload, payload_len);
300 }
301
302 return ACP_OK;
303}

305int acp_frame_create_command(acp_frame_t *frame, uint8_t msg_type, uint32_t sequence,
306 uint8_t priority, const void *payload, size_t payload_len)
307{
308 if (!frame)
309 {
310 return ACP_ERR_INVALID_PARAM;
311 }
312
313 if (payload_len > ACP_MAX_PAYLOAD_SIZE)
314 {
315 return ACP_ERR_PAYLOAD_TOO_LARGE;
316 }
317
318 frame->version = ACP_PROTOCOL_VERSION;
319 frame->type = msg_type;
320 frame->flags = (priority & 0x03) << 2; /* Pack priority into flags */
321 frame->sequence = sequence;
322 frame->length = (uint16_t)payload_len;
323
324 if (payload_len > 0 && payload)
325 {
326 memcpy(frame->payload, payload, payload_len);
327 }
328
329 return ACP_OK;
330}

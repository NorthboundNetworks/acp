54uint64_t acp_platform_get_time_ms(void)
55{
56 struct timespec ts;
57 if (clock_gettime(CLOCK_MONOTONIC, &ts) == 0)
58 {
59 return (uint64_t)ts.tv_sec * 1000 + ts.tv_nsec / 1000000;
60 }
61 return 0;
62}

67void acp_platform_sleep_ms(uint32_t ms)
68{
69 struct timespec ts;
70 ts.tv_sec = ms / 1000;
71 ts.tv_nsec = (ms % 1000) * 1000000;
72 nanosleep(&ts, NULL);
73}

82acp_mutex_t *acp_platform_mutex_create(void)
83{
84 pthread_mutex_t *mutex = malloc(sizeof(pthread_mutex_t));
85 if (mutex != NULL)
86 {
87 if (pthread_mutex_init(mutex, NULL) != 0)

89 free(mutex);

92 }

138int acp_platform_keystore_get_key(uint32_t key_id, uint8_t *key_out, size_t *key_len)
139{
140 if (key_out == NULL || key_len == NULL)
141 {
142 return ACP_ERR_INVALID_PARAM;
143 }
144
145 char filename[256];
146 snprintf(filename, sizeof(filename), "/tmp/acp_keys/key_%08x.bin", key_id);
147
148 FILE *file = fopen(filename, "rb");
149 if (file == NULL)
150 {
151 return ACP_ERR_KEY_NOT_FOUND;
152 }
153
154 fseek(file, 0, SEEK_END);
155 long file_size = ftell(file);
156 fseek(file, 0, SEEK_SET);
157
158 if (file_size <= 0 || (size_t)file_size > *key_len)
159 {
160 fclose(file);
161 return ACP_ERR_BUFFER_TOO_SMALL;
162 }
163
164 size_t read_size = fread(key_out, 1, file_size, file);
165 fclose(file);
166
167 if (read_size != (size_t)file_size)
168 {
169 return ACP_ERR_KEYSTORE_READ;
170 }
171
172 *key_len = read_size;
173 return ACP_OK;
174}

179int acp_platform_keystore_store_key(uint32_t key_id, const uint8_t *key, size_t key_len)
180{
181 if (key == NULL || key_len == 0)
182 {
183 return ACP_ERR_INVALID_PARAM;
184 }
185
186 /* Ensure keystore directory exists */
187 mkdir("/tmp/acp_keys", 0700);
188
189 char filename[256];
190 snprintf(filename, sizeof(filename), "/tmp/acp_keys/key_%08x.bin", key_id);
191
192 FILE *file = fopen(filename, "wb");
193 if (file == NULL)
194 {
195 return ACP_ERR_KEYSTORE_WRITE;
196 }
197
198 size_t written = fwrite(key, 1, key_len, file);
199 fclose(file);
200
201 if (written != key_len)
202 {
203 return ACP_ERR_KEYSTORE_WRITE;
204 }
205
206 return ACP_OK;
207}

33int test_cobs_basic(void)
34{
35 /* Test case: "Hello\0World" */
36 const uint8_t input[] = {'H', 'e', 'l', 'l', 'o', 0, 'W', 'o', 'r', 'l', 'd'};
37 uint8_t encoded[20];
38 uint8_t decoded[20];
39 size_t encoded_len, decoded_len;
40
41 printf("Testing COBS encode/decode...\n");
42
43 /* Encode */
44 int result = acp_cobs_encode(input, sizeof(input), encoded, sizeof(encoded), &encoded_len);
45 if (result != ACP_OK)
46 {
47 printf("COBS encode failed: %d\n", result);
48 return -1;
49 }
50
51 printf("Input: ");
52 for (size_t i = 0; i < sizeof(input); i++)
53 {
54 printf("%02X ", input[i]);
55 }
56 printf("\n");
57
58 printf("Encoded: ");
59 for (size_t i = 0; i < encoded_len; i++)
60 {
61 printf("%02X ", encoded[i]);
62 }
63 printf("(length: %zu)\n", encoded_len);
64
65 /* Decode */
66 result = acp_cobs_decode(encoded, encoded_len, decoded, sizeof(decoded), &decoded_len);
67 if (result != ACP_OK)
68 {
69 printf("COBS decode failed: %d\n", result);
70 return -1;
71 }
72
73 printf("Decoded: ");
74 for (size_t i = 0; i < decoded_len; i++)
75 {
76 printf("%02X ", decoded[i]);
77 }
78 printf("(length: %zu)\n", decoded_len);
79
80 /* Verify */
81 if (decoded_len != sizeof(input) || memcmp(input, decoded, sizeof(input)) != 0)

83 printf("COBS test FAILED - data mismatch\n");
84 return -1;
85 }
86
87 printf("COBS test PASSED\n");

91int test_cobs_streaming(void)
92{
93 printf("\nTesting COBS streaming decoder...\n");
94
95 /* Create test frame: encode "Test" */
96 const uint8_t test_data[] = "Test";
97 uint8_t encoded[10];
98 size_t encoded_len;
99
100 int result = acp_cobs_encode(test_data, 4, encoded, sizeof(encoded), &encoded_len);
101 if (result != ACP_OK)
102 {
103 printf("Failed to encode test data\n");
104 return -1;
105 }
106
107 /* Create framed data (add delimiters) */
108 uint8_t framed[12];
109 framed[0] = ACP_COBS_DELIMITER; /* Frame start */
110 memcpy(framed + 1, encoded, encoded_len);
111 framed[encoded_len + 1] = ACP_COBS_DELIMITER; /* Frame end */
112
113 /* Setup streaming decoder */
114 uint8_t decoder_buffer[10];
115 acp_cobs_decoder_t decoder;
116 result = acp_cobs_decoder_init(&decoder, decoder_buffer, sizeof(decoder_buffer));
117 if (result != ACP_OK)
118 {
119 printf("Failed to initialize decoder\n");
120 return -1;
121 }
122
123 /* Feed bytes one by one */
124 for (size_t i = 0; i < encoded_len + 2; i++)
125 {
126 int feed_result = acp_cobs_decoder_feed_byte(&decoder, framed[i]);
127 if (feed_result < 0)
128 {
129 printf("Decoder error at byte %zu: %d\n", i, feed_result);
130 return -1;
131 }
132 if (feed_result == 1)
133 {
134 printf("Frame complete at byte %zu\n", i);
135 break;
136 }
137 }
138
139 /* Get decoded frame */
140 uint8_t decoded[10];
141 size_t decoded_len;
142 result = acp_cobs_decoder_get_frame(&decoder, decoded, sizeof(decoded), &decoded_len);
143 if (result != ACP_OK)
144 {
145 printf("Failed to get decoded frame: %d\n", result);
146 return -1;
147 }
148
149 /* Verify */
150 if (decoded_len != 4 || memcmp(test_data, decoded, 4) != 0)
151 {
152 printf("Streaming decoder test FAILED\n");
153 return -1;
154 }
155
156 printf("Streaming decoder test PASSED\n");
157 return 0;
158}

174int scp_crypto_hmac_generate(const uint8_t *key,
175 const uint8_t *message,
176 uint16_t message_len,
177 uint8_t *tag_out)
178{
179
180 if (!key || !message || !tag_out || message_len == 0)
181 {
182 return SCP_CRYPTO_ERR_INVALID_PARAM;
183 }
184
185 if (!crypto_state.initialized)
186 {
187 return SCP_CRYPTO_ERR_HARDWARE;
188 }
189
190 uint8_t hmac_full[SCP_HMAC_FULL_SIZE];
191 int ret;
192
193 // Calculate HMAC-SHA256
194 ret = mbedtls_md_hmac(mbedtls_md_info_from_type(MBEDTLS_MD_SHA256),
195 key, SCP_HMAC_KEY_SIZE,
196 message, message_len,
197 hmac_full);

199 if (ret != 0)

201 ESP_LOGE(TAG, "HMAC generation failed: -0x%04X", -ret);
202 return SCP_CRYPTO_ERR_HARDWARE;
203 }
204
205 // Truncate to 128 bits (16 bytes) for protocol efficiency
206 memcpy(tag_out, hmac_full, SCP_HMAC_TAG_SIZE);
207
208 // Clear full HMAC from memory
209 scp_crypto_secure_memclear(hmac_full, sizeof(hmac_full));
210
211 crypto_state.stats.hmac_operations++;
212
213 ESP_LOGD(TAG, "HMAC generated for %u bytes", message_len);
214 return SCP_CRYPTO_OK;

217int scp_crypto_hmac_verify(const uint8_t *key,
218 const uint8_t *message,
219 uint16_t message_len,
220 const uint8_t *tag_expected)
221{
222
223 if (!key || !message || !tag_expected || message_len == 0)
224 {
225 return SCP_CRYPTO_ERR_INVALID_PARAM;
226 }
227
228 uint8_t tag_calculated[SCP_HMAC_TAG_SIZE];
229
230 // Generate HMAC for comparison
231 int ret = scp_crypto_hmac_generate(key, message, message_len, tag_calculated);
232 if (ret != SCP_CRYPTO_OK)
233 {
234 return ret;
235 }
236
237 // Constant-time comparison to prevent timing attacks
238 bool match = scp_crypto_secure_memcmp(tag_calculated, tag_expected, SCP_HMAC_TAG_SIZE);
239
240 // Clear calculated tag from memory
241 scp_crypto_secure_memclear(tag_calculated, sizeof(tag_calculated));
242
243 if (match)
244 {
245 crypto_state.stats.auth_successes++;
246 ESP_LOGD(TAG, "HMAC verification successful");
247 return SCP_CRYPTO_OK;
248 }
249 else
250 {
251 crypto_state.stats.auth_failures++;
252 ESP_LOGW(TAG, "HMAC verification failed");
253 return SCP_CRYPTO_ERR_AUTH_FAIL;
254 }

500

502 esp_err_t err = nvs_entry_find(NVS_DEFAULT_PART_NAME, SCP_NVS_NAMESPACE, NVS_TYPE_BLOB, &it);
503 *count_out = 0;
504
505 while (err == ESP_OK && it != NULL && *count_out < max_keys)
506 {
507 nvs_entry_info_t info;
508 nvs_entry_info(it, &info);
509
510 // Check if this is a key entry

512 {
513 // Extract key ID
514 uint32_t key_id = (uint32_t)strtoul(info.key + strlen(SCP_NVS_KEY_PREFIX), NULL, 10);
515
516 // Load the key
517 if (scp_crypto_keystore_load(key_id, &key_list[*count_out]) == SCP_CRYPTO_OK)
518 {
519 (*count_out)++;
520 }
521 }
522
523 err = nvs_entry_next(&it);
524 }
525
526 if (it)
527 {
528 nvs_release_iterator(it);
529 }
530
531 ESP_LOGI(TAG, "Listed %u keys from store", *count_out);
532 return SCP_CRYPTO_OK;

57/* SHA-256 helper macros */
58#define ROTR(x, n) (((x) >> (n)) | ((x) << (32 - (n))))
59#define CH(x, y, z) (((x) & (y)) ^ (~(x) & (z)))
60#define MAJ(x, y, z) (((x) & (y)) ^ ((x) & (z)) ^ ((y) & (z)))
61#define EP0(x) (ROTR(x, 2) ^ ROTR(x, 13) ^ ROTR(x, 22))
62#define EP1(x) (ROTR(x, 6) ^ ROTR(x, 11) ^ ROTR(x, 25))
63#define SIG0(x) (ROTR(x, 7) ^ ROTR(x, 18) ^ ((x) >> 3))
64#define SIG1(x) (ROTR(x, 17) ^ ROTR(x, 19) ^ ((x) >> 10))

69void acp_sha256_init(acp_sha256_ctx_t *ctx)
70{
71 if (!ctx)
72 return;
73
74 memcpy(ctx->state, H0, sizeof(H0));
75 ctx->bit_len = 0;
76 ctx->buffer_len = 0;
77 memset(ctx->buffer, 0, sizeof(ctx->buffer));
78}

83static void acp_sha256_process_block(acp_sha256_ctx_t *ctx, const uint8_t *data)
84{
85 uint32_t w[64];
86 uint32_t a, b, c, d, e, f, g, h;
87 uint32_t t1, t2;
88 int i;
89
90 /* Prepare the message schedule */
91 for (i = 0; i < 16; i++)
92 {
93 w[i] = ((uint32_t)data[i * 4] << 24) |
94 ((uint32_t)data[i * 4 + 1] << 16) |
95 ((uint32_t)data[i * 4 + 2] << 8) |
96 ((uint32_t)data[i * 4 + 3]);
97 }
98
99 for (i = 16; i < 64; i++)
100 {
101 w[i] = SIG1(w[i - 2]) + w[i - 7] + SIG0(w[i - 15]) + w[i - 16];
102 }
103
104 /* Initialize working variables */
105 a = ctx->state[0];
106 b = ctx->state[1];
107 c = ctx->state[2];
108 d = ctx->state[3];
109 e = ctx->state[4];
110 f = ctx->state[5];
111 g = ctx->state[6];
112 h = ctx->state[7];
113
114 /* Compression function main loop */
115 for (i = 0; i < 64; i++)
116 {
117 t1 = h + EP1(e) + CH(e, f, g) + K[i] + w[i];
118 t2 = EP0(a) + MAJ(a, b, c);
119
120 h = g;
121 g = f;
122 f = e;
123 e = d + t1;
124 d = c;
125 c = b;
126 b = a;
127 a = t1 + t2;
128 }
129
130 /* Add the compressed chunk to the current hash value */
131 ctx->state[0] += a;
132 ctx->state[1] += b;
133 ctx->state[2] += c;
134 ctx->state[3] += d;
135 ctx->state[4] += e;
136 ctx->state[5] += f;
137 ctx->state[6] += g;
138 ctx->state[7] += h;
139}

144void acp_sha256_update(acp_sha256_ctx_t *ctx, const uint8_t *data, size_t len)
145{
146 if (!ctx || !data)
147 return;
148
149 size_t i = 0;
150
151 /* Process data byte by byte if buffer is partially filled */
152 while (i < len && ctx->buffer_len > 0 && ctx->buffer_len < ACP_SHA256_BLOCK_SIZE)
153 {
154 ctx->buffer[ctx->buffer_len++] = data[i++];
155
156 if (ctx->buffer_len == ACP_SHA256_BLOCK_SIZE)
157 {
158 acp_sha256_process_block(ctx, ctx->buffer);
159 ctx->buffer_len = 0;
160 ctx->bit_len += 512;
161 }
162 }
163
164 /* Process complete blocks directly */
165 while (i + ACP_SHA256_BLOCK_SIZE <= len)
166 {
167 acp_sha256_process_block(ctx, &data[i]);
168 i += ACP_SHA256_BLOCK_SIZE;
169 ctx->bit_len += 512;
170 }
171
172 /* Buffer remaining bytes */
173 while (i < len)
174 {
175 ctx->buffer[ctx->buffer_len++] = data[i++];
176 }
177}

182void acp_sha256_final(acp_sha256_ctx_t *ctx, uint8_t *hash)
183{
184 if (!ctx || !hash)
185 return;
186
187 uint64_t bit_len = ctx->bit_len + (ctx->buffer_len * 8);
188 size_t i;
189
190 /* Pad the message */
191 ctx->buffer[ctx->buffer_len++] = 0x80;
192
193 /* If we don't have room for the length, pad to end and process */
194 if (ctx->buffer_len > 56)
195 {
196 while (ctx->buffer_len < ACP_SHA256_BLOCK_SIZE)
197 {
198 ctx->buffer[ctx->buffer_len++] = 0x00;
199 }
200 acp_sha256_process_block(ctx, ctx->buffer);
201 ctx->buffer_len = 0;
202 }
203
204 /* Pad to 56 bytes */
205 while (ctx->buffer_len < 56)
206 {
207 ctx->buffer[ctx->buffer_len++] = 0x00;
208 }
209
210 /* Append length in bits as big-endian 64-bit number */
211 for (i = 0; i < 8; i++)
212 {
213 ctx->buffer[56 + i] = (uint8_t)(bit_len >> (56 - i * 8));
214 }
215
216 acp_sha256_process_block(ctx, ctx->buffer);
217
218 /* Output hash in big-endian format */
219 for (i = 0; i < 8; i++)
220 {
221 hash[i * 4] = (uint8_t)(ctx->state[i] >> 24);
222 hash[i * 4 + 1] = (uint8_t)(ctx->state[i] >> 16);
223 hash[i * 4 + 2] = (uint8_t)(ctx->state[i] >> 8);
224 hash[i * 4 + 3] = (uint8_t)(ctx->state[i]);
225 }
226}

231void acp_sha256(const uint8_t *data, size_t len, uint8_t *hash)
232{
233 acp_sha256_ctx_t ctx;
234 acp_sha256_init(&ctx);
235 acp_sha256_update(&ctx, data, len);
236 acp_sha256_final(&ctx, hash);
237
238 /* Clear sensitive data */
239 acp_crypto_clear(&ctx, sizeof(ctx));
240}

246void acp_hmac_sha256(const uint8_t *key, size_t key_len,
247 const uint8_t *data, size_t data_len,
248 uint8_t *mac)
249{
250 uint8_t k_pad[ACP_SHA256_BLOCK_SIZE];
251 uint8_t inner_hash[ACP_SHA256_SIZE];
252 acp_sha256_ctx_t ctx;
253 size_t i;
254
255 if (!key || !data || !mac)
256 return;
257
258 /* Prepare the key */
259 memset(k_pad, 0, sizeof(k_pad));
260
261 if (key_len > ACP_SHA256_BLOCK_SIZE)
262 {
263 /* Hash the key if it's too long */
264 acp_sha256(key, key_len, k_pad);
265 }
266 else
267 {
268 /* Use key directly, zero-padded */
269 memcpy(k_pad, key, key_len);
270 }
271
272 /* Create inner padding (key XOR ipad) */
273 for (i = 0; i < ACP_SHA256_BLOCK_SIZE; i++)
274 {
275 k_pad[i] ^= 0x36;
276 }
277
278 /* Inner hash: SHA256(K XOR ipad || text) */
279 acp_sha256_init(&ctx);
280 acp_sha256_update(&ctx, k_pad, ACP_SHA256_BLOCK_SIZE);
281 acp_sha256_update(&ctx, data, data_len);
282 acp_sha256_final(&ctx, inner_hash);
283
284 /* Restore key and create outer padding (key XOR opad) */
285 for (i = 0; i < ACP_SHA256_BLOCK_SIZE; i++)
286 {
287 k_pad[i] ^= 0x36; /* Remove ipad */
288 k_pad[i] ^= 0x5c; /* Add opad */
289 }
290
291 /* Outer hash: SHA256(K XOR opad || inner_hash) */
292 acp_sha256_init(&ctx);
293 acp_sha256_update(&ctx, k_pad, ACP_SHA256_BLOCK_SIZE);
294 acp_sha256_update(&ctx, inner_hash, ACP_SHA256_SIZE);
295 acp_sha256_final(&ctx, mac); /* Output full 32-byte HMAC to mac buffer */
296
297 /* Clear sensitive data */
298 acp_crypto_clear(k_pad, sizeof(k_pad));
299 acp_crypto_clear(inner_hash, sizeof(inner_hash));
300 acp_crypto_clear(&ctx, sizeof(ctx));
301}

307int acp_crypto_memcmp_ct(const void *a, const void *b, size_t len)
308{
309 const uint8_t *pa = (const uint8_t *)a;
310 const uint8_t *pb = (const uint8_t *)b;
311 uint8_t result = 0;
312 size_t i;
313
314 if (!a || !b)
315 return -1;
316
317 /* Constant-time comparison */
318 for (i = 0; i < len; i++)
319 {
320 result |= pa[i] ^ pb[i];
321 }
322
323 return result;
324}

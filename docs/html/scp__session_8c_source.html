132static bool is_sequence_valid(scp_session_t *session, uint8_t seq_num)
133{
134 // For simplicity, we use a simple forward-only sequence check
135 // In production, a sliding window approach would be more robust
136
137 if (seq_num == session->last_seq_num)
138 {
139 // Duplicate sequence number
140 ESP_LOGW(TAG, "Duplicate sequence number %u for session %u",
141 seq_num, session->session_id);
142 return false;
143 }
144
145 // Allow sequence number wrapping (0-255)
146 uint8_t expected_min = (session->last_seq_num + 1) & 0xFF;
147 uint8_t expected_max = (session->last_seq_num + 10) & 0xFF; // Allow up to 10 ahead
148
149 if (expected_max > expected_min)
150 {
151 // No wrap-around
152 if (seq_num >= expected_min && seq_num <= expected_max)
153 {
154 return true;
155 }
156 }
157 else
158 {
159 // Wrap-around case
160 if (seq_num >= expected_min || seq_num <= expected_max)
161 {
162 return true;
163 }
164 }
165
166 ESP_LOGW(TAG, "Invalid sequence number %u for session %u (last: %u)",
167 seq_num, session->session_id, session->last_seq_num);
168 return false;
169}

178static bool update_rate_limit(scp_session_t *session, uint32_t current_time)
179{
180 scp_rate_limit_t *rl = &session->rate_limit;
181
182 // Check if we're in penalty timeout
183 if (rl->penalty_until > current_time)
184 {
185 ESP_LOGD(TAG, "Session %u in penalty timeout", session->session_id);
186 return false;
187 }
188
189 // Reset penalty state if expired
190 if (rl->penalty_until > 0 && current_time >= rl->penalty_until)
191 {
192 rl->penalty_until = 0;
193 rl->command_count = 0;
194 rl->burst_used = 0;
195 ESP_LOGI(TAG, "Penalty timeout expired for session %u", session->session_id);
196 }
197
198 // Check if we need to start a new rate window
199 uint32_t window_age = current_time - rl->window_start;
200 if (window_age >= 1000)
201 { // 1 second window
202 rl->window_start = current_time;
203 rl->command_count = 0;
204 // Don't reset burst_used - it accumulates over burst window
205 }
206
207 // Check burst window reset
208 if (window_age >= SCP_RATE_BURST_WINDOW)
209 {
210 rl->burst_used = 0;
211 }
212
213 // Check sustained rate limit
214 if (rl->command_count >= SCP_RATE_SUSTAINED_CPS)
215 {
216 // Check if we can use burst allowance
217 if (rl->burst_used >= SCP_RATE_BURST_CPS)
218 {
219 // Rate limit exceeded - apply penalty
220 rl->penalty_until = current_time + SCP_RATE_PENALTY_TIME;
221 session->rate_limit_hits++;
222 global_stats.rate_limit_events++;
223
224 ESP_LOGW(TAG, "Rate limit exceeded for session %u (penalty: %u ms)",
225 session->session_id, SCP_RATE_PENALTY_TIME);
226 return false;
227 }
228 }
229
230 // Allow command - update counters
231 rl->command_count++;
232 if (rl->command_count > SCP_RATE_SUSTAINED_CPS)
233 {
234 rl->burst_used++;
235 }
236
237 return true;
238}

268 return false; // No configuration changes

274 // Observers cannot execute any commands (telemetry only)
275 return false;

278 return false;

316bool scp_session_init(void)
317{
318 if (session_system_initialized)
319 {
320 ESP_LOGW(TAG, "Session system already initialized");
321 return true;
322 }
323
324 ESP_LOGI(TAG, "Initializing SCP session management");
325
326 // Create mutex for thread safety

328 if (!session_mutex)

331 return false;

333

336 session_manager.active_count = 0;

338
339 // Initialize all sessions as inactive

342 session_manager.sessions[i].state = SCP_SESSION_STATE_INACTIVE;
343 }
344
345 // Initialize global statistics
346 memset(&global_stats, 0, sizeof(global_stats));
347
348 session_system_initialized = true;
349 ESP_LOGI(TAG, "Session system initialization complete");
350
351 return true;

354bool scp_session_create(uint8_t operator_role,
355 uint8_t connection_type,
356 const uint8_t *auth_key,
357 uint32_t *session_id_out)
358{
359
360 if (!session_system_initialized || !auth_key || !session_id_out)
361 {
362 ESP_LOGE(TAG, "Invalid parameters for session creation");
363 return false;
364 }
365
366 // Validate role (explicit membership to avoid unsigned range warnings)
367 if (!(operator_role == SCP_ROLE_OBSERVER || operator_role == SCP_ROLE_PILOT || operator_role == SCP_ROLE_COMMANDER))
368 {
369 ESP_LOGE(TAG, "Invalid operator role: %u", operator_role);
370 return false;
371 }
372
373 // Validate connection type (explicit membership)
374 if (!(connection_type == SCP_CONN_UART || connection_type == SCP_CONN_USB || connection_type == SCP_CONN_SPI))
375 {
376 ESP_LOGE(TAG, "Invalid connection type: %u", connection_type);
377 return false;
378 }
379
380 if (xSemaphoreTake(session_mutex, portMAX_DELAY) != pdTRUE)
381 {
382 ESP_LOGE(TAG, "Failed to acquire session mutex");
383 return false;
384 }
385
386 bool success = false;
387
388 // Find free session slot
389 scp_session_t *session = find_free_session_slot();
390 if (!session)
391 {
392 ESP_LOGW(TAG, "No free session slots available");
393 goto cleanup;
394 }
395
396 // Allocate session ID
397 uint32_t new_session_id = session_manager.next_session_id++;
398 if (session_manager.next_session_id == SCP_SESSION_ID_INVALID)
399 {
400 session_manager.next_session_id = 1; // Skip invalid ID
401 }
402
403 // Initialize session
404 init_session(session, new_session_id, operator_role, connection_type, auth_key);
405
406 // Update manager state
407 session_manager.active_count++;
408 session_manager.system_uptime_ms = get_current_time_ms();
409
410 // Update global statistics
411 global_stats.total_sessions++;
412 switch (operator_role)
413 {
414 case SCP_ROLE_COMMANDER:
415 global_stats.commander_sessions++;
416 break;
417 case SCP_ROLE_PILOT:
418 global_stats.pilot_sessions++;
419 break;
420 case SCP_ROLE_OBSERVER:
421 global_stats.observer_sessions++;
422 break;
423 }
424
425 *session_id_out = new_session_id;
426 success = true;
427
428 ESP_LOGI(TAG, "Created session %u: role=%s, conn=%s, active=%u/%u",
429 new_session_id,
430 scp_role_get_name(operator_role),
431 scp_connection_get_name(connection_type),
432 session_manager.active_count,
433 SCP_MAX_SESSIONS);
434
435cleanup:
436 xSemaphoreGive(session_mutex);
437 return success;
438}

440bool scp_session_authenticate(uint32_t session_id,
441 uint8_t seq_num,
442 const uint8_t *auth_tag,
443 const uint8_t *message_data,
444 uint16_t message_len)
445{
446
447 if (!session_system_initialized || !message_data)
448 {
449 return false;
450 }
451
452 if (xSemaphoreTake(session_mutex, portMAX_DELAY) != pdTRUE)
453 {
454 ESP_LOGE(TAG, "Failed to acquire session mutex for auth");
455 return false;
456 }
457
458 bool authenticated = false;
459 scp_session_t *session = find_session_by_id(session_id);
460
461 if (!session)
462 {
463 ESP_LOGW(TAG, "Authentication failed: session %u not found", session_id);
464 goto cleanup;
465 }
466
467 if (session->state != SCP_SESSION_STATE_ACTIVE)
468 {
469 ESP_LOGW(TAG, "Authentication failed: session %u not active (state=%u)",
470 session_id, session->state);
471 goto cleanup;
472 }
473
474 // Check sequence number (replay protection)
475 if (!is_sequence_valid(session, seq_num))
476 {
477 session->auth_failures++;
478 global_stats.auth_failures++;
479 ESP_LOGW(TAG, "Authentication failed: invalid sequence number");
480 goto cleanup;
481 }
482
483 // Verify HMAC if present
484 if (auth_tag)
485 {
486 int crypto_result = scp_crypto_hmac_verify(session->auth_key,
487 message_data,
488 message_len,
489 auth_tag);
490 if (crypto_result != SCP_CRYPTO_OK)
491 {
492 session->auth_failures++;
493 global_stats.auth_failures++;
494 ESP_LOGW(TAG, "Authentication failed: HMAC verification failed");
495 goto cleanup;
496 }
497 }
498
499 // Update session activity
500 uint32_t current_time = get_current_time_ms();
501 session->last_activity = current_time;
502 session->last_seq_num = seq_num;
503 session->messages_received++;
504 session->expires_at = current_time + SCP_SESSION_TIMEOUT;
505
506 authenticated = true;
507 ESP_LOGD(TAG, "Authentication successful for session %u", session_id);
508
509cleanup:
510 xSemaphoreGive(session_mutex);
511 return authenticated;
512}

602{
603 if (!session_system_initialized)
604 {
605 return false;
606 }
607
608 if (xSemaphoreTake(session_mutex, portMAX_DELAY) != pdTRUE)
609 {
610 return false;
611 }
612
613 bool closed = false;
614 scp_session_t *session = find_session_by_id(session_id);
615
616 if (session)
617 {
618 uint8_t old_role = session->operator_role;
619
620 // Clear sensitive data
621 scp_crypto_secure_memclear(session->auth_key, sizeof(session->auth_key));
622
623 // Mark as inactive
624 session->state = SCP_SESSION_STATE_INACTIVE;
625 session->session_id = SCP_SESSION_ID_INVALID;
626
627 // Update counters
628 if (session_manager.active_count > 0)
629 {
630 session_manager.active_count--;
631 }
632
633 // Update role statistics
634 switch (old_role)
635 {
636 case SCP_ROLE_COMMANDER:
637 if (global_stats.commander_sessions > 0)
638 global_stats.commander_sessions--;
639 break;
640 case SCP_ROLE_PILOT:
641 if (global_stats.pilot_sessions > 0)
642 global_stats.pilot_sessions--;
643 break;
644 case SCP_ROLE_OBSERVER:
645 if (global_stats.observer_sessions > 0)
646 global_stats.observer_sessions--;
647 break;
648 }
649
650 closed = true;
651 ESP_LOGI(TAG, "Closed session %u (%s), active sessions: %u",
652 session_id, scp_role_get_name(old_role), session_manager.active_count);
653 }

660{
661 if (!session_system_initialized)
662 {
663 return 0;
664 }
665
666 if (xSemaphoreTake(session_mutex, portMAX_DELAY) != pdTRUE)

669 }

687 xSemaphoreGive(session_mutex); // Release mutex for close call

689 xSemaphoreTake(session_mutex, portMAX_DELAY); // Re-acquire
690
691 cleaned_count++;
692 global_stats.expired_sessions++;
693 }
694 }
695
696 if (cleaned_count > 0)
697 {
698 ESP_LOGI(TAG, "Cleaned up %u expired sessions", cleaned_count);
699 }
700
701 xSemaphoreGive(session_mutex);
702 return cleaned_count;

404bool scp_validate_frame_limits(const scp_frame_header_t *header)
405{
406 if (!header)
407 {
408 return false;
409 }
410
411 // Check sync pattern
412 if (header->sync[0] != SCP_SYNC_BYTE1 || header->sync[1] != SCP_SYNC_BYTE2)
413 {
414 return false;
415 }
416
417 // Check protocol version
418 if (header->version != SCP_VERSION_1_1)
419 {
420 return false;
421 }
422
423 // Check message type
424 if (!scp_is_valid_message_type(header->msg_type))
425 {
426 return false;
427 }
428
429 // Check payload length
430 uint16_t payload_len = ntohs(header->length);
431 if (payload_len > SCP_MAX_PAYLOAD_SIZE)
432 {
433 return false;
434 }
435
436 // Check priority in flags
437 uint8_t priority = SCP_GET_PRIORITY(header->flags);
438 if (!scp_is_valid_priority(priority))
439 {
440 return false;
441 }
442
443 // Check for reserved flag bits
444 uint8_t reserved_flags = header->flags & ~(SCP_FLAG_AUTH_PRESENT |
445 SCP_FLAG_COMPRESSED |
446 SCP_FLAG_PRIORITY_MASK);
447 if (reserved_flags != 0)
448 {
449 return false;
450 }
451
452 return true;
453}

462uint16_t scp_get_max_frame_size(uint16_t payload_size, bool has_auth)
463{
464 uint16_t raw_size = sizeof(scp_frame_header_t) + payload_size + SCP_CRC16_SIZE;
465
466 if (has_auth)
467 {
468 raw_size += SCP_AUTH_TAG_SIZE;
469 }
470
471 // COBS worst-case overhead: 1 byte per 254 + 2 bytes
472 uint16_t cobs_overhead = (raw_size / 254) + 2;
473
474 return raw_size + cobs_overhead;
475}

542bool scp_constants_init(void)
543{
544 ESP_LOGI(TAG, "Initializing SCP constants");
545
546 // Validate critical protocol constants
547 static_assert(SCP_MAX_FRAME_SIZE >= 256, "Frame size too small");
548 static_assert(SCP_MAX_PAYLOAD_SIZE < SCP_MAX_FRAME_SIZE, "Payload size inconsistent");
549 static_assert(SCP_AUTH_TAG_SIZE == 16, "Auth tag size must be 16 bytes");
550 static_assert(SCP_CRC16_SIZE == 2, "CRC size must be 2 bytes");
551
552 // Validate session constants
553 static_assert(SCP_MAX_SESSIONS >= 1 && SCP_MAX_SESSIONS <= 8, "Invalid session count");
554 static_assert(SCP_SESSION_KEY_SIZE == 32, "Session key must be 32 bytes");
555
556 // Validate rate limiting constants
557 static_assert(SCP_RATE_SUSTAINED_CPS > 0, "Sustained rate must be positive");
558 static_assert(SCP_RATE_BURST_CPS >= SCP_RATE_SUSTAINED_CPS, "Burst rate must be >= sustained");
559 static_assert(SCP_SESSION_TIMEOUT > 0, "Session timeout must be positive");
560
561 ESP_LOGI(TAG, "SCP constants validation passed");
562 return true;
563}

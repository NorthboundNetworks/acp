180#define ACP_IS_SUCCESS(err) ((err) == ACP_OK)
181
188#define ACP_IS_ERROR(err) ((err) < ACP_OK)
189
196#define ACP_IS_RECOVERABLE(err) \
197 ((err) == ACP_ERR_NEED_MORE_DATA || \
198 (err) == ACP_ERR_BUFFER_TOO_SMALL || \
199 (err) == ACP_ERR_TIMEOUT || \
200 (err) == ACP_ERR_RESOURCE_BUSY)
201
208#define ACP_IS_SECURITY_ERROR(err) \
209 ((err) >= ACP_ERR_AUTH_BASE && (err) < ACP_ERR_PLATFORM_BASE)

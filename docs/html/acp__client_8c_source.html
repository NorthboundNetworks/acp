84static int demo_telemetry_frame(void)
85{
86 printf("\n=== Telemetry Frame Demo ===\n");
87
88 // Sample telemetry data
89 telemetry_data_t telemetry = {
90 .voltage_mv = 12500, // 12.5V
91 .current_ma = 2300, // 2.3A
92 .temperature_c = 45, // 45°C
93 .status_flags = 0x07 // Systems online
94 };
95
96 // Encode frame (no authentication for telemetry)
97 uint8_t encoded_buffer[ACP_MAX_FRAME_SIZE];
98 size_t encoded_len = sizeof(encoded_buffer);
99
100 acp_result_t result = acp_encode_frame(
101 ACP_FRAME_TYPE_TELEMETRY,
102 0, // No auth flags for telemetry
103 (uint8_t *)&telemetry,
104 sizeof(telemetry),
105 NULL, // No session (unauthenticated)
106 encoded_buffer,
107 &encoded_len);
108
109 if (result != ACP_OK)
110 {
111 printf("Encode failed: %d\n", result);
112 return -1;
113 }
114
115 printf("Encoded %zu bytes\n", encoded_len);
116
117 // Decode frame
118 acp_frame_t decoded_frame;
119 size_t consumed = 0;
120
121 result = acp_decode_frame(
122 encoded_buffer,
123 encoded_len,
124 &decoded_frame,
125 &consumed,
126 NULL // No session (unauthenticated)
127 );
128
129 if (result != ACP_OK)
130 {
131 printf("Decode failed: %d\n", result);
132 return -1;
133 }
134
135 printf("Decoded frame (%zu bytes consumed):\n", consumed);
136 print_frame_info(&decoded_frame);
137
138 // Verify payload integrity
139 if (decoded_frame.length == sizeof(telemetry) &&
140 memcmp(decoded_frame.payload, &telemetry, sizeof(telemetry)) == 0)
141 {
142 printf("✓ Telemetry payload verified\n");
143 }
144 else
145 {
146 printf("✗ Telemetry payload mismatch\n");
147 return -1;
148 }
149
150 return 0;
151}

156static int demo_authenticated_command(void)
157{
158 printf("\n=== Authenticated Command Demo ===\n");
159
160 // Initialize session with test key
161 acp_session_t session;
162 uint8_t test_key[] = "test_key_32_bytes_for_hmac_demo!";
163 uint64_t nonce = 0x1234567890ABCDEF;
164
165 acp_result_t result = acp_session_init(&session, 1, test_key, sizeof(test_key), nonce);
166 if (result != ACP_OK)
167 {
168 printf("Session init failed: %d\n", result);
169 return -1;
170 }
171
172 printf("Session initialized with key_id=1, nonce=0x%016llx\n",
173 (unsigned long long)nonce);
174
175 // Sample command data
176 command_data_t command = {
177 .command_id = 0x42, // Set mode command
178 .target_system = 1, // Target system 1
179 .parameter = 1000 // Parameter value
180 };
181
182 // Encode authenticated command frame
183 uint8_t encoded_buffer[ACP_MAX_FRAME_SIZE];
184 size_t encoded_len = sizeof(encoded_buffer);
185
186 result = acp_encode_frame(
187 ACP_FRAME_TYPE_COMMAND,
188 ACP_FLAG_AUTHENTICATED,
189 (uint8_t *)&command,
190 sizeof(command),
191 &session,
192 encoded_buffer,
193 &encoded_len);
194
195 if (result != ACP_OK)
196 {
197 printf("Authenticated encode failed: %d\n", result);
198 return -1;
199 }
200
201 printf("Encoded authenticated frame: %zu bytes\n", encoded_len);
202
203 // Create new session for decoding (simulating receiver)
204 acp_session_t rx_session;
205 result = acp_session_init(&rx_session, 1, test_key, sizeof(test_key), nonce);
206 if (result != ACP_OK)
207 {
208 printf("RX session init failed: %d\n", result);
209 return -1;
210 }
211
212 // Decode authenticated frame
213 acp_frame_t decoded_frame;
214 size_t consumed = 0;
215
216 result = acp_decode_frame(
217 encoded_buffer,
218 encoded_len,
219 &decoded_frame,
220 &consumed,
221 &rx_session);
222
223 if (result != ACP_OK)
224 {
225 printf("Authenticated decode failed: %d\n", result);
226 return -1;
227 }
228
229 printf("✓ Authentication verified\n");
230 print_frame_info(&decoded_frame);
231
232 // Test replay protection - try to decode same frame again
233 printf("\nTesting replay protection...\n");
234 result = acp_decode_frame(
235 encoded_buffer,
236 encoded_len,
237 &decoded_frame,
238 &consumed,
239 &rx_session);
240
241 if (result == ACP_ERR_REPLAY)
242 {
243 printf("✓ Replay protection working\n");
244 }
245 else
246 {
247 printf("✗ Replay protection failed: %d\n", result);
248 return -1;
249 }
250
251 return 0;
252}

58uint64_t acp_platform_get_time_ms(void) {
59 FILETIME ft;
60 ULARGE_INTEGER ull;
61
62 /* Get system time as FILETIME (100-nanosecond intervals since Jan 1, 1601) */
63 GetSystemTimeAsFileTime(&ft);
64
65 /* Convert to 64-bit value */
66 ull.LowPart = ft.dwLowDateTime;
67 ull.HighPart = ft.dwHighDateTime;
68
69 /* Convert to milliseconds since Unix epoch (Jan 1, 1970) */
70 /* FILETIME epoch is 11644473600 seconds before Unix epoch */
71 const uint64_t EPOCH_DIFFERENCE_MS = 11644473600000ULL;
72 return (ull.QuadPart / 10000ULL) - EPOCH_DIFFERENCE_MS;
73}

78uint64_t acp_platform_get_monotonic_us(void) {
79 static LARGE_INTEGER frequency = {0};
80 LARGE_INTEGER counter;
81
82 /* Initialize frequency on first call */
83 if (frequency.QuadPart == 0) {
84 if (!QueryPerformanceFrequency(&frequency)) {
85 /* Fallback to GetTickCount64 if high-res timer not available */
86 return GetTickCount64() * 1000ULL;
87 }
88 }
89
90 /* Get current counter value */
91 if (!QueryPerformanceCounter(&counter)) {
92 return GetTickCount64() * 1000ULL;
93 }
94
95 /* Convert to microseconds */
96 return (counter.QuadPart * 1000000ULL) / frequency.QuadPart;
97}

114 if (!mutex) {
115 return -1;

119 if (!cs) {
120 return -1;

132 if (mutex && mutex->handle) {
133 CRITICAL_SECTION *cs = (CRITICAL_SECTION *)mutex->handle;
134 DeleteCriticalSection(cs);

136 mutex->handle = NULL;

186void acp_platform_log(acp_log_level_t level, const char *file, int line,
187 const char *func, const char *fmt, ...) {
188 /* Get current timestamp */
189 SYSTEMTIME st;
190 GetLocalTime(&st);
191
192 /* Log level strings */
193 const char *level_str;
194 switch (level) {
195 case ACP_LOG_ERROR: level_str = "ERROR"; break;
196 case ACP_LOG_WARN: level_str = "WARN "; break;
197 case ACP_LOG_INFO: level_str = "INFO "; break;
198 case ACP_LOG_DEBUG: level_str = "DEBUG"; break;
199 default: level_str = "?????"; break;
200 }
201
202 /* Print timestamp and level */
203 fprintf(stderr, "[%04d-%02d-%02d %02d:%02d:%02d.%03d] [ACP][%s] ",
204 st.wYear, st.wMonth, st.wDay,
205 st.wHour, st.wMinute, st.wSecond, st.wMilliseconds,
206 level_str);
207
208 /* Print location info for errors and debug */
209 if (level == ACP_LOG_ERROR || level == ACP_LOG_DEBUG) {
210 const char *filename = strrchr(file, '\\');
211 if (!filename) filename = strrchr(file, '/');
212 if (!filename) filename = file;
213 else filename++;
214
215 fprintf(stderr, "%s:%d:%s() ", filename, line, func);
216 }
217
218 /* Print formatted message */
219 va_list args;
220 va_start(args, fmt);
221 vfprintf(stderr, fmt, args);
222 va_end(args);
223
224 fprintf(stderr, "\n");
225 fflush(stderr);
226}

239int acp_platform_keystore_init(void) {
240 /* Create keystore directory if it doesn't exist */
241 DWORD attrs = GetFileAttributesA(ACP_WIN32_KEYSTORE_DIR);
242 if (attrs == INVALID_FILE_ATTRIBUTES) {
243 if (!CreateDirectoryA(ACP_WIN32_KEYSTORE_DIR, NULL)) {
244 DWORD error = GetLastError();
245 if (error != ERROR_ALREADY_EXISTS) {
246 return -1;
247 }
248 }
249 } else if (!(attrs & FILE_ATTRIBUTE_DIRECTORY)) {
250 /* Path exists but is not a directory */
251 return -1;
252 }
253
254 return 0;
255}

260int acp_platform_keystore_load_key(uint32_t key_id, uint8_t *key_buffer, size_t buffer_size) {
261 if (!key_buffer || buffer_size < 32) {
262 return -1;
263 }
264
265 /* Open keystore file */
266 HANDLE hFile = CreateFileA(ACP_WIN32_KEYSTORE_FILE,
267 GENERIC_READ,
268 FILE_SHARE_READ,
269 NULL,
270 OPEN_EXISTING,
271 FILE_ATTRIBUTE_NORMAL,
272 NULL);
273
274 if (hFile == INVALID_HANDLE_VALUE) {
275 return -1;
276 }
277
278 /* Read keystore header to find key */
279 struct keystore_header {
280 uint32_t magic; /* 'ACPK' */
281 uint32_t version; /* Keystore format version */
282 uint32_t key_count; /* Number of keys */
283 uint32_t reserved; /* Reserved for future use */
284 } header;
285
286 DWORD bytes_read;
287 if (!ReadFile(hFile, &header, sizeof(header), &bytes_read, NULL) ||
288 bytes_read != sizeof(header) ||
289 header.magic != 0x4B504341) { /* 'ACPK' in little-endian */
290 CloseHandle(hFile);
291 return -1;
292 }
293
294 /* Search for key */
295 for (uint32_t i = 0; i < header.key_count; i++) {
296 struct key_entry {
297 uint32_t key_id;
298 uint32_t key_len;
299 uint8_t key_data[32];
300 } entry;
301
302 if (!ReadFile(hFile, &entry, sizeof(entry), &bytes_read, NULL) ||
303 bytes_read != sizeof(entry)) {
304 CloseHandle(hFile);
305 return -1;
306 }
307
308 if (entry.key_id == key_id && entry.key_len <= buffer_size) {
309 memcpy(key_buffer, entry.key_data, entry.key_len);
310 CloseHandle(hFile);
311 return (int)entry.key_len;
312 }
313 }
314
315 CloseHandle(hFile);
316 return -1; /* Key not found */
317}

322int acp_platform_keystore_store_key(uint32_t key_id, const uint8_t *key_data, size_t key_len) {
323 if (!key_data || key_len == 0 || key_len > 32) {
324 return -1;
325 }
326
327 /* For this implementation, we'll create a simple append-only keystore */
328 /* In production, you'd want to handle key updates and compaction */
329
330 HANDLE hFile = CreateFileA(ACP_WIN32_KEYSTORE_FILE,
331 GENERIC_WRITE,
332 0,
333 NULL,
334 OPEN_ALWAYS,
335 FILE_ATTRIBUTE_NORMAL,
336 NULL);
337
338 if (hFile == INVALID_HANDLE_VALUE) {
339 return -1;
340 }
341
342 /* Check if file is new */
343 DWORD file_size = GetFileSize(hFile, NULL);
344 if (file_size == 0) {
345 /* Write header for new keystore */
346 struct keystore_header header = {
347 .magic = 0x4B504341, /* 'ACPK' */
348 .version = 1,
349 .key_count = 1,
350 .reserved = 0
351 };
352
353 DWORD bytes_written;
354 if (!WriteFile(hFile, &header, sizeof(header), &bytes_written, NULL)) {
355 CloseHandle(hFile);
356 return -1;
357 }
358 } else {
359 /* Update key count in existing file */
360 SetFilePointer(hFile, 8, NULL, FILE_BEGIN); /* Offset of key_count */
361 uint32_t key_count;
362 DWORD bytes_read;
363 if (ReadFile(hFile, &key_count, sizeof(key_count), &bytes_read, NULL)) {
364 key_count++;
365 SetFilePointer(hFile, 8, NULL, FILE_BEGIN);
366 WriteFile(hFile, &key_count, sizeof(key_count), &bytes_read, NULL);
367 }
368
369 /* Seek to end for appending */
370 SetFilePointer(hFile, 0, NULL, FILE_END);
371 }
372
373 /* Write key entry */
374 struct key_entry {
375 uint32_t key_id;
376 uint32_t key_len;
377 uint8_t key_data[32];
378 } entry = {0};
379
380 entry.key_id = key_id;
381 entry.key_len = (uint32_t)key_len;
382 memcpy(entry.key_data, key_data, key_len);
383
384 DWORD bytes_written;
385 BOOL result = WriteFile(hFile, &entry, sizeof(entry), &bytes_written, NULL);
386
387 CloseHandle(hFile);
388 return result ? 0 : -1;
389}

404int acp_platform_keystore_list_keys(uint32_t *key_ids, size_t max_keys) {
405 if (!key_ids || max_keys == 0) {
406 return -1;
407 }
408
409 HANDLE hFile = CreateFileA(ACP_WIN32_KEYSTORE_FILE,
410 GENERIC_READ,
411 FILE_SHARE_READ,
412 NULL,
413 OPEN_EXISTING,
414 FILE_ATTRIBUTE_NORMAL,
415 NULL);
416
417 if (hFile == INVALID_HANDLE_VALUE) {
418 return 0; /* No keystore file = no keys */
419 }
420
421 /* Read header */
422 struct keystore_header {
423 uint32_t magic;
424 uint32_t version;
425 uint32_t key_count;
426 uint32_t reserved;
427 } header;
428
429 DWORD bytes_read;
430 if (!ReadFile(hFile, &header, sizeof(header), &bytes_read, NULL) ||
431 bytes_read != sizeof(header) ||
432 header.magic != 0x4B504341) {
433 CloseHandle(hFile);
434 return -1;
435 }
436
437 /* Read key IDs */
438 size_t keys_found = 0;
439 for (uint32_t i = 0; i < header.key_count && keys_found < max_keys; i++) {
440 uint32_t key_id;
441 if (!ReadFile(hFile, &key_id, sizeof(key_id), &bytes_read, NULL) ||
442 bytes_read != sizeof(key_id)) {
443 break;
444 }
445
446 key_ids[keys_found++] = key_id;
447
448 /* Skip rest of key entry */
449 SetFilePointer(hFile, sizeof(uint32_t) + 32, NULL, FILE_CURRENT);
450 }
451
452 CloseHandle(hFile);
453 return (int)keys_found;
454}

Implementation Details

Database Schema Extensions (Phase 1)

-- Task execution tracking
CREATE TABLE task_executions (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
task_id VARCHAR(36) NOT NULL REFERENCES research_tasks(id),
worker_type VARCHAR(50) NOT NULL,
worker_id VARCHAR(100),
status VARCHAR(20) NOT NULL DEFAULT 'queued',
progress_percent INTEGER DEFAULT 0,
current_step TEXT,
started_at TIMESTAMPTZ,
completed_at TIMESTAMPTZ,
error_message TEXT,
retry_count INTEGER DEFAULT 0,
result_data JSONB,
metadata JSONB DEFAULT '{}',
created_at TIMESTAMPTZ DEFAULT NOW(),
updated_at TIMESTAMPTZ DEFAULT NOW()
);
-- Worker process registry
CREATE TABLE worker_registry (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
worker_id VARCHAR(100) UNIQUE NOT NULL,
worker_type VARCHAR(50) NOT NULL,
status VARCHAR(20) NOT NULL DEFAULT 'active',
last_heartbeat TIMESTAMPTZ DEFAULT NOW(),
capabilities JSONB DEFAULT '{}',
performance_metrics JSONB DEFAULT '{}',
created_at TIMESTAMPTZ DEFAULT NOW()
);
-- Task results storage
CREATE TABLE task_results (
id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
task_id VARCHAR(36) NOT NULL REFERENCES research_tasks(id),
execution_id UUID NOT NULL REFERENCES task_executions(id),
result_type VARCHAR(50) NOT NULL,
content_type VARCHAR(100) NOT NULL,
file_path TEXT,
data JSONB,
metadata JSONB DEFAULT '{}',
checksum VARCHAR(64),
size_bytes BIGINT,
created_at TIMESTAMPTZ DEFAULT NOW()
);
-- Add indexes for performance
CREATE INDEX idx_task_executions_task_id ON task_executions(task_id);
CREATE INDEX idx_task_executions_worker_type ON task_executions(worker_type);
CREATE INDEX idx_task_executions_status ON task_executions(status);
CREATE INDEX idx_worker_registry_type ON worker_registry(worker_type);
CREATE INDEX idx_worker_registry_status ON worker_registry(status);
CREATE INDEX idx_task_results_task_id ON task_results(task_id);
CREATE INDEX idx_task_results_execution_id ON task_results(execution_id);

Worker Framework (Phase 3)

Create a base worker class that all specific workers inherit from:

from abc import ABC, abstractmethod
import asyncio
from typing import Dict, Any
class BaseWorker(ABC):
def __init__(self, worker_id: str, worker_type: str):
self.worker_id = worker_id
self.worker_type = worker_type
@abstractmethod
async def process_task(self, task_data: Dict[str, Any]) -> Dict[str, Any]:
"""Process a task and return results"""
pass
async def update_progress(self, percent: int, step: str):
"""Update task progress in database"""
pass
async def register_worker(self):
"""Register worker in worker_registry table"""
pass
async def heartbeat(self):
"""Send heartbeat to maintain worker registration"""
pass

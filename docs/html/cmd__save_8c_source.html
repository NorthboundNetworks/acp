29// Save all current configuration parameters to NVS
30esp_err_t cmd_save_persist_config(void)
31{
32 esp_err_t ret;
33 nvs_handle_t nvs_handle;
34
35 // Open NVS namespace
36 ret = nvs_open(CONFIG_NVS_NAMESPACE, NVS_READWRITE, &nvs_handle);
37 if (ret != ESP_OK)
38 {
39 ESP_LOGE(TAG, "Failed to open NVS namespace: %s", esp_err_to_name(ret));
40 return ret;
41 }
42
43 // Get parameter registry for iteration
44 const config_param_def_t *params = NULL;
45 size_t param_count = 0;
46 ret = config_get_parameter_registry(&params, &param_count);
47 if (ret != ESP_OK)
48 {
49 ESP_LOGE(TAG, "Failed to get parameter registry");
50 nvs_close(nvs_handle);
51 return ret;
52 }
53
54 // Persist each parameter using current values
55 for (size_t i = 0; i < param_count; i++)
56 {
57 const config_param_def_t *param = &params[i];
58
59 // Skip read-only parameters - they shouldn't be persisted as user config
60 if (param->read_only)
61 {
62 continue;
63 }
64
65 // Get current value as string
66 char value_str[256];
67 ret = config_get_param_value_string(param->name, value_str, sizeof(value_str));
68 if (ret != ESP_OK)
69 {
70 ESP_LOGE(TAG, "Failed to get value for parameter %s", param->name);
71 nvs_close(nvs_handle);
72 return ret;
73 }
74
75 // Store based on parameter type for efficient storage
76 switch (param->type)
77 {
78 case PARAM_TYPE_INT:
79 {
80 int32_t int_val = (int32_t)strtol(value_str, NULL, 10);
81 ret = nvs_set_i32(nvs_handle, param->name, int_val);
82 break;
83 }
84 case PARAM_TYPE_FLOAT:
85 {
86 float float_val = strtof(value_str, NULL);
87 ret = nvs_set_blob(nvs_handle, param->name, &float_val, sizeof(float));
88 break;
89 }
90 case PARAM_TYPE_BOOL:
91 {
92 uint8_t bool_val = (strcmp(value_str, "true") == 0) ? 1 : 0;
93 ret = nvs_set_u8(nvs_handle, param->name, bool_val);
94 break;
95 }
96 case PARAM_TYPE_STRING:
97 {
98 ret = nvs_set_str(nvs_handle, param->name, value_str);
99 break;
100 }
101 default:
102 ESP_LOGE(TAG, "Unknown parameter type for %s", param->name);
103 ret = ESP_ERR_INVALID_ARG;
104 break;
105 }
106
107 if (ret != ESP_OK)
108 {
109 ESP_LOGE(TAG, "Failed to set NVS value for %s: %s",
110 param->name, esp_err_to_name(ret));
111 nvs_close(nvs_handle);
112 return ret;
113 }
114 }
115
116 // Commit all changes atomically
117 ret = nvs_commit(nvs_handle);
118 if (ret != ESP_OK)
119 {
120 ESP_LOGE(TAG, "Failed to commit NVS changes: %s", esp_err_to_name(ret));
121 nvs_close(nvs_handle);
122 return ret;
123 }
124
125 // Close NVS handle
126 nvs_close(nvs_handle);
127
128 ESP_LOGI(TAG, "Successfully persisted %zu configuration parameters", param_count);
129 return ESP_OK;
130}

132// CLI save command handler
133esp_err_t cmd_save_handler(int argc, char **argv)
134{
135 // Save command takes no arguments
136 if (argc > 1)
137 {
138 cli_transport_printf("ERROR: save command takes no arguments\n");
139 return ESP_ERR_INVALID_ARG;
140 }
141
142 // Persist current configuration
143 esp_err_t ret = cmd_save_persist_config();
144
145 if (ret != ESP_OK)
146 {
147 // Handle specific error cases per CLI contracts
148 if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NO_MEM)
149 {
150 cli_transport_printf("ERROR: 507 insufficient storage\n");
151 return ESP_ERR_NO_MEM;
152 }
153 else
154 {
155 cli_transport_printf("ERROR: 500 persistence error\n");
156 return ESP_FAIL;
157 }
158 }
159
160 // Success output per CLI contracts
161 cli_transport_printf("SAVED\n");
162 return ESP_OK;
163}

78static size_t mock_serial_write(mock_serial_t *serial, const uint8_t *data, size_t len)
79{
80 size_t written = 0;
81
82 while (written < len && serial->available < serial->buffer_size)
83 {
84 serial->buffer[serial->write_pos] = data[written];
85 serial->write_pos = (serial->write_pos + 1) % serial->buffer_size;
86 serial->available++;
87 written++;
88 }
89
90 return written;
91}

96static size_t mock_serial_read(mock_serial_t *serial, uint8_t *data, size_t len)
97{
98 size_t read_count = 0;
99
100 while (read_count < len && serial->available > 0)
101 {
102 data[read_count] = serial->buffer[serial->read_pos];
103 serial->read_pos = (serial->read_pos + 1) % serial->buffer_size;
104 serial->available--;
105 read_count++;
106 }
107
108 return read_count;
109}

199static int receive_frames(mock_serial_t *serial)
200{
201 printf("\nReceiving frames...\n");
202
203 uint8_t receive_buffer[512];
204 size_t buffer_pos = 0;
205 int frame_count = 0;
206
207 // Stream processing loop
208 while (mock_serial_available(serial) > 0)
209 {
210 // Read more data from serial port
211 size_t available_space = sizeof(receive_buffer) - buffer_pos;
212 if (available_space == 0)
213 {
214 printf("Receive buffer full, resetting\n");
215 buffer_pos = 0;
216 available_space = sizeof(receive_buffer);
217 }
218
219 size_t bytes_read = mock_serial_read(serial,
220 &receive_buffer[buffer_pos],
221 available_space);
222
223 if (bytes_read == 0)
224 {
225 break; // No more data
226 }
227
228 buffer_pos += bytes_read;
229
230 printf("Read %zu bytes, buffer now has %zu bytes\n", bytes_read, buffer_pos);
231
232 // Try to decode frames from current buffer
233 size_t processed = 0;
234
235 while (processed < buffer_pos)
236 {
237 acp_frame_t frame;
238 size_t consumed = 0;
239
240 acp_result_t result = acp_decode_frame(
241 &receive_buffer[processed],
242 buffer_pos - processed,
243 &frame,
244 &consumed,
245 NULL // No authentication for this test
246 );
247
248 if (result == ACP_OK)
249 {
250 // Successfully decoded a frame
251 frame_count++;
252 printf("\n✓ Frame %d decoded (%zu bytes consumed):\n",
253 frame_count, consumed);
254 printf(" Type: %u, Flags: 0x%02x, Length: %u\n",
255 frame.type, frame.flags, frame.length);
256 printf(" Payload: ");
257 for (size_t i = 0; i < frame.length; i++)
258 {
259 printf("%02x ", frame.payload[i]);
260 }
261 printf("\n");
262
263 processed += consumed;
264 }
265 else if (result == ACP_ERR_NEED_MORE_DATA)
266 {
267 // Need more data to complete frame
268 printf("Need more data (have %zu bytes)\n", buffer_pos - processed);
269 break;
270 }
271 else
272 {
273 // Frame decode error - skip one byte and try again
274 printf("✗ Frame decode error %d, skipping byte 0x%02x\n",
275 result, receive_buffer[processed]);
276 processed++;
277 }
278 }
279
280 // Move remaining data to start of buffer
281 if (processed > 0 && processed < buffer_pos)
282 {
283 memmove(receive_buffer, &receive_buffer[processed], buffer_pos - processed);
284 buffer_pos -= processed;
285 }
286 else if (processed >= buffer_pos)
287 {
288 buffer_pos = 0;
289 }
290 }
291
292 printf("\nTotal frames received: %d\n", frame_count);
293
294 if (buffer_pos > 0)
295 {
296 printf("Remaining buffer data (%zu bytes): ", buffer_pos);
297 for (size_t i = 0; i < buffer_pos && i < 16; i++)
298 {
299 printf("%02x ", receive_buffer[i]);
300 }
301 printf("\n");
302 }
303
304 return frame_count;
305}

310static int test_partial_reads(mock_serial_t *serial)
311{
312 printf("\n=== Partial Read Test ===\n");
313
314 // Send a test frame
315 uint8_t test_payload[] = "Hello ACP World!";
316 uint8_t frame_buffer[128];
317 size_t frame_len = sizeof(frame_buffer);
318
319 acp_result_t result = acp_encode_frame(
320 ACP_FRAME_TYPE_SYSTEM,
321 0,
322 test_payload,
323 sizeof(test_payload) - 1, // Exclude null terminator
324 NULL,
325 frame_buffer,
326 &frame_len);
327
328 if (result != ACP_OK)
329 {
330 printf("Test frame encode failed: %d\n", result);
331 return -1;
332 }
333
334 printf("Sending test frame in chunks...\n");
335
336 // Send frame in small chunks to test partial read handling
337 for (size_t i = 0; i < frame_len; i += 3)
338 {
339 size_t chunk_size = (i + 3 < frame_len) ? 3 : frame_len - i;
340 mock_serial_write(serial, &frame_buffer[i], chunk_size);
341 printf("Sent chunk %zu: %zu bytes\n", i / 3 + 1, chunk_size);
342
343 // Try to decode with each partial chunk
344 uint8_t read_buffer[128];
345 size_t total_read = 0;
346
347 while (mock_serial_available(serial) > 0 && total_read < sizeof(read_buffer))
348 {
349 size_t read_this_time = mock_serial_read(serial,
350 &read_buffer[total_read], 1);
351 total_read += read_this_time;
352 }
353
354 acp_frame_t frame;
355 size_t consumed = 0;
356 result = acp_decode_frame(read_buffer, total_read, &frame, &consumed, NULL);
357
358 if (result == ACP_OK)
359 {
360 printf("✓ Complete frame decoded on chunk %zu\n", i / 3 + 1);
361 break;
362 }
363 else if (result == ACP_ERR_NEED_MORE_DATA)
364 {
365 printf("Partial frame, need more data\n");
366 // Put data back for next iteration
367 mock_serial_write(serial, read_buffer, total_read);
368 }
369 else
370 {
371 printf("Decode error: %d\n", result);
372 }
373 }
374
375 return 0;
376}

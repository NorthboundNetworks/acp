100static void tlm_update_jitter_stats(int32_t jitter_ms)
101{
102 if (jitter_ms > s_gateway.tlm.max_jitter_ms)
103 s_gateway.tlm.max_jitter_ms = jitter_ms;
104 if (jitter_ms < s_gateway.tlm.min_jitter_ms)
105 s_gateway.tlm.min_jitter_ms = jitter_ms;
106
107 s_gateway.tlm.jitter_samples[s_gateway.tlm.jitter_idx] = (uint32_t)(jitter_ms < 0 ? -jitter_ms : jitter_ms);
108 s_gateway.tlm.jitter_idx = (s_gateway.tlm.jitter_idx + 1) % 16;
109 uint32_t sum = 0;
110 for (int i = 0; i < 16; i++)
111 sum += s_gateway.tlm.jitter_samples[i];
112 s_gateway.tlm.avg_jitter_ms = (int32_t)(sum / 16);
113}

257static uint8_t send_status_to_session(uint32_t session_id, const scp_tlm_status_t *status)
258{
259 // Build frame header per current protocol
260 scp_frame_header_t hdr = {
261 .sync = {SCP_SYNC_BYTE1, SCP_SYNC_BYTE2},
262 .version = SCP_VERSION_1_1,
263 .msg_type = SCP_TLM_STATUS,
264 .seq_num = 0,
265 .flags = (uint8_t)((SCP_PRIORITY_NORMAL << SCP_FLAG_PRIORITY_SHIFT)),
266 .length = htons((uint16_t)sizeof(*status))};
267
268 uint16_t out_len = 0;
269 bool ok = scp_framer_encode(&hdr,
270 (const uint8_t *)status,
271 (uint16_t)sizeof(*status),
272 NULL, // no auth tag yet
273 s_gateway.tx_buffer,
274 (uint16_t)sizeof(s_gateway.tx_buffer),
275 &out_len);
276 if (!ok)
277 {
278 ESP_LOGW(TAG, "Failed to encode TLM_STATUS for session %lu", session_id);
279 return 0;
280 }
281
282 // Minimal transport: write binary frame to stdout (same console as logs)
283 // This allows a host tool to read frames over the USB serial console.
284 size_t wrote = fwrite(s_gateway.tx_buffer, 1, out_len, stdout);
285 fflush(stdout);
286 if (wrote != out_len)
287 {
288 ESP_LOGW(TAG, "Console write short: %u/%u", (unsigned)wrote, out_len);
289 }
290
291 ESP_LOGD(TAG, "Sent TLM_STATUS frame (len=%u) for session %lu", out_len, session_id);
292 s_gateway.stats.messages_sent++; // count bytes/messages generically
293 s_gateway.stats.tlm_status_sent++;
294 return 1;
295}

297uint8_t gcs_gateway_send_tlm_status_targeted(uint32_t target_session_id)
298{
299 scp_tlm_status_t st;
300 tlm_populate_status(&st);
301
302 scp_session_t list[SCP_MAX_SESSIONS];
303 uint8_t n = scp_session_get_list(list, SCP_MAX_SESSIONS);
304 uint8_t sent = 0;
305 uint32_t now = gw_now_ms();
306
307 for (uint8_t i = 0; i < n; i++)
308 {
309 uint32_t sid = list[i].session_id;
310 if (target_session_id != 0 && sid != target_session_id)
311 continue;
312
313 // Apply per-session config
314 uint8_t eff_rate = s_gateway.tlm.session[i].rate_hz ? s_gateway.tlm.session[i].rate_hz : s_gateway.tlm.rate_hz;
315 if (!s_gateway.tlm.session[i].configured)
316 {
317 // default enabled
318 s_gateway.tlm.session[i].enabled = true;
319 s_gateway.tlm.session[i].configured = true;
320 s_gateway.tlm.session[i].session_id = sid;
321 }
322 if (!s_gateway.tlm.session[i].enabled)
323 continue;
324
325 uint32_t interval = (eff_rate > 0) ? (1000 / eff_rate) : s_gateway.tlm.target_interval_ms;
326 if (now - s_gateway.tlm.session[i].last_send_ms < interval)
327 continue;
328
329 sent += send_status_to_session(sid, &st);
330 s_gateway.tlm.session[i].last_send_ms = now;
331 s_gateway.tlm.session[i].messages_sent++;
332 }
333
334 // Timing accounting (global)
335 if (sent > 0)
336 {
337 int32_t elapsed = (int32_t)(now - s_gateway.tlm.last_status_send_ms);
338 if (s_gateway.tlm.last_status_send_ms != 0)
339 {
340 int32_t jitter = elapsed - (int32_t)s_gateway.tlm.target_interval_ms;
341 if (jitter < -50)
342 {
343 s_gateway.tlm.rate_limit_drops++;
344 }
345 else
346 {
347 s_gateway.tlm.total_transmissions++;
348 if (jitter >= -50 && jitter <= 50)
349 s_gateway.tlm.on_time_transmissions++;
350 if (jitter > 50)
351 s_gateway.tlm.late_transmissions++;
352 tlm_update_jitter_stats(jitter);
353 }
354 }
355 s_gateway.tlm.last_status_send_ms = now;
356 }
357
358 return sent;
359}

374bool gcs_gateway_send_audit(uint8_t event_type,
375 uint32_t source_session,
376 uint8_t result_code,
377 const uint8_t *details)
378{
379 scp_session_t list[SCP_MAX_SESSIONS];
380 uint8_t n = scp_session_get_list(list, SCP_MAX_SESSIONS);
381 if (n == 0)
382 return false;
383
384 scp_tlm_audit_t audit = {0};
385 audit.event_type = event_type;
386 audit.source_session = source_session;
387 audit.operator_role = 0; // Unknown here (could be filled per-session)
388 audit.result_code = result_code;
389 audit.timestamp_ms = gw_now_ms();
390 if (details)
391 {
392 memcpy(audit.details, details, sizeof(audit.details));
393 }
394
395 uint8_t sent = 0;
396 for (uint8_t i = 0; i < n; i++)
397 {
398 scp_frame_header_t hdr = {
399 .sync = {SCP_SYNC_BYTE1, SCP_SYNC_BYTE2},
400 .version = SCP_VERSION_1_1,
401 .msg_type = SCP_TLM_AUDIT,
402 .seq_num = 0,
403 .flags = (uint8_t)((SCP_PRIORITY_NORMAL << SCP_FLAG_PRIORITY_SHIFT)),
404 .length = htons((uint16_t)sizeof(audit))};
405
406 uint16_t out_len = 0;
407 bool ok = scp_framer_encode(&hdr,
408 (const uint8_t *)&audit,
409 (uint16_t)sizeof(audit),
410 NULL,
411 s_gateway.tx_buffer,
412 (uint16_t)sizeof(s_gateway.tx_buffer),
413 &out_len);
414 if (ok)
415 {
416 size_t wrote = fwrite(s_gateway.tx_buffer, 1, out_len, stdout);
417 fflush(stdout);
418 if (wrote != out_len)
419 {
420 ESP_LOGW(TAG, "Console write short (audit): %u/%u", (unsigned)wrote, out_len);
421 }
422 s_gateway.stats.messages_sent++;
423 s_gateway.stats.audit_events_sent++;
424 sent++;
425 }
426 }
427
428 return (sent > 0);
429}

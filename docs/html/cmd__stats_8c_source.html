114bool cmd_stats_validate_performance(void)
115{
116 cpu_stats_t cpu_stats;

118
119 rt_metrics_get_cpu_stats(&cpu_stats);
120 rt_metrics_get_loop_stats(&loop_stats);
121
122 // Constitution requirements
123 bool cpu_ok = cpu_stats.within_budget; // ≤70% per Constitution
124 bool timing_ok = loop_stats.within_tolerance; // ±5% per Constitution
125
126 ESP_LOGI(TAG, "Performance validation: CPU=%s (%d%%), Timing=%s (%dHz)",
127 cpu_ok ? "PASS" : "FAIL", cpu_stats.total_cpu_percent,
128 timing_ok ? "PASS" : "FAIL", loop_stats.current_hz);

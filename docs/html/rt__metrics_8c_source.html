70void rt_metrics_loop_end(uint16_t target_hz)
71{
72 if (g_loop_start_time_us == 0)
73 {
74 return; // No start time recorded
75 }
76
77 uint32_t loop_time_us = get_time_us() - g_loop_start_time_us;
78 g_loop_start_time_us = 0;
79
80 // Update loop statistics
81 g_loop_stats.target_hz = target_hz;
82
83 // Convert loop time to frequency (with overflow protection)
84 if (loop_time_us > 0 && loop_time_us < 1000000)
85 {
86 g_loop_stats.current_hz = 1000000 / loop_time_us;
87 }
88 else if (loop_time_us >= 1000000)
89 {
90 g_loop_stats.current_hz = 1; // Very slow loop
91 }
92
93 // Add to rolling window for statistical analysis
94 g_loop_times_us[g_loop_sample_index] = loop_time_us;
95 g_loop_sample_index = (g_loop_sample_index + 1) % SAMPLE_WINDOW_SIZE;
96 if (g_loop_sample_index == 0)
97 {
98 g_loop_samples_full = true;
99 }
100
101 // Calculate comprehensive statistics from window
102 uint8_t sample_count = g_loop_samples_full ? SAMPLE_WINDOW_SIZE : g_loop_sample_index;
103 if (sample_count > 10) // Need reasonable sample size for p95
104 {
105 uint32_t sum = 0, min = UINT32_MAX, max = 0;
106
107 // Create sorted copy for percentile calculation
108 uint32_t sorted_times[SAMPLE_WINDOW_SIZE];
109 for (uint8_t i = 0; i < sample_count; i++)
110 {
111 uint32_t time = g_loop_times_us[i];
112 sorted_times[i] = time;
113 sum += time;
114 if (time < min)
115 min = time;
116 if (time > max)
117 max = time;
118 }
119
120 // Simple bubble sort for percentile calculation (small array, acceptable)
121 for (uint8_t i = 0; i < sample_count - 1; i++)
122 {
123 for (uint8_t j = 0; j < sample_count - i - 1; j++)
124 {
125 if (sorted_times[j] > sorted_times[j + 1])
126 {
127 uint32_t temp = sorted_times[j];
128 sorted_times[j] = sorted_times[j + 1];
129 sorted_times[j + 1] = temp;
130 }
131 }
132 }
133
134 // Calculate statistics
135 g_loop_stats.avg_loop_time_us = sum / sample_count;
136 g_loop_stats.min_loop_time_us = min;
137 g_loop_stats.max_loop_time_us = max;
138
139 // Calculate p95 (95th percentile) for Constitution compliance
140 uint8_t p95_index = (sample_count * 95) / 100;
141 if (p95_index >= sample_count)
142 p95_index = sample_count - 1;
143 uint32_t p95_time_us = sorted_times[p95_index];
144
145 // Convert timing to frequencies (with bounds checking)
146 g_loop_stats.min_hz = (max > 0 && max < 1000000) ? 1000000 / max : 0;
147 g_loop_stats.max_hz = (min > 0) ? 1000000 / min : 0;
148
149 // Check tolerance (±5% per Constitution)
150 uint16_t tolerance_hz = (target_hz * LOOP_TOLERANCE_PERCENT) / 100;
151 uint16_t min_acceptable = (target_hz > tolerance_hz) ? target_hz - tolerance_hz : 1;
152 uint16_t max_acceptable = target_hz + tolerance_hz;
153
154 // Current frequency must be within tolerance
155 g_loop_stats.within_tolerance = (g_loop_stats.current_hz >= min_acceptable) &&
156 (g_loop_stats.current_hz <= max_acceptable);
157
158 // Additional check: p95 timing must also be reasonable
159 uint32_t target_period_us = 1000000 / target_hz;
160 uint32_t max_acceptable_time_us = target_period_us + (target_period_us * LOOP_TOLERANCE_PERCENT / 100);
161 bool p95_acceptable = (p95_time_us <= max_acceptable_time_us);
162
163 g_loop_stats.within_tolerance = g_loop_stats.within_tolerance && p95_acceptable;
164
165 // Store p95 for stats reporting (convert to milliseconds for readability)
166 g_loop_stats.p95_time_ms = (float)p95_time_us / 1000.0f;
167
168 // Periodic logging for debugging (rate limited)
169 static uint32_t last_log = 0;
170 uint32_t now = get_time_ms();
171 if (now - last_log > 10000) // Every 10 seconds
172 {
173 ESP_LOGD("rt_metrics", "Loop stats: target=%dHz current=%dHz avg=%.2fms p95=%.2fms tolerance=%s",
174 target_hz, g_loop_stats.current_hz,
175 g_loop_stats.avg_loop_time_us / 1000.0f,
176 g_loop_stats.p95_time_ms,
177 g_loop_stats.within_tolerance ? "OK" : "FAIL");
178 last_log = now;
179 }
180 }
181
182 // Update legacy interface
183 g_loop_rate = g_loop_stats.current_hz;
184}

199void rt_metrics_update_cpu_usage(void)
200{
201 uint32_t current_ms = get_time_ms();
202
203 // Sample CPU usage periodically
204 if (current_ms - g_last_cpu_sample_ms >= CPU_SAMPLE_PERIOD_MS)
205 {
206 // Use FreeRTOS runtime statistics with improved delta calculations
207 // Following reference implementation pattern from Espfc
208
209#if (CONFIG_FREERTOS_USE_TRACE_FACILITY) && (CONFIG_FREERTOS_GENERATE_RUN_TIME_STATS)
210 UBaseType_t task_count = uxTaskGetNumberOfTasks();
211 if (task_count > 0)
212 {
213 // Allocate buffer for all tasks
214 TaskStatus_t *task_array = (TaskStatus_t *)pvPortMalloc(task_count * sizeof(TaskStatus_t));
215 uint32_t total_runtime = 0;
216 uint32_t idle_runtime_core0 = 0;
217 uint32_t idle_runtime_core1 = 0;
218
219 if (task_array != NULL)
220 {
221 // Get handles for idle tasks on each core
222 TaskHandle_t idle0 = xTaskGetIdleTaskHandleForCore(0);
223 TaskHandle_t idle1 = xTaskGetIdleTaskHandleForCore(1);
224
225 // Get system state with total runtime counter
226 UBaseType_t filled = uxTaskGetSystemState(task_array, task_count, &total_runtime);
227
228 if (filled == 0 || total_runtime == 0)
229 {
230 // Prime runtime stats system by calling formatting function
231#if (CONFIG_FREERTOS_USE_STATS_FORMATTING_FUNCTIONS)
232 char stats_buffer[512];
233 vTaskGetRunTimeStats(stats_buffer);
234#endif
235 // Retry getting system state
236 filled = uxTaskGetSystemState(task_array, task_count, &total_runtime);
237 }
238
239 // Extract idle runtime for each core by matching task handles
240 for (UBaseType_t i = 0; i < filled; ++i)
241 {
242 if (task_array[i].xHandle == idle0)
243 {
244 idle_runtime_core0 = task_array[i].ulRunTimeCounter;
245 }
246 else if (task_array[i].xHandle == idle1)
247 {
248 idle_runtime_core1 = task_array[i].ulRunTimeCounter;
249 }
250 }
251
252 uint32_t total_idle = idle_runtime_core0 + idle_runtime_core1;
253
254 // Calculate CPU usage using deltas to avoid overflow and get accurate measurements
255 if (total_runtime > 0)
256 {
257 if (g_prev_total_runtime == 0)
258 {
259 // Initialize previous counters on first valid measurement
260 g_prev_total_runtime = total_runtime;
261 g_prev_idle_runtime = total_idle;
262 ESP_LOGI("rt_metrics", "CPU measurement initialized (total=%lu, idle=%lu)",
263 total_runtime, total_idle);
264 }
265 else if (total_runtime > g_prev_total_runtime)
266 {
267 // Calculate deltas and CPU utilization
268 // Note: On dual-core system, we need to account for 2 idle tasks
269 uint32_t delta_total = total_runtime - g_prev_total_runtime;
270 uint32_t delta_idle = (total_idle > g_prev_idle_runtime) ? (total_idle - g_prev_idle_runtime) : 0;
271
272 // On dual-core, total possible CPU time is 2x delta_total
273 // (one core can be 100% busy while other is 100% idle)
274 uint32_t max_possible_time = delta_total * 2; // Two cores
275 uint32_t delta_busy = (max_possible_time > delta_idle) ? (max_possible_time - delta_idle) : 0;
276
277 if (max_possible_time > 0)
278 {
279 uint8_t measured_cpu = (uint8_t)((delta_busy * 100UL) / max_possible_time);
280
281 // Clamp CPU percentage to sane limits
282 if (measured_cpu > 100)
283 {
284 measured_cpu = 100;
285 }
286
287 g_cpu_stats.total_cpu_percent = measured_cpu;
288 // For now, flight CPU equals total CPU - future enhancement can track per-task
289 g_cpu_stats.flight_cpu_percent = measured_cpu;
290
291 // Debug output for troubleshooting (rate limited)
292 static uint32_t last_debug = 0;
293 if (current_ms - last_debug > 2000)
294 { // Every 2 seconds
295 ESP_LOGI("rt_metrics", "CPU: %d%% (delta_total=%lu, delta_idle=%lu, max_time=%lu, delta_busy=%lu)",
296 measured_cpu, delta_total, delta_idle, max_possible_time, delta_busy);
297 last_debug = current_ms;
298 }
299 }
300
301 // Update previous counters for next measurement
302 g_prev_total_runtime = total_runtime;
303 g_prev_idle_runtime = total_idle;
304 }
305 else
306 {
307 // Runtime counter wrapped or reset - reinitialize
308 ESP_LOGW("rt_metrics", "Runtime counter reset detected, reinitializing");
309 g_prev_total_runtime = total_runtime;
310 g_prev_idle_runtime = total_idle;
311 }
312 }
313 else
314 {
315 // Runtime stats not working properly
316 static uint32_t last_warning = 0;
317 if (current_ms - last_warning > 5000)
318 { // Warn every 5 seconds
319 ESP_LOGI("rt_metrics", "FreeRTOS runtime stats total=0 - check configuration");
320 last_warning = current_ms;
321 }
322 }
323
324 vPortFree(task_array);
325 }
326 else
327 {
328 ESP_LOGW("rt_metrics", "Failed to allocate memory for task array");
329 }
330 }
331#else
332 // FreeRTOS runtime stats not enabled - cannot measure real CPU usage
333 static uint32_t last_config_warning = 0;
334 if (current_ms - last_config_warning > 30000)
335 { // Warn every 30 seconds
336 ESP_LOGW("rt_metrics", "FreeRTOS runtime stats disabled - CPU measurement unavailable");
337 last_config_warning = current_ms;
338 }
339#endif
340
341 // Update derived state
342 g_cpu_stats.within_budget = (g_cpu_stats.total_cpu_percent <= CPU_BUDGET_MAX_PERCENT);
343
344 // Update legacy interface
345 g_cpu = g_cpu_stats.total_cpu_percent;
346
347 g_last_cpu_sample_ms = current_ms;
348 }
349}

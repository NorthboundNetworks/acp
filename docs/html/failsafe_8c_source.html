29void failsafe_update(uint32_t timestamp_ms) {
30 scout_mode_t current_mode = mode_manager_get_mode();
31
32 switch (g_state) {
33 case FAILSAFE_STATE_NORMAL:
34 // Check for link loss while in flight
35 if (current_mode == SCOUT_MODE_FLIGHT && !g_link_ok && g_was_link_ok) {
36 g_link_loss_start_ms = timestamp_ms;
37 g_state = FAILSAFE_STATE_LINK_LOSS_DETECTED;
38 strcpy(g_fault_reason, "Ground link lost");
39 }
40 break;
41
42 case FAILSAFE_STATE_LINK_LOSS_DETECTED:
43 // Check if link has been lost for the threshold time
44 if (timestamp_ms - g_link_loss_start_ms >= LINK_LOSS_TIMEOUT_MS) {
45 g_state = FAILSAFE_STATE_DISARMING;
46 g_disarm_start_ms = timestamp_ms;
47 g_disarm_requested = true;
48 } else if (g_link_ok) {
49 // Link recovered before timeout
50 g_state = FAILSAFE_STATE_NORMAL;
51 strcpy(g_fault_reason, "Link recovered");
52 }
53 break;
54
55 case FAILSAFE_STATE_DISARMING:
56 // T028: Implement disarm on link-loss â‰¤1s
57 // Force immediate disarm due to link loss
58 if (current_mode == SCOUT_MODE_FLIGHT) {
59 // Force transition to CONFIG mode (effective disarm)
60 mode_manager_set_mode(SCOUT_MODE_CONFIG);
61
62 // Update safety gates to reflect disarmed state
63 safety_manager_update_link_status(false);
64 safety_manager_report_sensor_fault("LINK", "Disarmed due to link loss timeout");
65
66 g_state = FAILSAFE_STATE_DISARMED;
67 strcpy(g_fault_reason, "Disarmed due to link loss within 1s");
68 } else if (timestamp_ms - g_disarm_start_ms >= DISARM_MAX_TIME_MS) {
69 // Fallback timeout if mode change didn't complete
70 g_state = FAILSAFE_STATE_DISARMED;
71 strcpy(g_fault_reason, "Disarm timeout - manual recovery required");
72 }
73 break;
74
75 case FAILSAFE_STATE_DISARMED:
76 // Stay disarmed until manual recovery
77 if (g_link_ok && current_mode == SCOUT_MODE_CONFIG) {
78 // Allow recovery when back in CONFIG mode with good link
79 g_state = FAILSAFE_STATE_NORMAL;
80 g_disarm_requested = false;
81 strcpy(g_fault_reason, "System recovered");
82 }
83 break;
84 }
85}

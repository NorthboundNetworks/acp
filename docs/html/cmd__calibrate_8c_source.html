58static void cal_list_handler(int argc, char **argv)
59{
60 if (argc == 1)
61 {
62 // Show calibration status for all sensors
63 cli_transport_printf("Calibration Status:\n");
64 cli_transport_printf("==================\n\n");
65
66 // IMU status
67 mpu6050_t *imu = mpu6050_get_instance();
68 if (imu != NULL)

70 sensor_health_t health = mpu6050_get_health(imu);
71 const char *status_str = (health == SENSOR_HEALTH_OK) ? "CALIBRATED" : (health == SENSOR_HEALTH_CAL_REQ) ? "REQUIRED"
72 : "FAULT";
73 cli_transport_printf("IMU (MPU-6050): %s\n", status_str);
74 }
75 else
76 {
77 cli_transport_printf("IMU (MPU-6050): UNAVAILABLE\n");
78 }
79
80 // Magnetometer status
81 hmc5883l_t *mag = hmc5883l_get_instance();
82 if (mag != NULL)
83 {
84 sensor_health_t health = hmc5883l_get_health(mag);
85 const char *status_str = (health == SENSOR_HEALTH_OK) ? "CALIBRATED" : (health == SENSOR_HEALTH_CAL_REQ) ? "REQUIRED"
86 : "FAULT";
87 cli_transport_printf("MAG (HMC5883L): %s\n", status_str);
88 }
89 else
90 {
91 cli_transport_printf("MAG (HMC5883L): UNAVAILABLE\n");
92 }
93
94 cli_transport_printf("\nAvailable calibration commands:\n");
95 cli_transport_printf(" cal imu - Calibrate IMU (level surface required)\n");
96 cli_transport_printf(" cal mag - Calibrate magnetometer (3D rotation required)\n");
97 cli_transport_printf("\nNote: Calibration only allowed when disarmed and in CONFIG mode.\n");
98 }
99 else if (argc == 2)
100 {
101 // Handle specific sensor calibration
102 const char *sensor = argv[1];
103
104 if (strcmp(sensor, "imu") == 0)
105 {
106 cal_imu_handler(argc, argv);
107 }
108 else if (strcmp(sensor, "mag") == 0)
109 {
110 cal_mag_handler(argc, argv);
111 }
112 else
113 {
114 cli_transport_printf("ERROR: Unknown sensor '%s'. Available: imu, mag\n", sensor);
115 }
116 }
117}

119static void cal_imu_handler(int argc, char **argv)
120{
121 if (g_cal_state.in_progress)
122 {
123 cli_transport_printf("ERROR: Calibration already in progress for %s\n",
124 (g_cal_state.sensor == SENSOR_TYPE_IMU) ? "IMU" : "another sensor");
125 return;
126 }
127
128 if (!check_calibration_preconditions("IMU"))
129 {
130 return;
131 }
132
133 mpu6050_t *imu = mpu6050_get_instance();
134 if (imu == NULL)
135 {
136 cli_transport_printf("ERROR: IMU sensor not available\n");
137 return;
138 }
139
140 cli_transport_printf("Starting IMU calibration...\n");
141 g_cal_state.in_progress = true;
142 g_cal_state.sensor = SENSOR_TYPE_IMU;
143 g_cal_state.start_time = xTaskGetTickCount();
144
145 esp_err_t ret = mpu6050_calibrate_level(imu, imu_calibration_callback, NULL);
146
147 g_cal_state.in_progress = false;
148
149 if (ret == ESP_OK)
150 {
151 cli_transport_printf("RESULT: IMU calibration completed successfully\n");
152 cli_transport_printf("Note: Use 'save' command to persist calibration to flash\n");
153 }
154 else
155 {
156 const char *error_msg;
157 switch (ret)
158 {
159 case ESP_ERR_INVALID_STATE:
160 error_msg = "Motion detected during calibration";
161 break;
162 case ESP_ERR_TIMEOUT:
163 error_msg = "Calibration timeout";
164 break;
165 default:
166 error_msg = "Sensor communication failure";
167 break;
168 }
169 cli_transport_printf("RESULT: IMU calibration FAILED - %s\n", error_msg);
170 }
171}

227static bool check_calibration_preconditions(const char *sensor_name)
228{
229 // Check system mode
230 scout_mode_t mode = mode_manager_get_mode();
231 if (mode != SCOUT_MODE_CONFIG)
232 {
233 cli_transport_printf("ERROR: Calibration only permitted in CONFIG mode\n");
234 return false;
235 }
236
237 // Check armed status
238 safety_gates_t gates;
239 safety_manager_get_gates(&gates);
240 if (!gates.arming_allowed)
241 {
242 // This is actually checking if we're disarmed indirectly
243 // In a real system, we'd have an explicit armed/disarmed state
244 // For now, we assume if arming is blocked by other reasons, we're disarmed
245 ESP_LOGD(TAG, "System appears disarmed, calibration allowed");
246 }
247
248 // Additional safety check - ensure system is stable
249 // In real implementation, might check for recent movement, etc.
250
251 return true;
252}

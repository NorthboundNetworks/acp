48int test_telemetry_frame_processing(void)
49{
50 printf("=== Testing ACP Telemetry Frame Processing ===\n");
51
52 /* Create sample telemetry data */
53 telemetry_data_t telem_data = {
54 .timestamp = 1234567890,
55 .temperature = 2150, /* 21.5°C */
56 .voltage = 1205, /* 12.05V */
57 .status = 0x01, /* Online */
58 .reserved = 0};
59
60 printf("Sample telemetry data:\n");
61 printf(" Timestamp: %u\n", telem_data.timestamp);
62 printf(" Temperature: %d (%.1f°C)\n", telem_data.temperature, telem_data.temperature / 100.0);
63 printf(" Voltage: %d (%.2fV)\n", telem_data.voltage, telem_data.voltage / 100.0);
64 printf(" Status: 0x%02X\n", telem_data.status);
65
66 /* Step 1: Create ACP frame */
67 acp_frame_t frame;
68 int result = acp_frame_create_telemetry(&frame, ACP_FRAME_TYPE_TELEMETRY, 42,
69 &telem_data, sizeof(telem_data));
70 if (result != ACP_OK)
71 {
72 printf("ERROR: Failed to create telemetry frame: %d\n", result);
73 return -1;
74 }
75
76 printf("\nCreated ACP frame:\n");
77 printf(" Version: 0x%02X\n", frame.version);
78 printf(" Type: 0x%02X\n", frame.type);
79 printf(" Flags: 0x%02X\n", frame.flags);
80 printf(" Sequence: %u\n", frame.sequence);
81 printf(" Length: %u bytes\n", frame.length);
82
83 /* Step 2: Encode frame */
84 uint8_t encoded_buffer[256];
85 size_t encoded_len;
86 result = acp_frame_encode(&frame, encoded_buffer, sizeof(encoded_buffer), &encoded_len);
87 if (result != ACP_OK)
88 {
89 printf("ERROR: Failed to encode frame: %d\n", result);
90 return -1;
91 }
92
93 printf("\nEncoded frame (%zu bytes):\n", encoded_len);
94 printf(" Raw bytes: ");
95 for (size_t i = 0; i < encoded_len && i < 32; i++)
96 {
97 printf("%02X ", encoded_buffer[i]);
98 if ((i + 1) % 16 == 0)
99 printf("\n ");
100 }
101 if (encoded_len > 32)
102 printf("... (%zu more)", encoded_len - 32);
103 printf("\n");
104
105 /* Verify frame has delimiters */
106 if (encoded_buffer[0] != 0x00 || encoded_buffer[encoded_len - 1] != 0x00)
107 {
108 printf("ERROR: Frame missing COBS delimiters\n");
109 return -1;
110 }
111 printf(" Frame has proper COBS delimiters: ✓\n");
112
113 /* Step 3: Decode frame */
114 acp_frame_t decoded_frame;
115 size_t consumed_bytes;
116 result = acp_frame_decode(encoded_buffer, encoded_len, &decoded_frame, &consumed_bytes);
117 if (result != ACP_OK)
118 {
119 printf("ERROR: Failed to decode frame: %d\n", result);
120 return -1;
121 }
122
123 printf("\nDecoded frame:\n");
124 printf(" Version: 0x%02X\n", decoded_frame.version);
125 printf(" Type: 0x%02X\n", decoded_frame.type);
126 printf(" Flags: 0x%02X\n", decoded_frame.flags);
127 printf(" Sequence: %u\n", decoded_frame.sequence);
128 printf(" Length: %u bytes\n", decoded_frame.length);
129 printf(" Consumed: %zu bytes\n", consumed_bytes);
130
131 /* Step 4: Verify integrity */
132 if (decoded_frame.version != frame.version ||
133 decoded_frame.type != frame.type ||
134 decoded_frame.flags != frame.flags ||
135 decoded_frame.sequence != frame.sequence ||
136 decoded_frame.length != frame.length)
137 {
138 printf("ERROR: Frame header mismatch\n");
139 return -1;
140 }
141
142 if (memcmp(decoded_frame.payload, frame.payload, frame.length) != 0)
143 {
144 printf("ERROR: Payload data mismatch\n");
145 return -1;
146 }
147
148 printf(" Frame header verification: ✓\n");
149 printf(" Payload integrity check: ✓\n");
150
151 /* Step 5: Verify telemetry data */
152 const telemetry_data_t *decoded_telem = (const telemetry_data_t *)decoded_frame.payload;
153
154 printf("\nDecoded telemetry data:\n");
155 printf(" Timestamp: %u\n", decoded_telem->timestamp);
156 printf(" Temperature: %d (%.1f°C)\n", decoded_telem->temperature, decoded_telem->temperature / 100.0);
157 printf(" Voltage: %d (%.2fV)\n", decoded_telem->voltage, decoded_telem->voltage / 100.0);
158 printf(" Status: 0x%02X\n", decoded_telem->status);
159
160 if (memcmp(&telem_data, decoded_telem, sizeof(telemetry_data_t)) != 0)
161 {
162 printf("ERROR: Telemetry data corruption\n");
163 return -1;
164 }
165
166 printf(" Telemetry data integrity: ✓\n");
167
168 printf("\n✅ User Story 1 - Telemetry Frame Processing: SUCCESS\n");
169 return 0;
170}

196 printf("ERROR: Actual size exceeds prediction\n");
197 return -1;

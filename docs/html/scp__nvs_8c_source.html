77static uint32_t calculate_config_checksum(const scp_config_t *config)
78{
79 if (!config)
80 return 0;
81
82 uint32_t checksum = 0x5C501234; // SCP magic number
83 const uint8_t *data = (const uint8_t *)config;
84
85 // Calculate checksum excluding the checksum field itself
86 size_t size = sizeof(scp_config_t) - sizeof(uint32_t);
87
88 for (size_t i = 0; i < size; i++)
89 {
90 checksum = checksum * 31 + data[i];
91 }
92
93 return checksum;
94}

99static bool validate_config(const scp_config_t *config)
100{
101 if (!config)
102 return false;
103
104 // Validate telemetry rate
105 if (config->telemetry_rate_hz == 0 || config->telemetry_rate_hz > 10)
106 {
107 ESP_LOGE(TAG, "Invalid telemetry rate: %u Hz (must be 1-10)", config->telemetry_rate_hz);
108 return false;
109 }
110
111 // Validate session timeout
112 if (config->session_timeout_ms < 60000 || config->session_timeout_ms > 3600000)
113 {
114 ESP_LOGE(TAG, "Invalid session timeout: %u ms (must be 60s-1h)", config->session_timeout_ms);
115 return false;
116 }
117
118 // Validate rate limiting
119 if (config->rate_limit_sustained == 0 || config->rate_limit_sustained > 50)
120 {
121 ESP_LOGE(TAG, "Invalid sustained rate: %u (must be 1-50)", config->rate_limit_sustained);
122 return false;
123 }
124
125 if (config->rate_limit_burst < config->rate_limit_sustained || config->rate_limit_burst > 100)
126 {
127 ESP_LOGE(TAG, "Invalid burst rate: %u (must be >= sustained and <= 100)", config->rate_limit_burst);
128 return false;
129 }
130
131 // Validate protocol version
132 if (config->protocol_version != SCP_VERSION_1_1)
133 {
134 ESP_LOGE(TAG, "Unsupported protocol version: 0x%02X", config->protocol_version);
135 return false;
136 }
137
138 // Verify checksum
139 uint32_t expected_checksum = calculate_config_checksum(config);
140 if (config->config_checksum != expected_checksum)
141 {
142 ESP_LOGE(TAG, "Configuration checksum mismatch: 0x%08X != 0x%08X",
143 config->config_checksum, expected_checksum);
144 return false;
145 }
146
147 return true;
148}

157bool scp_nvs_init(void)
158{
159 if (nvs_initialized)
160 {
161 ESP_LOGW(TAG, "SCP NVS already initialized");
162 return true;
163 }
164
165 ESP_LOGI(TAG, "Initializing SCP NVS storage");
166
167 // Initialize default configuration
168 init_default_config(&current_config);
169 current_config.config_checksum = calculate_config_checksum(&current_config);
170
171 // Open NVS handle
172 esp_err_t err = nvs_open(SCP_NVS_NAMESPACE, NVS_READWRITE, &config_nvs_handle);
173 if (err != ESP_OK)
174 {
175 ESP_LOGE(TAG, "Failed to open NVS namespace: %s", esp_err_to_name(err));
176 return false;
177 }
178
179 nvs_initialized = true;
180
181 // Try to load existing configuration
182 if (scp_nvs_load_config())
183 {
184 ESP_LOGI(TAG, "Loaded existing SCP configuration from NVS");
185 }
186 else
187 {
188 ESP_LOGI(TAG, "Using default SCP configuration");
189 // Save default configuration to NVS
190 scp_nvs_save_config();
191 }
192
193 ESP_LOGI(TAG, "SCP NVS initialization complete");
194 return true;
195}

200bool scp_nvs_load_config(void)
201{
202 if (!nvs_initialized)
203 {
204 ESP_LOGE(TAG, "NVS not initialized");
205 return false;
206 }
207
208 scp_config_t loaded_config;
209 size_t required_size = sizeof(scp_config_t);
210
211 esp_err_t err = nvs_get_blob(config_nvs_handle, "config", &loaded_config, &required_size);
212 if (err == ESP_ERR_NVS_NOT_FOUND)
213 {
214 ESP_LOGI(TAG, "No existing configuration found in NVS");
215 return false;
216 }
217
218 if (err != ESP_OK)
219 {
220 ESP_LOGE(TAG, "Failed to load configuration: %s", esp_err_to_name(err));
221 return false;
222 }
223
224 // Validate loaded configuration
225 if (!validate_config(&loaded_config))
226 {
227 ESP_LOGE(TAG, "Loaded configuration is invalid, using defaults");
228 return false;
229 }
230
231 // Configuration is valid, update current config
232 memcpy(&current_config, &loaded_config, sizeof(scp_config_t));
233
234 ESP_LOGI(TAG, "Configuration loaded: rate=%u Hz, timeout=%u ms, debug=%s",
235 current_config.telemetry_rate_hz,
236 current_config.session_timeout_ms,
237 current_config.debug_logging_enabled ? "ON" : "OFF");
238
239 return true;
240}

246{

252

256 // Validate before saving

262

264 if (err != ESP_OK)

273 ESP_LOGE(TAG, "Failed to commit configuration: %s", esp_err_to_name(err));

42esp_err_t bus_io_init_i2c(void)
43{
44 if (i2c_initialized) {
45 ESP_LOGW(TAG, "I2C already initialized");
46 return ESP_OK;
47 }
48
49 i2c_config_t conf = {
50 .mode = I2C_MODE_MASTER,
51 .sda_io_num = I2C_SDA_PIN,
52 .scl_io_num = I2C_SCL_PIN,
53 .sda_pullup_en = GPIO_PULLUP_ENABLE,
54 .scl_pullup_en = GPIO_PULLUP_ENABLE,
55 .master.clk_speed = I2C_MASTER_FREQ,
56 };
57
58 esp_err_t ret = i2c_param_config(I2C_MASTER_NUM, &conf);
59 if (ret != ESP_OK) {
60 ESP_LOGE(TAG, "I2C param config failed: %s", esp_err_to_name(ret));
61 return ret;
62 }
63
64 ret = i2c_driver_install(I2C_MASTER_NUM, conf.mode, 0, 0, 0);
65 if (ret != ESP_OK) {
66 ESP_LOGE(TAG, "I2C driver install failed: %s", esp_err_to_name(ret));
67 return ret;
68 }
69
70 i2c_initialized = true;
71 ESP_LOGI(TAG, "I2C initialized (SCL=%d, SDA=%d, freq=%dHz)",
72 I2C_SCL_PIN, I2C_SDA_PIN, I2C_MASTER_FREQ);
73
74 return ESP_OK;
75}

113 uart_config_t uart_config = {

119 .source_clk = UART_SCLK_DEFAULT,

142 uart_gps_initialized = true;
143 ESP_LOGI(TAG, "UART GPS initialized (TX=%d, RX=%d, baud=%d)",
144 UART_GPS_TX_PIN, UART_GPS_RX_PIN, UART_GPS_BAUD);

149esp_err_t bus_io_add_spi_device(spi_device_handle_t *handle, int cs_pin, int clock_speed_hz)
150{
151 if (!spi_initialized) {
152 ESP_LOGE(TAG, "SPI bus not initialized");
153 return ESP_ERR_INVALID_STATE;
154 }
155
156 spi_device_interface_config_t devcfg = {
157 .clock_speed_hz = clock_speed_hz,
158 .mode = 0, // SPI mode 0
159 .spics_io_num = cs_pin,
160 .queue_size = 7,
161 .command_bits = 0,
162 .address_bits = 8, // Most sensors use 8-bit register addresses
163 .dummy_bits = 0,
164 };
165
166 esp_err_t ret = spi_bus_add_device(SPI2_HOST, &devcfg, handle);
167 if (ret != ESP_OK) {
168 ESP_LOGE(TAG, "SPI add device failed (cs=%d): %s", cs_pin, esp_err_to_name(ret));
169 return ret;
170 }
171
172 ESP_LOGI(TAG, "SPI device added (CS=%d, speed=%dHz)", cs_pin, clock_speed_hz);
173 return ESP_OK;
174}

181esp_err_t bus_io_i2c_read_register(uint8_t device_addr, uint8_t reg_addr, uint8_t *data, size_t len)
182{
183 if (!i2c_initialized) {
184 ESP_LOGE(TAG, "I2C not initialized");
185 return ESP_ERR_INVALID_STATE;
186 }
187
188 i2c_cmd_handle_t cmd = i2c_cmd_link_create();
189
190 // Write register address
191 i2c_master_start(cmd);
192 i2c_master_write_byte(cmd, (device_addr << 1) | I2C_MASTER_WRITE, true);
193 i2c_master_write_byte(cmd, reg_addr, true);
194
195 // Read data
196 i2c_master_start(cmd);
197 i2c_master_write_byte(cmd, (device_addr << 1) | I2C_MASTER_READ, true);
198 i2c_master_read(cmd, data, len, I2C_MASTER_LAST_NACK);
199 i2c_master_stop(cmd);
200
201 esp_err_t ret = i2c_master_cmd_begin(I2C_MASTER_NUM, cmd, pdMS_TO_TICKS(100));
202 i2c_cmd_link_delete(cmd);
203
204 return ret;
205}

227esp_err_t bus_io_spi_read_register(spi_device_handle_t handle, uint8_t reg_addr, uint8_t *data, size_t len)
228{
229 if (handle == NULL) {
230 ESP_LOGE(TAG, "Invalid SPI device handle");
231 return ESP_ERR_INVALID_ARG;
232 }
233
234 spi_transaction_t trans = {
235 .addr = reg_addr | 0x80, // Set read bit for most sensors
236 .length = len * 8, // Length in bits
237 .rx_buffer = data,
238 };
239
240 return spi_device_transmit(handle, &trans);
241}

243esp_err_t bus_io_spi_write_register(spi_device_handle_t handle, uint8_t reg_addr, uint8_t data)
244{
245 if (handle == NULL) {
246 ESP_LOGE(TAG, "Invalid SPI device handle");
247 return ESP_ERR_INVALID_ARG;
248 }
249
250 spi_transaction_t trans = {
251 .addr = reg_addr & 0x7F, // Clear read bit for write
252 .length = 8, // 8 bits
253 .tx_data[0] = data,
254 .flags = SPI_TRANS_USE_TXDATA,
255 };
256
257 return spi_device_transmit(handle, &trans);

260esp_err_t bus_io_uart_gps_read(uint8_t *data, size_t max_len, size_t *bytes_read, uint32_t timeout_ms)
261{
262 if (!uart_gps_initialized) {
263 ESP_LOGE(TAG, "UART GPS not initialized");
264 return ESP_ERR_INVALID_STATE;
265 }
266
267 int len = uart_read_bytes(UART_GPS_NUM, data, max_len, pdMS_TO_TICKS(timeout_ms));
268 if (len < 0) {
269 *bytes_read = 0;
270 return ESP_ERR_TIMEOUT;
271 }
272
273 *bytes_read = len;

277void bus_io_deinit_all(void)
278{
279 if (uart_gps_initialized) {
280 uart_driver_delete(UART_GPS_NUM);
281 uart_gps_initialized = false;
282 ESP_LOGI(TAG, "UART GPS deinitialized");
283 }
284
285 if (spi_initialized) {
286 if (spi_baro_handle) {
287 spi_bus_remove_device(spi_baro_handle);
288 spi_baro_handle = NULL;
289 }
290 spi_bus_free(SPI2_HOST);
291 spi_initialized = false;
292 ESP_LOGI(TAG, "SPI bus deinitialized");

294
295 if (i2c_initialized) {

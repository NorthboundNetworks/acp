124esp_err_t gimbal_init(void)
125{
126 if (s_gimbal_initialized)
127 {
128 ESP_LOGW(TAG, "Gimbal service already initialized");
129 return ESP_OK;
130 }
131
132 ESP_LOGI(TAG, "Initializing gimbal service...");
133
134 // Setup servo PWM outputs
135 esp_err_t ret = gimbal_setup_servo_pwm();
136 if (ret != ESP_OK)
137 {
138 ESP_LOGE(TAG, "Failed to setup servo PWM: %s", esp_err_to_name(ret));
139 return ret;
140 }
141
142 // Initialize default state
143 s_current_mode = GIMBAL_MODE_FIXED;
144 s_current_attitude = (gimbal_attitude_t){0.0f, 0.0f, 0.0f};
145 s_target_attitude = (gimbal_attitude_t){0.0f, 0.0f, 0.0f};
146 s_last_update_ms = gimbal_get_time_ms();
147 s_fault_detected = false;
148 s_sweep_direction = 1.0f;
149
150 // Set initial servo positions to center
151 gimbal_set_servo_angle(GIMBAL_PITCH_LEDC_CHANNEL, 0.0f);
152 gimbal_set_servo_angle(GIMBAL_YAW_LEDC_CHANNEL, 0.0f);
153
154 s_gimbal_initialized = true;
155 ESP_LOGI(TAG, "Gimbal service initialized successfully");
156
157 return ESP_OK;
158}

171 gimbal_set_servo_angle(GIMBAL_YAW_LEDC_CHANNEL, 0.0f);

177 s_gimbal_initialized = false;
178 ESP_LOGI(TAG, "Gimbal service shut down");
179}

186esp_err_t gimbal_set_mode(gimbal_mode_t mode)
187{
188 if (!s_gimbal_initialized)
189 {
190 ESP_LOGE(TAG, "Gimbal service not initialized");
191 return ESP_ERR_INVALID_STATE;
192 }
193
194 if (mode < GIMBAL_MODE_FIXED || mode > GIMBAL_MODE_SEARCH_SWEEP)
195 {
196 ESP_LOGE(TAG, "Invalid gimbal mode: %d", mode);
197 return ESP_ERR_INVALID_ARG;
198 }
199
200 const char *mode_names[] = {"FIXED", "FOLLOW_HEADING", "SEARCH_SWEEP"};
201 ESP_LOGI(TAG, "Setting gimbal mode to %s", mode_names[mode]);
202
203 s_current_mode = mode;
204
205 // Reset sweep state when entering sweep mode
206 if (mode == GIMBAL_MODE_SEARCH_SWEEP)
207 {
208 s_sweep_direction = 1.0f;
209 s_target_attitude.yaw = s_sweep_config.center_yaw - s_sweep_config.sweep_range;
210 }
211
212 return ESP_OK;
213}

220esp_err_t gimbal_set_attitude(const gimbal_attitude_t *attitude)
221{
222 if (!s_gimbal_initialized)
223 {
224 ESP_LOGE(TAG, "Gimbal service not initialized");
225 return ESP_ERR_INVALID_STATE;
226 }
227
228 if (attitude == NULL)
229 {
230 ESP_LOGE(TAG, "Null attitude pointer");
231 return ESP_ERR_INVALID_ARG;
232 }
233
234 // Validate attitude limits
235 if (attitude->pitch < GIMBAL_PITCH_MIN_DEG || attitude->pitch > GIMBAL_PITCH_MAX_DEG ||
236 attitude->yaw < GIMBAL_YAW_MIN_DEG || attitude->yaw > GIMBAL_YAW_MAX_DEG ||
237 attitude->roll < GIMBAL_ROLL_MIN_DEG || attitude->roll > GIMBAL_ROLL_MAX_DEG)
238 {
239 ESP_LOGE(TAG, "Attitude out of limits: pitch=%.1f, yaw=%.1f, roll=%.1f",
240 attitude->pitch, attitude->yaw, attitude->roll);
241 return ESP_ERR_INVALID_ARG;
242 }
243
244 ESP_LOGD(TAG, "Setting target attitude: pitch=%.1f°, yaw=%.1f°, roll=%.1f°",
245 attitude->pitch, attitude->yaw, attitude->roll);
246
247 s_target_attitude = *attitude;
248
249 return ESP_OK;
250}

252esp_err_t gimbal_get_status(gimbal_status_t *status)
253{
254 if (status == NULL)
255 {
256 return ESP_ERR_INVALID_ARG;
257 }
258
259 status->mode = s_current_mode;
260 status->current_attitude = s_current_attitude;
261 status->target_attitude = s_target_attitude;
262 status->last_update_ms = s_last_update_ms;
263 status->fault_detected = s_fault_detected;
264
265 // Check if gimbal is moving (within tolerance)
266 float pitch_diff = fabs(s_target_attitude.pitch - s_current_attitude.pitch);
267 float yaw_diff = fabs(s_target_attitude.yaw - s_current_attitude.yaw);
268 status->is_moving = (pitch_diff > GIMBAL_POSITION_TOLERANCE_DEG) ||
269 (yaw_diff > GIMBAL_POSITION_TOLERANCE_DEG);
270
271 return ESP_OK;
272}

316esp_err_t gimbal_update(float vehicle_heading)
317{
318 if (!s_gimbal_initialized)
319 {
320 return ESP_ERR_INVALID_STATE;
321 }
322
323 uint32_t current_time = gimbal_get_time_ms();
324 uint32_t dt_ms = current_time - s_last_update_ms;
325
326 // Skip update if called too frequently
327 if (dt_ms < GIMBAL_UPDATE_PERIOD_MS)
328 {
329 return ESP_OK;
330 }
331
332 s_last_update_ms = current_time;
333
334 // Update based on current mode
335 switch (s_current_mode)
336 {
337 case GIMBAL_MODE_FIXED:
338 gimbal_update_fixed_mode();
339 break;
340
341 case GIMBAL_MODE_FOLLOW_HEADING:
342 gimbal_update_follow_heading_mode(vehicle_heading);
343 break;
344
345 case GIMBAL_MODE_SEARCH_SWEEP:
346 gimbal_update_search_sweep_mode();
347 break;
348
349 default:
350 ESP_LOGW(TAG, "Unknown gimbal mode: %d", s_current_mode);
351 break;
352 }
353
354 // Update servo positions
355 esp_err_t ret1 = gimbal_set_servo_angle(GIMBAL_PITCH_LEDC_CHANNEL, s_current_attitude.pitch);
356 esp_err_t ret2 = gimbal_set_servo_angle(GIMBAL_YAW_LEDC_CHANNEL, s_current_attitude.yaw);
357
358 // Check for servo errors
359 if (ret1 != ESP_OK || ret2 != ESP_OK)
360 {
361 s_fault_detected = true;
362 ESP_LOGE(TAG, "Servo control error: pitch=%s, yaw=%s",
363 esp_err_to_name(ret1), esp_err_to_name(ret2));
364 return ESP_FAIL;
365 }
366
367 return ESP_OK;
368}

374static esp_err_t gimbal_setup_servo_pwm(void)
375{
376 // Configure LEDC timer
377 ledc_timer_config_t timer_config = {
378 .speed_mode = LEDC_LOW_SPEED_MODE,
379 .timer_num = GIMBAL_LEDC_TIMER,
380 .duty_resolution = GIMBAL_PWM_RESOLUTION,
381 .freq_hz = GIMBAL_SERVO_FREQ_HZ,
382 .clk_cfg = LEDC_AUTO_CLK};
383
384 esp_err_t ret = ledc_timer_config(&timer_config);
385 if (ret != ESP_OK)
386 {
387 ESP_LOGE(TAG, "LEDC timer config failed: %s", esp_err_to_name(ret));
388 return ret;
389 }
390
391 // Configure pitch servo channel
392 ledc_channel_config_t pitch_channel = {
393 .speed_mode = LEDC_LOW_SPEED_MODE,
394 .channel = GIMBAL_PITCH_LEDC_CHANNEL,
395 .timer_sel = GIMBAL_LEDC_TIMER,
396 .intr_type = LEDC_INTR_DISABLE,
397 .gpio_num = GIMBAL_PITCH_PIN,
398 .duty = 0,
399 .hpoint = 0};
400
401 ret = ledc_channel_config(&pitch_channel);
402 if (ret != ESP_OK)
403 {
404 ESP_LOGE(TAG, "Pitch servo config failed: %s", esp_err_to_name(ret));
405 return ret;
406 }
407
408 // Configure yaw servo channel
409 ledc_channel_config_t yaw_channel = {
410 .speed_mode = LEDC_LOW_SPEED_MODE,
411 .channel = GIMBAL_YAW_LEDC_CHANNEL,
412 .timer_sel = GIMBAL_LEDC_TIMER,
413 .intr_type = LEDC_INTR_DISABLE,
414 .gpio_num = GIMBAL_YAW_PIN,

427
428 return ESP_OK;

431static esp_err_t gimbal_set_servo_angle(int channel, float angle_deg)
432{
433 // Constrain angle to servo limits (-90 to +90 degrees)
434 angle_deg = gimbal_constrain_angle(angle_deg, -90.0f, 90.0f);
435
436 // Convert angle to pulse width (microseconds)
437 float pulse_width_us = SERVO_CENTER_PULSE_WIDTH_US +
438 (angle_deg / 90.0f) * (SERVO_MAX_PULSE_WIDTH_US - SERVO_CENTER_PULSE_WIDTH_US);
439
440 // Convert pulse width to PWM duty cycle
441 uint32_t duty = (uint32_t)((pulse_width_us / SERVO_PERIOD_US) * GIMBAL_PWM_MAX_DUTY);
442
443 // Set PWM duty cycle
444 esp_err_t ret = ledc_set_duty(LEDC_LOW_SPEED_MODE, channel, duty);
445 if (ret == ESP_OK)
446 {
447 ret = ledc_update_duty(LEDC_LOW_SPEED_MODE, channel);
448 }
449
450 if (ret != ESP_OK)
451 {
452 ESP_LOGE(TAG, "Failed to set servo angle: channel=%d, angle=%.1f°, duty=%lu",
453 channel, angle_deg, duty);
454 }
455
456 return ret;
457}

468static void gimbal_update_fixed_mode(void)
469{
470 // In fixed mode, smoothly move towards target attitude
471 float dt_sec = GIMBAL_UPDATE_PERIOD_MS / 1000.0f;
472 float max_movement = GIMBAL_MAX_RATE_DEG_PER_SEC * dt_sec;
473
474 // Update pitch
475 float pitch_error = s_target_attitude.pitch - s_current_attitude.pitch;
476 if (fabs(pitch_error) > max_movement)
477 {
478 s_current_attitude.pitch += (pitch_error > 0) ? max_movement : -max_movement;
479 }
480 else
481 {
482 s_current_attitude.pitch = s_target_attitude.pitch;
483 }
484
485 // Update yaw
486 float yaw_error = s_target_attitude.yaw - s_current_attitude.yaw;
487 if (fabs(yaw_error) > max_movement)
488 {
489 s_current_attitude.yaw += (yaw_error > 0) ? max_movement : -max_movement;
490 }
491 else
492 {
493 s_current_attitude.yaw = s_target_attitude.yaw;
494 }
495
496 // Roll is typically fixed at 0 for gimbal applications
497 s_current_attitude.roll = 0.0f;
498}

510static void gimbal_update_search_sweep_mode(void)
511{
512 if (!s_sweep_config.enabled)
513 {
514 return;
515 }
516
517 float dt_sec = GIMBAL_UPDATE_PERIOD_MS / 1000.0f;
518 float movement = s_sweep_config.sweep_rate * dt_sec * s_sweep_direction;
519
520 // Update yaw position
521 s_current_attitude.yaw += movement;
522
523 // Check sweep limits and reverse direction if needed
524 float min_yaw = s_sweep_config.center_yaw - s_sweep_config.sweep_range;
525 float max_yaw = s_sweep_config.center_yaw + s_sweep_config.sweep_range;
526
527 if (s_current_attitude.yaw >= max_yaw)
528 {
529 s_current_attitude.yaw = max_yaw;
530 s_sweep_direction = -1.0f;
531 }
532 else if (s_current_attitude.yaw <= min_yaw)
533 {
534 s_current_attitude.yaw = min_yaw;
535 s_sweep_direction = 1.0f;
536 }
537
538 // Keep pitch at target, roll at 0
539 s_current_attitude.pitch = s_target_attitude.pitch;
540 s_current_attitude.roll = 0.0f;
541}

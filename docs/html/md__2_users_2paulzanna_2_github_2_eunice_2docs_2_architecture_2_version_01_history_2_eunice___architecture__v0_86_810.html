1) Executive Summary

v0.6.10 introduces a comprehensive search and knowledge discovery system with full-text search, semantic search capabilities, faceted filtering, AI-powered query understanding, and intelligent result ranking. The system enables researchers to efficiently discover relevant tasks, protocols, findings, and insights across the entire Eunice platform.

7) Search Engine Implementation

7.1 Elasticsearch Integration

class ElasticsearchService {
private client: Client
async search(query: SearchQuery): Promise<SearchResponse> {
const esQuery = this.buildElasticsearchQuery(query)
try {
const response = await this.client.search({
index: this.getRelevantIndices(query.filters.contentTypes),
body: esQuery,
timeout: '5s'
})
return this.transformElasticsearchResponse(response, query)
} catch (error) {
console.error('Elasticsearch query failed:', error)
throw new SearchError('Search temporarily unavailable', 'SEARCH_ENGINE_ERROR')
}
}
private buildElasticsearchQuery(query: SearchQuery): any {
const mustQueries = []
const filterQueries = []
// Main search query
if (query.query) {
mustQueries.push({
multi_match: {
query: query.query,
fields: [
'title^3',
'content^2',
'tags^2',
'metadata.description',
'metadata.keywords'
],
type: 'best_fields',
fuzziness: 'AUTO'
}
})
}
// Apply filters
if (query.filters.contentTypes?.length) {
filterQueries.push({
terms: { 'type': query.filters.contentTypes }
})
}
if (query.filters.dateRange) {
filterQueries.push({
range: {
[query.filters.dateRange.field]: {
gte: query.filters.dateRange.from?.toISOString(),
lte: query.filters.dateRange.to?.toISOString()
}
}
})
}
// Build aggregations for facets
const aggregations = this.buildAggregations(query.aggregations)
return {
query: {
bool: {
must: mustQueries,
filter: filterQueries
}
},
highlight: query.highlight ? {
fields: {
title: {},
content: { fragment_size: 150, number_of_fragments: 3 }
}
} : undefined,
sort: this.buildSortCriteria(query.sort),
from: (query.pagination.page - 1) * query.pagination.size,
size: query.pagination.size,
aggs: aggregations
}
}
}

7.2 Semantic Search Service

class SemanticSearchService {
private embeddingService: EmbeddingService
private vectorDb: VectorDatabase
async semanticSearch(query: string, filters: SearchFilters): Promise<SearchResult[]> {
// Generate query embedding
const queryEmbedding = await this.embeddingService.generateEmbedding(query)
// Perform vector similarity search
const similarDocuments = await this.vectorDb.searchSimilar(
queryEmbedding,
{
limit: 50,
threshold: 0.7,
filters: this.convertFiltersToVectorFilters(filters)
}
)
// Enhance results with metadata
const enhancedResults = await this.enhanceWithMetadata(similarDocuments)
return enhancedResults.map(doc => ({
id: doc.id,
type: doc.type,
title: doc.metadata.title,
content: doc.chunk_text,
highlights: [], // Semantic search doesn't provide traditional highlights
score: doc.similarity_score,
metadata: doc.metadata,
searchType: 'semantic'
}))
}
async hybridSearch(query: SearchQuery): Promise<SearchResult[]> {
// Execute both text and semantic search in parallel
const [textResults, semanticResults] = await Promise.all([
this.elasticsearchService.search(query),
this.semanticSearch(query.query, query.filters)
])
// Fuse results using rank-based fusion
const fusedResults = this.fuseSearchResults(
textResults.results,
semanticResults,
{ textWeight: 0.6, semanticWeight: 0.4 }
)
return fusedResults
}
private fuseSearchResults(
textResults: SearchResult[],
semanticResults: SearchResult[],
weights: { textWeight: number, semanticWeight: number }
): SearchResult[] {
const resultMap = new Map<string, SearchResult>()
// Process text search results
textResults.forEach((result, index) => {
const fusedScore = weights.textWeight * (1 / (index + 1))
resultMap.set(result.id, {
...result,
score: fusedScore,
searchSources: ['text']
})
})
// Process semantic search results
semanticResults.forEach((result, index) => {
const semanticScore = weights.semanticWeight * (1 / (index + 1))
if (resultMap.has(result.id)) {
// Combine scores for documents found in both searches
const existing = resultMap.get(result.id)!
existing.score += semanticScore
existing.searchSources!.push('semantic')
} else {
resultMap.set(result.id, {
...result,
score: semanticScore,
searchSources: ['semantic']
})
}
})
// Sort by combined score and return
return Array.from(resultMap.values())
.sort((a, b) => b.score - a.score)
}
}

7.3 Knowledge Graph Search

class KnowledgeGraphService {
private neo4jDriver: Driver
async findRelatedEntities(
entityId: string,
relationshipTypes?: string[],
maxDepth: number = 2
): Promise<KnowledgeGraph> {
const session = this.neo4jDriver.session()
try {
const query = `
MATCH (start:Entity {id: $entityId})
CALL apoc.path.subgraphNodes(start, {
relationshipFilter: $relationshipTypes,
maxLevel: $maxDepth
})
YIELD node
OPTIONAL MATCH (node)-[r]-(related)
WHERE related IN nodes
RETURN node, collect(DISTINCT r) as relationships, collect(DISTINCT related) as relatedNodes
`
const result = await session.run(query, {
entityId,
relationshipTypes: relationshipTypes?.join('|') || '',
maxDepth
})
return this.buildKnowledgeGraph(result.records)
} finally {
await session.close()
}
}
async searchByPath(
startEntity: string,
endEntity: string,
pathTypes: string[]
): Promise<EntityPath[]> {
const session = this.neo4jDriver.session()
try {
const query = `
MATCH path = (start:Entity {id: $startEntity})-[r*1..3]-(end:Entity {id: $endEntity})
WHERE ALL(rel IN r WHERE type(rel) IN $pathTypes)
RETURN path, length(path) as pathLength
ORDER BY pathLength
LIMIT 10
`
const result = await session.run(query, {
startEntity,
endEntity,
pathTypes
})
return result.records.map(record => this.extractEntityPath(record))
} finally {
await session.close()
}
}
}

8) Query Processing Intelligence

8.1 Natural Language Query Parser

class QueryProcessor {
private nlpPipeline: NLPPipeline
async processQuery(query: string, context?: SearchContext): Promise<ProcessedQuery> {
// Extract intent and entities
const analysis = await this.nlpPipeline.analyze(query)
// Expand query with synonyms and related terms
const expandedTerms = await this.expandQuery(query, analysis.entities)
// Build structured query
const structuredQuery = this.buildStructuredQuery(analysis, expandedTerms)
// Apply contextual filters
if (context) {
structuredQuery.filters = this.applyContextualFilters(
structuredQuery.filters,
context
)
}
return structuredQuery
}
private async expandQuery(query: string, entities: Entity[]): Promise<string[]> {
const expansions = []
// Add synonyms
const synonyms = await this.synonymService.getSynonyms(query)
expansions.push(...synonyms)
// Add related domain terms
for (const entity of entities) {
const relatedTerms = await this.getRelatedTerms(entity.text, entity.type)
expansions.push(...relatedTerms)
}
// Add acronym expansions
const acronymExpansions = await this.expandAcronyms(query)
expansions.push(...acronymExpansions)
return expansions
}
private buildStructuredQuery(analysis: NLPAnalysis, expansions: string[]): ProcessedQuery {
const filters: SearchFilters = {}
// Extract temporal filters
if (analysis.dateEntities.length > 0) {
filters.dateRange = this.extractDateRange(analysis.dateEntities)
}
// Extract person filters
const personEntities = analysis.entities.filter(e => e.type === 'PERSON')
if (personEntities.length > 0) {
filters.authors = personEntities.map(e => e.text)
}
// Determine search type based on query characteristics
const searchType = this.determineSearchType(analysis)
return {
originalQuery: analysis.originalText,
processedQuery: this.buildQueryString(analysis, expansions),
searchType,
filters,
intent: analysis.intent,
entities: analysis.entities,
confidence: analysis.confidence
}
}
}

9) Result Ranking and Personalization

9.1 Machine Learning Ranker

class MLRankingService {
private model: RankingModel
async rankResults(
results: SearchResult[],
query: SearchQuery,
user?: User
): Promise<SearchResult[]> {
// Extract features for each result
const rankedResults = await Promise.all(
results.map(async result => {
const features = await this.extractFeatures(result, query, user)
const score = await this.model.predict(features)
return {
...result,
mlScore: score,
features: features
}
})
)
// Sort by ML score and return
return rankedResults.sort((a, b) => b.mlScore - a.mlScore)
}
private async extractFeatures(
result: SearchResult,
query: SearchQuery,
user?: User
): Promise<FeatureVector> {
const features: FeatureVector = {
// Text relevance features
titleMatch: this.calculateTitleMatch(result.title, query.query),
contentMatch: this.calculateContentMatch(result.content, query.query),
exactMatch: result.content.toLowerCase().includes(query.query.toLowerCase()),
// Document quality features
documentLength: result.content.length,
hasImages: result.metadata.images?.length > 0,
authorReputation: await this.getAuthorReputation(result.metadata.author),
// Freshness features
daysSinceCreated: this.daysSince(result.metadata.createdAt),
daysSinceModified: this.daysSince(result.metadata.modifiedAt),
// User interaction features
clickThroughRate: await this.getCTR(result.id),
userPreviousViews: user ? await this.getUserViewCount(user.id, result.id) : 0,
// Personalization features
userTopicMatch: user ? this.calculateTopicMatch(result.metadata.topics, user.interests) : 0,
collaboratorMatch: user ? await this.getCollaboratorMatch(result.metadata.author, user.id) : 0
}
return features
}
}

9.2 Personalization Engine

class PersonalizationService {
async personalizeResults(
results: SearchResult[],
userId: string
): Promise<SearchResult[]> {
const userProfile = await this.getUserProfile(userId)
if (!userProfile.personalizationEnabled) {
return results
}
// Apply personalization boosters
const personalizedResults = results.map(result => {
let personalizedScore = result.score
// Boost based on topic interests
const topicBoost = this.calculateTopicBoost(
result.metadata.topics,
userProfile.topicInterests
)
personalizedScore *= (1 + topicBoost)
// Boost based on preferred content types
if (userProfile.preferredContentTypes.includes(result.type)) {
personalizedScore *= 1.2
}
// Boost based on collaboration history
const collaborationBoost = this.calculateCollaborationBoost(
result.metadata.author,
userProfile.collaborators
)
personalizedScore *= (1 + collaborationBoost)
return {
...result,
personalizedScore,
personalizedBoosts: {
topic: topicBoost,
contentType: userProfile.preferredContentTypes.includes(result.type),
collaboration: collaborationBoost
}
}
})
// Re-sort by personalized score
return personalizedResults.sort((a, b) => b.personalizedScore - a.personalizedScore)
}
}

10) Performance Optimization

10.1 Caching Strategy

class SearchCacheService {
private redis: Redis
private cacheKeyGenerator: CacheKeyGenerator
async getCachedResults(query: SearchQuery): Promise<SearchResponse | null> {
const cacheKey = this.cacheKeyGenerator.generate(query)
const cached = await this.redis.get(cacheKey)
if (cached) {
const result = JSON.parse(cached)
// Check if cache is still valid
if (this.isCacheValid(result.timestamp, query)) {
return result.data
}
}
return null
}
async cacheResults(
query: SearchQuery,
results: SearchResponse,
ttl: number = 300 // 5 minutes default
): Promise<void> {
const cacheKey = this.cacheKeyGenerator.generate(query)
const cacheData = {
data: results,
timestamp: Date.now(),
ttl
}
await this.redis.setex(cacheKey, ttl, JSON.stringify(cacheData))
}
private isCacheValid(timestamp: number, query: SearchQuery): boolean {
const age = Date.now() - timestamp
// Different TTL based on query characteristics
if (query.filters.dateRange?.to &&
query.filters.dateRange.to > new Date(Date.now() - 24 * 60 * 60 * 1000)) {
// Recent date filters - shorter cache
return age < 60 * 1000 // 1 minute
}
if (query.sort?.some(s => s.field === 'created_at' && s.order === 'desc')) {
// Sorting by recent - shorter cache
return age < 5 * 60 * 1000 // 5 minutes
}
// Default cache validity
return age < 15 * 60 * 1000 // 15 minutes
}
}

10.2 Index Optimization

class IndexOptimizationService {
async optimizeSearchIndices(): Promise<void> {
// Optimize Elasticsearch indices
await this.optimizeElasticsearchIndices()
// Rebuild vector indices
await this.rebuildVectorIndices()
// Update knowledge graph indices
await this.updateKnowledgeGraphIndices()
// Clean up old cache entries
await this.cleanupSearchCaches()
}
private async optimizeElasticsearchIndices(): Promise<void> {
const indices = await this.elasticsearchClient.cat.indices({ format: 'json' })
for (const index of indices) {
if (index['docs.count'] > 100000) {
// Force merge large indices
await this.elasticsearchClient.indices.forcemerge({
index: index.index,
max_num_segments: 1
})
}
// Update index settings for better performance
await this.elasticsearchClient.indices.putSettings({
index: index.index,
body: {
refresh_interval: '30s', // Reduce refresh frequency
number_of_replicas: 1
}
})
}
}
}

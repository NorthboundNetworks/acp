10esp_err_t hmac_sha256_compute(const uint8_t *key, size_t key_len,
11 const uint8_t *data, size_t data_len,
12 uint8_t *hmac_out)
13{
14 if (key == NULL || data == NULL || hmac_out == NULL)
15 {
16 ESP_LOGE(TAG, "Invalid parameters for HMAC computation");
17 return ESP_ERR_INVALID_ARG;
18 }
19
20 if (key_len == 0 || data_len == 0)
21 {
22 ESP_LOGE(TAG, "Key or data length is zero");
23 return ESP_ERR_INVALID_ARG;
24 }
25
26 const mbedtls_md_info_t *md_info = mbedtls_md_info_from_type(MBEDTLS_MD_SHA256);
27 if (md_info == NULL)
28 {
29 ESP_LOGE(TAG, "Failed to get SHA256 MD info");
30 return ESP_FAIL;
31 }
32
33 int ret = mbedtls_md_hmac(md_info, key, key_len, data, data_len, hmac_out);
34 if (ret != 0)
35 {
36 ESP_LOGE(TAG, "mbedTLS HMAC computation failed: -0x%04x", -ret);
37 return ESP_FAIL;
38 }
39
40 ESP_LOGD(TAG, "HMAC-SHA256 computed successfully");
41 return ESP_OK;
42}

44esp_err_t hmac_sha256_compute_hex_key(const char *key_hex,
45 const uint8_t *data, size_t data_len,
46 uint8_t *hmac_out)
47{
48 if (key_hex == NULL || data == NULL || hmac_out == NULL)
49 {
50 ESP_LOGE(TAG, "Invalid parameters for HMAC computation with hex key");
51 return ESP_ERR_INVALID_ARG;
52 }
53
54 // Convert hex key to binary
55 size_t hex_len = strlen(key_hex);
56 if (hex_len == 0 || hex_len % 2 != 0)
57 {
58 ESP_LOGE(TAG, "Invalid hex key length: %zu", hex_len);
59 return ESP_ERR_INVALID_ARG;
60 }
61
62 size_t binary_key_len = hex_len / 2;
63 uint8_t *binary_key = malloc(binary_key_len);
64 if (binary_key == NULL)
65 {
66 ESP_LOGE(TAG, "Failed to allocate memory for binary key");
67 return ESP_ERR_NO_MEM;
68 }
69
70 esp_err_t ret = hmac_hex_to_binary(key_hex, binary_key, binary_key_len, NULL);
71 if (ret != ESP_OK)
72 {
73 ESP_LOGE(TAG, "Failed to convert hex key to binary");
74 free(binary_key);
75 return ret;
76 }
77
78 // Compute HMAC with binary key
79 ret = hmac_sha256_compute(binary_key, binary_key_len, data, data_len, hmac_out);
80
81 // Clear and free binary key
82 memset(binary_key, 0, binary_key_len);
83 free(binary_key);
84
85 return ret;
86}

88esp_err_t hmac_sha256_verify(const uint8_t *key, size_t key_len,
89 const uint8_t *data, size_t data_len,
90 const uint8_t *expected_hmac)
91{
92 if (expected_hmac == NULL)
93 {
94 ESP_LOGE(TAG, "Invalid parameters for HMAC verification");
95 return ESP_ERR_INVALID_ARG;
96 }
97
98 uint8_t computed_hmac[32];
99 esp_err_t ret = hmac_sha256_compute(key, key_len, data, data_len, computed_hmac);
100 if (ret != ESP_OK)
101 {
102 return ret;
103 }
104
105 // Compare HMACs using constant-time comparison to prevent timing attacks
106 int result = 0;
107 for (int i = 0; i < 32; i++)
108 {
109 result |= (computed_hmac[i] ^ expected_hmac[i]);
110 }
111
112 // Clear computed HMAC
113 memset(computed_hmac, 0, sizeof(computed_hmac));
114
115 if (result == 0)
116 {
117 ESP_LOGD(TAG, "HMAC verification successful");
118 return ESP_OK;
119 }
120 else
121 {
122 ESP_LOGW(TAG, "HMAC verification failed");
123 return ESP_ERR_INVALID_CRC;
124 }
125}

127esp_err_t hmac_to_hex_string(const uint8_t *hmac_bytes, char *hex_out)
128{
129 if (hmac_bytes == NULL || hex_out == NULL)

144esp_err_t hmac_hex_to_binary(const char *hex_string, uint8_t *binary_out,
145 size_t binary_out_size, size_t *actual_len)
146{
147 if (hex_string == NULL || binary_out == NULL)
148 {
149 ESP_LOGE(TAG, "Invalid parameters for hex to binary conversion");
150 return ESP_ERR_INVALID_ARG;
151 }
152
153 size_t hex_len = strlen(hex_string);
154 if (hex_len == 0 || hex_len % 2 != 0)
155 {
156 ESP_LOGE(TAG, "Invalid hex string length: %zu", hex_len);
157 return ESP_ERR_INVALID_ARG;
158 }
159
160 size_t binary_len = hex_len / 2;
161 if (binary_len > binary_out_size)
162 {
163 ESP_LOGE(TAG, "Output buffer too small: need %zu bytes, have %zu",
164 binary_len, binary_out_size);
165 return ESP_ERR_INVALID_SIZE;
166 }
167
168 for (size_t i = 0; i < binary_len; i++)
169 {
170 char hex_byte[3] = {hex_string[i * 2], hex_string[i * 2 + 1], '\0'};
171
172 // Validate hex characters
173 if (!isxdigit((unsigned char)hex_byte[0]) || !isxdigit((unsigned char)hex_byte[1]))
174 {
175 ESP_LOGE(TAG, "Invalid hex character in string at position %zu", i * 2);
176 return ESP_ERR_INVALID_ARG;
177 }
178
179 // Convert hex byte to binary
180 char *endptr;
181 long value = strtol(hex_byte, &endptr, 16);
182 if (*endptr != '\0')
183 {
184 ESP_LOGE(TAG, "Failed to convert hex byte: %s", hex_byte);
185 return ESP_ERR_INVALID_ARG;
186 }
187
188 binary_out[i] = (uint8_t)value;
189 }
190
191 if (actual_len != NULL)
192 {
193 *actual_len = binary_len;
194 }
195
196 ESP_LOGD(TAG, "Converted %zu hex characters to %zu binary bytes", hex_len, binary_len);
197 return ESP_OK;
198}

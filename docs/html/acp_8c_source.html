97acp_result_t acp_encode_frame(
98 uint8_t type,
99 uint8_t flags,
100 const uint8_t *payload,
101 size_t payload_len,
102 acp_session_t *session,
103 uint8_t *output,
104 size_t *output_len)
105{
106 /* Parameter validation */
107 if (payload == NULL && payload_len > 0)
108 {
109 return ACP_ERR_INVALID_PARAM;
110 }
111 if (output == NULL || output_len == NULL)
112 {
113 return ACP_ERR_INVALID_PARAM;
114 }
115 if (payload_len > ACP_MAX_PAYLOAD_SIZE)
116 {
117 return ACP_ERR_PAYLOAD_TOO_LARGE;
118 }
119 if (!acp_is_valid_frame_type(type))
120 {
121 return ACP_ERR_INVALID_TYPE;
122 }
123
124 /* Check authentication requirements */
125 if (acp_frame_requires_auth(type) && !(flags & ACP_FLAG_AUTHENTICATED))
126 {
127 return ACP_ERR_AUTH_REQUIRED;
128 }
129 if ((flags & ACP_FLAG_AUTHENTICATED) && session == NULL)
130 {
131 return ACP_ERR_SESSION_NOT_INIT;
132 }
133
134 /* Construct frame structure */
135 acp_frame_t frame = {0};
136 frame.version = ACP_PROTOCOL_VERSION;
137 frame.type = type;
138 frame.flags = flags;
139 frame.length = (uint16_t)payload_len;
140
141 /* Set sequence number for authenticated frames */
142 if (flags & ACP_FLAG_AUTHENTICATED)
143 {
144 frame.sequence = session->next_sequence;
145 }
146
147 /* Copy payload */
148 if (payload_len > 0)
149 {
150 memcpy(frame.payload, payload, payload_len);
151 }
152
153 /* Encode the frame using the framer */
154 size_t frame_size;
155 int result = acp_frame_encode(&frame, output, *output_len, &frame_size);
156 if (result != ACP_OK)
157 {
158 return result;
159 }
160
161 /* For authenticated frames, append HMAC tag */
162 if (flags & ACP_FLAG_AUTHENTICATED)
163 {
164 /* Check if we have space for HMAC tag */
165 if (*output_len < frame_size + ACP_HMAC_TAG_LEN)
166 {
167 *output_len = frame_size + ACP_HMAC_TAG_LEN;
168 return ACP_ERR_BUFFER_TOO_SMALL;
169 }
170
171 /* Calculate HMAC over the encoded frame (excluding delimiters) */
172 uint8_t hmac_tag[32]; /* Full SHA-256 output */
173 acp_hmac_sha256(session->key, ACP_KEY_SIZE,
174 output + 1, frame_size - 2, /* Skip delimiters */
175 hmac_tag);
176
177 /* Append truncated HMAC tag after the complete frame */
178 memcpy(output + frame_size, hmac_tag, ACP_HMAC_TAG_LEN);
179 frame_size += ACP_HMAC_TAG_LEN;
180
181 /* Update session sequence number */
182 session->next_sequence++;
183 }
184
185 *output_len = frame_size;
186 return ACP_OK;
187}

192acp_result_t acp_decode_frame(
193 const uint8_t *input,
194 size_t input_len,
195 acp_frame_t *frame,
196 size_t *consumed,
197 acp_session_t *session)
198{
199 /* Parameter validation */
200 if (input == NULL || frame == NULL || consumed == NULL)
201 {
202 return ACP_ERR_INVALID_PARAM;
203 }
204 if (input_len == 0)
205 {
206 return ACP_ERR_NEED_MORE_DATA;
207 }
208
209 *consumed = 0;
210 memset(frame, 0, sizeof(*frame));
211
212 /* For authenticated frames, we need to handle HMAC verification */
213 /* First, try to find frame boundaries to determine if frame is authenticated */
214
215 /* Find frame start delimiter */
216 if (input[0] != ACP_COBS_DELIMITER)
217 {
218 return ACP_ERR_MALFORMED_FRAME;
219 }
220
221 /* Find frame end delimiter */
222 size_t frame_end = 0;
223 for (size_t i = 1; i < input_len; i++)
224 {
225 if (input[i] == ACP_COBS_DELIMITER)
226 {
227 frame_end = i;
228 break;
229 }
230 }
231
232 if (frame_end == 0)
233 {
234 return ACP_ERR_NEED_MORE_DATA;
235 }
236
237 /* Try to decode the frame without HMAC first to check if it's authenticated */
238 acp_frame_t temp_frame;
239 size_t frame_consumed;
240 int result = acp_frame_decode(input, frame_end + 1, &temp_frame, &frame_consumed);
241 if (result != ACP_OK)
242 {
243 return result;
244 }
245
246 /* If frame is authenticated, verify HMAC */
247 if (temp_frame.flags & ACP_FLAG_AUTHENTICATED)
248 {
249 if (session == NULL || !session->initialized)
250 {
251 return ACP_ERR_SESSION_NOT_INIT;
252 }
253
254 /* Check if we have HMAC tag after the frame */
255 size_t total_size = frame_consumed + ACP_HMAC_TAG_LEN;
256 if (input_len < total_size)
257 {
258 return ACP_ERR_NEED_MORE_DATA;
259 }
260
261 /* Verify HMAC over the encoded frame (excluding delimiters) */
262 uint8_t expected_hmac[32];
263 acp_hmac_sha256(session->key, ACP_KEY_SIZE,
264 input + 1, frame_consumed - 2, /* Skip delimiters */
265 expected_hmac);
266
267 /* Compare with received HMAC tag (constant-time) */
268 const uint8_t *received_hmac = input + frame_consumed;
269 if (acp_crypto_memcmp_ct(expected_hmac, received_hmac, ACP_HMAC_TAG_LEN) != 0)
270 {
271 return ACP_ERR_AUTH_FAILED;
272 }
273
274 /* Verify sequence number for replay protection */
275 if (temp_frame.sequence <= session->last_accepted_seq)
276 {
277 return ACP_ERR_REPLAY;
278 }
279
280 /* Update session state */
281 session->last_accepted_seq = temp_frame.sequence;
282 *consumed = total_size;
283 }
284 else
285 {
286 /* Unauthenticated frame */
287 *consumed = frame_consumed;
288
289 /* Enforce authentication policy for command frames */
290 if (acp_frame_requires_auth(temp_frame.type))
291 {
292 return ACP_ERR_AUTH_REQUIRED;
293 }
294 }
295
296 /* Copy the decoded frame */
297 *frame = temp_frame;
298 return ACP_OK;
299}

351uint16_t acp_htons(uint16_t host_val)
352{
353 /* Check if we're on a little-endian system */
354 static const uint16_t test = 0x0001;
355 const uint8_t *test_bytes = (const uint8_t *)&test;
356
357 if (test_bytes[0] == 0x01)
358 {
359 /* Little-endian: swap bytes */
360 return ((host_val & 0xFF) << 8) | ((host_val >> 8) & 0xFF);
361 }
362 else
363 {
364 /* Big-endian: no conversion needed */
365 return host_val;
366 }
367}

381uint32_t acp_htonl(uint32_t host_val)
382{
383 /* Check if we're on a little-endian system */
384 static const uint16_t test = 0x0001;
385 const uint8_t *test_bytes = (const uint8_t *)&test;
386
387 if (test_bytes[0] == 0x01)
388 {
389 /* Little-endian: swap bytes */
390 return ((host_val & 0x000000FFU) << 24) |
391 ((host_val & 0x0000FF00U) << 8) |
392 ((host_val & 0x00FF0000U) >> 8) |
393 ((host_val & 0xFF000000U) >> 24);
394 }
395 else
396 {
397 /* Big-endian: no conversion needed */
398 return host_val;
399 }
400}

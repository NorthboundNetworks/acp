594static void writeIntraframe(void)
595{
596 blackboxMainState_t *blackboxCurrent = blackboxHistory[0];
597
598 blackboxWrite('I');
599
600 blackboxWriteUnsignedVB(blackboxIteration);
601 blackboxWriteUnsignedVB(blackboxCurrent->time);
602
603 if (testBlackboxCondition(CONDITION(PID))) {
604 blackboxWriteSignedVBArray(blackboxCurrent->axisPID_P, XYZ_AXIS_COUNT);
605 blackboxWriteSignedVBArray(blackboxCurrent->axisPID_I, XYZ_AXIS_COUNT);
606
607 // Don't bother writing the current D term if the corresponding PID setting is zero
608 for (int x = 0; x < XYZ_AXIS_COUNT; x++) {
609 if (testBlackboxCondition(CONDITION(NONZERO_PID_D_0) + x)) {
610 blackboxWriteSignedVB(blackboxCurrent->axisPID_D[x]);
611 }
612 }
613
614 blackboxWriteSignedVBArray(blackboxCurrent->axisPID_F, XYZ_AXIS_COUNT);
615 }
616
617 if (testBlackboxCondition(CONDITION(RC_COMMANDS))) {
618 // Write roll, pitch and yaw first:
619 blackboxWriteSigned16VBArray(blackboxCurrent->rcCommand, 3);
620
621 /*
622 * Write the throttle separately from the rest of the RC data as it's unsigned.
623 * Throttle lies in range [PWM_RANGE_MIN..PWM_RANGE_MAX]:
624 */
625 blackboxWriteUnsignedVB(blackboxCurrent->rcCommand[THROTTLE]);
626 }
627
628 if (testBlackboxCondition(CONDITION(SETPOINT))) {
629 // Write setpoint roll, pitch, yaw, and throttle
630 blackboxWriteSigned16VBArray(blackboxCurrent->setpoint, 4);
631 }
632
633 if (testBlackboxCondition(CONDITION(VBAT))) {
634 /*
635 * Our voltage is expected to decrease over the course of the flight, so store our difference from
636 * the reference:
637 *
638 * Write 14 bits even if the number is negative (which would otherwise result in 32 bits)
639 */
640 blackboxWriteUnsignedVB((vbatReference - blackboxCurrent->vbatLatest) & 0x3FFF);
641 }
642
643 if (testBlackboxCondition(CONDITION(AMPERAGE_ADC))) {
644 // 12bit value directly from ADC
645 blackboxWriteSignedVB(blackboxCurrent->amperageLatest);
646 }
647
648#ifdef USE_MAG
649 if (testBlackboxCondition(CONDITION(MAG))) {
650 blackboxWriteSigned16VBArray(blackboxCurrent->magADC, XYZ_AXIS_COUNT);
651 }
652#endif
653
654#ifdef USE_BARO
655 if (testBlackboxCondition(CONDITION(BARO))) {
656 blackboxWriteSignedVB(blackboxCurrent->baroAlt);
657 }
658#endif
659
660#ifdef USE_RANGEFINDER
661 if (testBlackboxCondition(CONDITION(RANGEFINDER))) {
662 blackboxWriteSignedVB(blackboxCurrent->surfaceRaw);
663 }
664#endif
665
666 if (testBlackboxCondition(CONDITION(RSSI))) {
667 blackboxWriteUnsignedVB(blackboxCurrent->rssi);
668 }
669
670 if (testBlackboxCondition(CONDITION(GYRO))) {
671 blackboxWriteSigned16VBArray(blackboxCurrent->gyroADC, XYZ_AXIS_COUNT);
672 }
673
674 if (testBlackboxCondition(CONDITION(GYROUNFILT))) {
675 blackboxWriteSigned16VBArray(blackboxCurrent->gyroUnfilt, XYZ_AXIS_COUNT);
676 }
677
678 if (testBlackboxCondition(CONDITION(ACC))) {
679 blackboxWriteSigned16VBArray(blackboxCurrent->accADC, XYZ_AXIS_COUNT);
680 }
681
682 if (testBlackboxCondition(CONDITION(DEBUG_LOG))) {
683 blackboxWriteSigned16VBArray(blackboxCurrent->debug, DEBUG16_VALUE_COUNT);
684 }
685
686 if (isFieldEnabled(FIELD_SELECT(MOTOR))) {
687 //Motors can be below minimum output when disarmed, but that doesn't happen much
688 blackboxWriteUnsignedVB(blackboxCurrent->motor[0] - getMotorOutputLow());
689
690 //Motors tend to be similar to each other so use the first motor's value as a predictor of the others
691 const int motorCount = getMotorCount();
692 for (int x = 1; x < motorCount; x++) {
693 blackboxWriteSignedVB(blackboxCurrent->motor[x] - blackboxCurrent->motor[0]);
694 }
695
696 if (testBlackboxCondition(FLIGHT_LOG_FIELD_CONDITION_TRICOPTER)) {
697 //Assume the tail spends most of its time around the center
698 blackboxWriteSignedVB(blackboxCurrent->servo[5] - 1500);
699 }
700 }
701
702#ifdef USE_DSHOT_TELEMETRY
703 if (isFieldEnabled(FIELD_SELECT(RPM))) {
704 const int motorCount = getMotorCount();
705 for (int x = 0; x < motorCount; x++) {
706 if (testBlackboxCondition(CONDITION(MOTOR_1_HAS_RPM) + x)) {
707 blackboxWriteUnsignedVB(blackboxCurrent->erpm[x]);
708 }
709 }
710 }
711#endif
712
713 //Rotate our history buffers:
714
715 //The current state becomes the new "before" state
716 blackboxHistory[1] = blackboxHistory[0];
717 //And since we have no other history, we also use it for the "before, before" state
718 blackboxHistory[2] = blackboxHistory[0];
719 //And advance the current state over to a blank space ready to be filled
720 blackboxHistory[0] = ((blackboxHistory[0] - blackboxHistoryRing + 1) % 3) + blackboxHistoryRing;
721
722 blackboxLoggedAnyFrames = true;
723}

725static void blackboxWriteMainStateArrayUsingAveragePredictor(int arrOffsetInHistory, int count)
726{
727 int16_t *curr = (int16_t*) ((char*) (blackboxHistory[0]) + arrOffsetInHistory);
728 int16_t *prev1 = (int16_t*) ((char*) (blackboxHistory[1]) + arrOffsetInHistory);
729 int16_t *prev2 = (int16_t*) ((char*) (blackboxHistory[2]) + arrOffsetInHistory);
730
731 for (int i = 0; i < count; i++) {
732 // Predictor is the average of the previous two history states
733 int32_t predictor = (prev1[i] + prev2[i]) / 2;
734
735 blackboxWriteSignedVB(curr[i] - predictor);
736 }
737}

739static void writeInterframe(void)
740{
741 blackboxMainState_t *blackboxCurrent = blackboxHistory[0];
742 blackboxMainState_t *blackboxLast = blackboxHistory[1];
743
744 blackboxWrite('P');
745
746 //No need to store iteration count since its delta is always 1
747
748 /*
749 * Since the difference between the difference between successive times will be nearly zero (due to consistent
750 * looptime spacing), use second-order differences.
751 */
752 blackboxWriteSignedVB((int32_t) (blackboxHistory[0]->time - 2 * blackboxHistory[1]->time + blackboxHistory[2]->time));
753
754 int32_t deltas[8];
755 int32_t setpointDeltas[4];
756
757 if (testBlackboxCondition(CONDITION(PID))) {
758 arraySubInt32(deltas, blackboxCurrent->axisPID_P, blackboxLast->axisPID_P, XYZ_AXIS_COUNT);
759 blackboxWriteSignedVBArray(deltas, XYZ_AXIS_COUNT);
760
761 /*
762 * The PID I field changes very slowly, most of the time +-2, so use an encoding
763 * that can pack all three fields into one byte in that situation.
764 */
765 arraySubInt32(deltas, blackboxCurrent->axisPID_I, blackboxLast->axisPID_I, XYZ_AXIS_COUNT);
766 blackboxWriteTag2_3S32(deltas);
767
768 /*
769 * The PID D term is frequently set to zero for yaw, which makes the result from the calculation
770 * always zero. So don't bother recording D results when PID D terms are zero.
771 */
772 for (int x = 0; x < XYZ_AXIS_COUNT; x++) {
773 if (testBlackboxCondition(FLIGHT_LOG_FIELD_CONDITION_NONZERO_PID_D_0 + x)) {
774 blackboxWriteSignedVB(blackboxCurrent->axisPID_D[x] - blackboxLast->axisPID_D[x]);
775 }
776 }
777
778 arraySubInt32(deltas, blackboxCurrent->axisPID_F, blackboxLast->axisPID_F, XYZ_AXIS_COUNT);
779 blackboxWriteSignedVBArray(deltas, XYZ_AXIS_COUNT);
780 }
781
782 /*
783 * RC tends to stay the same or fairly small for many frames at a time, so use an encoding that
784 * can pack multiple values per byte:
785 */
786 for (int x = 0; x < 4; x++) {
787 deltas[x] = blackboxCurrent->rcCommand[x] - blackboxLast->rcCommand[x];
788 setpointDeltas[x] = blackboxCurrent->setpoint[x] - blackboxLast->setpoint[x];
789 }
790
791 if (testBlackboxCondition(CONDITION(RC_COMMANDS))) {
792 blackboxWriteTag8_4S16(deltas);
793 }
794 if (testBlackboxCondition(CONDITION(SETPOINT))) {
795 blackboxWriteTag8_4S16(setpointDeltas);
796 }
797
798 //Check for sensors that are updated periodically (so deltas are normally zero)
799 int optionalFieldCount = 0;
800
801 if (testBlackboxCondition(CONDITION(VBAT))) {
802 deltas[optionalFieldCount++] = (int32_t) blackboxCurrent->vbatLatest - blackboxLast->vbatLatest;
803 }
804
805 if (testBlackboxCondition(CONDITION(AMPERAGE_ADC))) {
806 deltas[optionalFieldCount++] = blackboxCurrent->amperageLatest - blackboxLast->amperageLatest;
807 }
808
809#ifdef USE_MAG
810 if (testBlackboxCondition(CONDITION(MAG))) {
811 for (int x = 0; x < XYZ_AXIS_COUNT; x++) {
812 deltas[optionalFieldCount++] = blackboxCurrent->magADC[x] - blackboxLast->magADC[x];
813 }
814 }
815#endif
816
817#ifdef USE_BARO
818 if (testBlackboxCondition(CONDITION(BARO))) {
819 deltas[optionalFieldCount++] = blackboxCurrent->baroAlt - blackboxLast->baroAlt;
820 }
821#endif
822
823#ifdef USE_RANGEFINDER
824 if (testBlackboxCondition(CONDITION(RANGEFINDER))) {
825 deltas[optionalFieldCount++] = blackboxCurrent->surfaceRaw - blackboxLast->surfaceRaw;
826 }
827#endif
828
829 if (testBlackboxCondition(CONDITION(RSSI))) {
830 deltas[optionalFieldCount++] = (int32_t) blackboxCurrent->rssi - blackboxLast->rssi;
831 }
832
833 blackboxWriteTag8_8SVB(deltas, optionalFieldCount);
834
835 //Since gyros, accs and motors are noisy, base their predictions on the average of the history:
836 if (testBlackboxCondition(CONDITION(GYRO))) {
837 blackboxWriteMainStateArrayUsingAveragePredictor(offsetof(blackboxMainState_t, gyroADC), XYZ_AXIS_COUNT);
838 }
839 if (testBlackboxCondition(CONDITION(GYROUNFILT))) {
840 blackboxWriteMainStateArrayUsingAveragePredictor(offsetof(blackboxMainState_t, gyroUnfilt), XYZ_AXIS_COUNT);
841 }
842 if (testBlackboxCondition(CONDITION(ACC))) {
843 blackboxWriteMainStateArrayUsingAveragePredictor(offsetof(blackboxMainState_t, accADC), XYZ_AXIS_COUNT);
844 }
845 if (testBlackboxCondition(CONDITION(DEBUG_LOG))) {
846 blackboxWriteMainStateArrayUsingAveragePredictor(offsetof(blackboxMainState_t, debug), DEBUG16_VALUE_COUNT);
847 }
848
849 if (isFieldEnabled(FIELD_SELECT(MOTOR))) {
850 blackboxWriteMainStateArrayUsingAveragePredictor(offsetof(blackboxMainState_t, motor), getMotorCount());
851
852 if (testBlackboxCondition(FLIGHT_LOG_FIELD_CONDITION_TRICOPTER)) {
853 blackboxWriteSignedVB(blackboxCurrent->servo[5] - blackboxLast->servo[5]);
854 }
855 }
856#ifdef USE_DSHOT_TELEMETRY
857 if (isFieldEnabled(FIELD_SELECT(RPM))) {
858 const int motorCount = getMotorCount();
859 for (int x = 0; x < motorCount; x++) {
860 if (testBlackboxCondition(CONDITION(MOTOR_1_HAS_RPM) + x)) {
861 blackboxWriteSignedVB(blackboxCurrent->erpm[x] - blackboxLast->erpm[x]);
862 }
863 }
864 }
865#endif
866
867 //Rotate our history buffers
868 blackboxHistory[2] = blackboxHistory[1];
869 blackboxHistory[1] = blackboxHistory[0];
870 blackboxHistory[0] = ((blackboxHistory[0] - blackboxHistoryRing + 1) % 3) + blackboxHistoryRing;
871
872 blackboxLoggedAnyFrames = true;
873}

875/* Write the contents of the global "slowHistory" to the log as an "S" frame. Because this data is logged so
876 * infrequently, delta updates are not reasonable, so we log independent frames. */
877static void writeSlowFrame(void)
878{
879 int32_t values[3];
880
881 blackboxWrite('S');
882
883 blackboxWriteUnsignedVB(slowHistory.flightModeFlags);
884 blackboxWriteUnsignedVB(slowHistory.stateFlags);
885
886 /*
887 * Most of the time these three values will be able to pack into one byte for us:
888 */
889 values[0] = slowHistory.failsafePhase;
890 values[1] = slowHistory.rxSignalReceived ? 1 : 0;
891 values[2] = slowHistory.rxFlightChannelsValid ? 1 : 0;
892 blackboxWriteTag2_3S32(values);
893
894 blackboxSlowFrameIterationTimer = 0;
895}

915STATIC_UNIT_TESTED bool writeSlowFrameIfNeeded(void)
916{
917 // Write the slow frame peridocially so it can be recovered if we ever lose sync
918 bool shouldWrite = blackboxSlowFrameIterationTimer >= blackboxSInterval;
919
920 if (shouldWrite) {
921 loadSlowState(&slowHistory);
922 } else {
923 blackboxSlowState_t newSlowState;
924
925 loadSlowState(&newSlowState);
926
927 // Only write a slow frame if it was different from the previous state
928 if (memcmp(&newSlowState, &slowHistory, sizeof(slowHistory)) != 0) {
929 // Use the new state as our new history
930 memcpy(&slowHistory, &newSlowState, sizeof(slowHistory));
931 shouldWrite = true;
932 }
933 }
934
935 if (shouldWrite) {
936 writeSlowFrame();
937 }
938 return shouldWrite;
939}

1216static bool sendFieldDefinition(char mainFrameChar, char deltaFrameChar, const void *fieldDefinitions,
1217 const void *secondFieldDefinition, int fieldCount, const uint8_t *conditions, const uint8_t *secondCondition)
1218{
1219 const blackboxFieldDefinition_t *def;
1220 unsigned int headerCount;
1221 static bool needComma = false;
1222 size_t definitionStride = (char*) secondFieldDefinition - (char*) fieldDefinitions;
1223 size_t conditionsStride = (char*) secondCondition - (char*) conditions;
1224
1225 if (deltaFrameChar) {
1226 headerCount = BLACKBOX_DELTA_FIELD_HEADER_COUNT;
1227 } else {
1228 headerCount = BLACKBOX_SIMPLE_FIELD_HEADER_COUNT;
1229 }
1230
1231 /*
1232 * We're chunking up the header data so we don't exceed our datarate. So we'll be called multiple times to transmit
1233 * the whole header.
1234 */
1235
1236 // On our first call we need to print the name of the header and a colon
1237 if (xmitState.u.fieldIndex == -1) {
1238 if (xmitState.headerIndex >= headerCount) {
1239 return false; //Someone probably called us again after we had already completed transmission
1240 }
1241
1242 uint32_t charsToBeWritten = strlen("H Field x :") + strlen(blackboxFieldHeaderNames[xmitState.headerIndex]);
1243
1244 if (blackboxDeviceReserveBufferSpace(charsToBeWritten) != BLACKBOX_RESERVE_SUCCESS) {
1245 return true; // Try again later
1246 }
1247
1248 blackboxHeaderBudget -= blackboxPrintf("H Field %c %s:", xmitState.headerIndex >= BLACKBOX_SIMPLE_FIELD_HEADER_COUNT ? deltaFrameChar : mainFrameChar, blackboxFieldHeaderNames[xmitState.headerIndex]);
1249
1250 xmitState.u.fieldIndex++;
1251 needComma = false;
1252 }
1253
1254 // The longest we expect an integer to be as a string:
1255 const uint32_t LONGEST_INTEGER_STRLEN = 2;
1256
1257 for (; xmitState.u.fieldIndex < fieldCount; xmitState.u.fieldIndex++) {
1258 def = (const blackboxFieldDefinition_t*) ((const char*)fieldDefinitions + definitionStride * xmitState.u.fieldIndex);
1259
1260 if (!conditions || testBlackboxCondition(conditions[conditionsStride * xmitState.u.fieldIndex])) {
1261 // First (over)estimate the length of the string we want to print
1262
1263 int32_t bytesToWrite = 1; // Leading comma
1264
1265 // The first header is a field name
1266 if (xmitState.headerIndex == 0) {
1267 bytesToWrite += strlen(def->name) + strlen("[]") + LONGEST_INTEGER_STRLEN;
1268 } else {
1269 //The other headers are integers
1270 bytesToWrite += LONGEST_INTEGER_STRLEN;
1271 }
1272
1273 // Now perform the write if the buffer is large enough
1274 if (blackboxDeviceReserveBufferSpace(bytesToWrite) != BLACKBOX_RESERVE_SUCCESS) {
1275 // Ran out of space!
1276 return true;
1277 }
1278
1279 blackboxHeaderBudget -= bytesToWrite;
1280
1281 if (needComma) {
1282 blackboxWrite(',');
1283 } else {
1284 needComma = true;
1285 }
1286
1287 // The first header is a field name
1288 if (xmitState.headerIndex == 0) {
1289 blackboxWriteString(def->name);
1290
1291 // Do we need to print an index in brackets after the name?
1292 if (def->fieldNameIndex != -1) {
1293 blackboxPrintf("[%d]", def->fieldNameIndex);
1294 }
1295 } else {
1296 //The other headers are integers
1297 blackboxPrintf("%d", def->arr[xmitState.headerIndex - 1]);
1298 }
1299 }
1300 }
1301
1302 // Did we complete this line?
1303 if (xmitState.u.fieldIndex == fieldCount && blackboxDeviceReserveBufferSpace(1) == BLACKBOX_RESERVE_SUCCESS) {
1304 blackboxHeaderBudget--;
1305 blackboxWrite('\n');
1306 xmitState.headerIndex++;
1307 xmitState.u.fieldIndex = -1;
1308 }
1309
1310 return xmitState.headerIndex < headerCount;
1311}

1679void blackboxLogEvent(FlightLogEvent event, flightLogEventData_t *data)
1680{
1681 // Only allow events to be logged after headers have been written
1682 if (!(blackboxState == BLACKBOX_STATE_RUNNING || blackboxState == BLACKBOX_STATE_PAUSED)) {
1683 return;
1684 }
1685
1686 //Shared header for event frames
1687 blackboxWrite('E');
1688 blackboxWrite(event);
1689
1690 //Now serialize the data for this specific frame type
1691 switch (event) {
1692 case FLIGHT_LOG_EVENT_SYNC_BEEP:
1693 blackboxWriteUnsignedVB(data->syncBeep.time);
1694 break;
1695 case FLIGHT_LOG_EVENT_FLIGHTMODE: // New flightmode flags write
1696 blackboxWriteUnsignedVB(data->flightMode.flags);
1697 blackboxWriteUnsignedVB(data->flightMode.lastFlags);
1698 break;
1699 case FLIGHT_LOG_EVENT_DISARM:
1700 blackboxWriteUnsignedVB(data->disarm.reason);
1701 break;
1702 case FLIGHT_LOG_EVENT_INFLIGHT_ADJUSTMENT:
1703 if (data->inflightAdjustment.floatFlag) {
1704 blackboxWrite(data->inflightAdjustment.adjustmentFunction + FLIGHT_LOG_EVENT_INFLIGHT_ADJUSTMENT_FUNCTION_FLOAT_VALUE_FLAG);
1705 blackboxWriteFloat(data->inflightAdjustment.newFloatValue);
1706 } else {
1707 blackboxWrite(data->inflightAdjustment.adjustmentFunction);
1708 blackboxWriteSignedVB(data->inflightAdjustment.newValue);
1709 }
1710 break;
1711 case FLIGHT_LOG_EVENT_LOGGING_RESUME:
1712 blackboxWriteUnsignedVB(data->loggingResume.logIteration);
1713 blackboxWriteUnsignedVB(data->loggingResume.currentTime);
1714 break;
1715 case FLIGHT_LOG_EVENT_LOG_END:
1716 blackboxWriteString("End of log");
1717 blackboxWrite(0);
1718 break;
1719 default:
1720 break;
1721 }
1722}

1724/* If an arming beep has played since it was last logged, write the time of the arming beep to the log as a synchronization point */
1725static void blackboxCheckAndLogArmingBeep(void)

1777// Called once every FC loop in order to keep track of how many FC loop iterations have passed

1784 blackboxLoopIndex = 0;

1792// Called once every FC loop in order to log the current state
1793STATIC_UNIT_TESTED void blackboxLogIteration(timeUs_t currentTimeUs)
1794{
1795 // Write a keyframe every blackboxIInterval frames so we can resynchronise upon missing frames
1796 if (blackboxShouldLogIFrame()) {
1797 /*
1798 * Don't log a slow frame if the slow data didn't change ("I" frames are already large enough without adding
1799 * an additional item to write at the same time). Unless we're *only* logging "I" frames, then we have no choice.
1800 */
1801 if (blackboxIsOnlyLoggingIntraframes()) {
1802 writeSlowFrameIfNeeded();
1803 }
1804
1805 loadMainState(currentTimeUs);
1806 writeIntraframe();
1807 } else {
1808 blackboxCheckAndLogArmingBeep();
1809 blackboxCheckAndLogFlightMode(); // Check for FlightMode status change event
1810
1811 if (blackboxShouldLogPFrame()) {
1812 /*
1813 * We assume that slow frames are only interesting in that they aid the interpretation of the main data stream.
1814 * So only log slow frames during loop iterations where we log a main frame.
1815 */
1816 writeSlowFrameIfNeeded();
1817
1818 loadMainState(currentTimeUs);
1819 writeInterframe();
1820 }
1821#ifdef USE_GPS
1822 if (featureIsEnabled(FEATURE_GPS) && isFieldEnabled(FIELD_SELECT(GPS))) {
1823 if (blackboxShouldLogGpsHomeFrame()) {
1824 writeGPSHomeFrame();
1825 writeGPSFrame(currentTimeUs);
1826 } else if (gpsSol.numSat != gpsHistory.GPS_numSat
1827 || gpsSol.llh.lat != gpsHistory.GPS_coord[GPS_LATITUDE]
1828 || gpsSol.llh.lon != gpsHistory.GPS_coord[GPS_LONGITUDE]) {
1829 //We could check for velocity changes as well but I doubt it changes independent of position
1830 writeGPSFrame(currentTimeUs);
1831 }
1832 }
1833#endif
1834 }
1835
1836 //Flush every iteration so that our runtime variance is minimized
1837 blackboxDeviceFlush();
1838}

1843void blackboxUpdate(timeUs_t currentTimeUs)
1844{
1845 static BlackboxState cacheFlushNextState;
1846
1847 switch (blackboxState) {
1848 case BLACKBOX_STATE_STOPPED:
1849 if (ARMING_FLAG(ARMED)) {
1850 blackboxOpen();
1851 blackboxStart();
1852 }
1853#ifdef USE_FLASHFS
1854 if (IS_RC_MODE_ACTIVE(BOXBLACKBOXERASE)) {
1855 blackboxSetState(BLACKBOX_STATE_START_ERASE);
1856 }
1857#endif
1858 break;
1859 case BLACKBOX_STATE_PREPARE_LOG_FILE:
1860 if (blackboxDeviceBeginLog()) {
1861 blackboxSetState(BLACKBOX_STATE_SEND_HEADER);
1862 }
1863 break;
1864 case BLACKBOX_STATE_SEND_HEADER:
1865 blackboxReplenishHeaderBudget();
1866 //On entry of this state, xmitState.headerIndex is 0 and startTime is intialised
1867
1868 /*
1869 * Once the UART has had time to init, transmit the header in chunks so we don't overflow its transmit
1870 * buffer, overflow the OpenLog's buffer, or keep the main loop busy for too long.
1871 */
1872 if (millis() > xmitState.u.startTime + 100) {
1873 if (blackboxDeviceReserveBufferSpace(BLACKBOX_TARGET_HEADER_BUDGET_PER_ITERATION) == BLACKBOX_RESERVE_SUCCESS) {
1874 for (int i = 0; i < BLACKBOX_TARGET_HEADER_BUDGET_PER_ITERATION && blackboxHeader[xmitState.headerIndex] != '\0'; i++, xmitState.headerIndex++) {
1875 blackboxWrite(blackboxHeader[xmitState.headerIndex]);
1876 blackboxHeaderBudget--;
1877 }
1878 if (blackboxHeader[xmitState.headerIndex] == '\0') {
1879 blackboxSetState(BLACKBOX_STATE_SEND_MAIN_FIELD_HEADER);
1880 }
1881 }
1882 }
1883 break;
1884 case BLACKBOX_STATE_SEND_MAIN_FIELD_HEADER:
1885 blackboxReplenishHeaderBudget();
1886 //On entry of this state, xmitState.headerIndex is 0 and xmitState.u.fieldIndex is -1
1887 if (!sendFieldDefinition('I', 'P', blackboxMainFields, blackboxMainFields + 1, ARRAYLEN(blackboxMainFields),
1888 &blackboxMainFields[0].condition, &blackboxMainFields[1].condition)) {
1889#ifdef USE_GPS
1890 if (featureIsEnabled(FEATURE_GPS) && isFieldEnabled(FIELD_SELECT(GPS))) {
1891 blackboxSetState(BLACKBOX_STATE_SEND_GPS_H_HEADER);
1892 } else
1893#endif
1894 blackboxSetState(BLACKBOX_STATE_SEND_SLOW_HEADER);
1895 }
1896 break;
1897#ifdef USE_GPS
1898 case BLACKBOX_STATE_SEND_GPS_H_HEADER:
1899 blackboxReplenishHeaderBudget();

2020 }
2021 }

64static int gimbal_cmd_mode(int argc, char **argv)
65{
66 if (argc < 2)
67 {
68 printf("Usage: gimbal mode <fixed|follow|sweep>\n");
69 printf(" fixed - Fixed attitude (manual control)\n");
70 printf(" follow - Follow vehicle heading\n");
71 printf(" sweep - Autonomous search sweep\n");
72 return -1;
73 }
74
75 if (!gimbal_is_initialized())
76 {
77 printf("ERROR: Gimbal service not initialized\n");
78 return -1;
79 }
80
81 const char *mode_str = argv[1];
82 gimbal_mode_t mode;
83
84 if (strcmp(mode_str, "fixed") == 0)
85 {
86 mode = GIMBAL_MODE_FIXED;
87 }
88 else if (strcmp(mode_str, "follow") == 0)
89 {
90 mode = GIMBAL_MODE_FOLLOW_HEADING;
91 }
92 else if (strcmp(mode_str, "sweep") == 0)
93 {
94 mode = GIMBAL_MODE_SEARCH_SWEEP;
95 }
96 else
97 {
98 printf("ERROR: Invalid mode '%s'. Use: fixed, follow, or sweep\n", mode_str);
99 return -1;
100 }
101
102 esp_err_t ret = gimbal_set_mode(mode);
103 if (ret == ESP_OK)
104 {
105 printf("Gimbal mode set to: %s\n", mode_str);
106 }
107 else
108 {
109 printf("ERROR: Failed to set gimbal mode: %s\n", esp_err_to_name(ret));
110 return -1;
111 }
112
113 return 0;
114}

116static int gimbal_cmd_set(int argc, char **argv)
117{
118 if (argc < 3)
119 {
120 printf("Usage: gimbal set <pitch> <yaw>\n");
121 printf(" pitch: Pitch angle in degrees (-90 to +90)\n");
122 printf(" yaw: Yaw angle in degrees (-180 to +180)\n");
123 printf(" Example: gimbal set -30 45\n");
124 return -1;
125 }
126
127 if (!gimbal_is_initialized())
128 {
129 printf("ERROR: Gimbal service not initialized\n");
130 return -1;
131 }
132
133 // Parse pitch and yaw angles
134 float pitch = atof(argv[1]);
135 float yaw = atof(argv[2]);
136
137 gimbal_attitude_t attitude = {
138 .pitch = pitch,
139 .yaw = yaw,
140 .roll = 0.0f // Roll is typically 0 for gimbal applications
141 };
142
143 esp_err_t ret = gimbal_set_attitude(&attitude);
144 if (ret == ESP_OK)
145 {
146 printf("Gimbal attitude set: pitch=%.1f°, yaw=%.1f°\n", pitch, yaw);
147
148 // Switch to fixed mode to use the new attitude
149 gimbal_set_mode(GIMBAL_MODE_FIXED);
150 printf("Mode switched to FIXED for manual control\n");
151 }
152 else
153 {
154 printf("ERROR: Failed to set gimbal attitude: %s\n", esp_err_to_name(ret));
155 printf("Check angle limits: pitch (-90 to +90°), yaw (-180 to +180°)\n");
156 return -1;
157 }
158
159 return 0;
160}

162static int gimbal_cmd_sweep(int argc, char **argv)
163{
164 if (argc < 3)
165 {
166 printf("Usage: gimbal sweep <range> <rate> [center]\n");
167 printf(" range: Sweep range in degrees (1 to 180)\n");
168 printf(" rate: Sweep rate in degrees/sec (1 to 90)\n");
169 printf(" center: Center yaw position (optional, default 0)\n");
170 printf(" Example: gimbal sweep 135 10 0\n");
171 return -1;
172 }
173
174 if (!gimbal_is_initialized())
175 {
176 printf("ERROR: Gimbal service not initialized\n");
177 return -1;
178 }
179
180 // Parse sweep parameters
181 float range = atof(argv[1]);
182 float rate = atof(argv[2]);
183 float center = (argc > 3) ? atof(argv[3]) : 0.0f;
184
185 gimbal_sweep_config_t config = {
186 .sweep_range = range,
187 .sweep_rate = rate,
188 .center_yaw = center,
189 .enabled = true};
190
191 esp_err_t ret = gimbal_configure_sweep(&config);
192 if (ret == ESP_OK)
193 {
194 printf("Sweep configured: range=±%.1f°, rate=%.1f°/s, center=%.1f°\n",
195 range, rate, center);
196
197 // Optionally switch to sweep mode
198 printf("Use 'gimbal mode sweep' to start sweeping\n");
199 }
200 else
201 {
202 printf("ERROR: Failed to configure sweep: %s\n", esp_err_to_name(ret));
203 printf("Check parameters: range (1-180°), rate (1-90°/s)\n");
204 return -1;
205 }
206
207 return 0;
208}

210static int gimbal_cmd_status(int argc, char **argv)
211{
212 (void)argc;
213 (void)argv;
214
215 if (!gimbal_is_initialized())
216 {
217 printf("Gimbal Status: NOT INITIALIZED\n");
218 printf("ERROR: Gimbal service not available\n");
219 return -1;
220 }
221
222 // Get current status
223 gimbal_status_t status;
224 esp_err_t ret = gimbal_get_status(&status);
225 if (ret != ESP_OK)
226 {
227 printf("ERROR: Failed to get gimbal status: %s\n", esp_err_to_name(ret));
228 return -1;
229 }
230
231 // Get sweep configuration
232 gimbal_sweep_config_t sweep_config;
233 gimbal_get_sweep_config(&sweep_config);
234
235 // Display status
236 printf("Gimbal Status: INITIALIZED\n");
237
238 // Mode
239 const char *mode_names[] = {"FIXED", "FOLLOW_HEADING", "SEARCH_SWEEP"};
240 printf("Mode: %s\n", mode_names[status.mode]);
241
242 // Attitudes
243 printf("Current Attitude: pitch=%.1f°, yaw=%.1f°, roll=%.1f°\n",
244 status.current_attitude.pitch, status.current_attitude.yaw, status.current_attitude.roll);
245 printf("Target Attitude: pitch=%.1f°, yaw=%.1f°, roll=%.1f°\n",
246 status.target_attitude.pitch, status.target_attitude.yaw, status.target_attitude.roll);
247
248 // Movement status
249 printf("Movement: %s\n", status.is_moving ? "ACTIVE" : "STATIONARY");
250
251 // Fault status
252 if (status.fault_detected)
253 {
254 printf("Fault Status: FAULT DETECTED\n");
255 }
256 else
257 {
258 printf("Fault Status: OK\n");
259 }
260
261 // Sweep configuration
262 printf("Sweep Config: range=±%.1f°, rate=%.1f°/s, center=%.1f°, enabled=%s\n",
263 sweep_config.sweep_range, sweep_config.sweep_rate, sweep_config.center_yaw,
264 sweep_config.enabled ? "true" : "false");
265
266 // Last update
267 printf("Last Update: %lu ms ago\n",
268 (gimbal_get_time_ms() - status.last_update_ms));
269
270 // Available commands
271 printf("\nAvailable Commands:\n");
272 printf(" gimbal mode <fixed|follow|sweep> - Change operating mode\n");
273 printf(" gimbal set <pitch> <yaw> - Set attitude (fixed mode)\n");
274 printf(" gimbal sweep <range> <rate> - Configure sweep parameters\n");
275
276 return 0;
277}

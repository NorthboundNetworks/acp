144/* Static assertions to ensure wire header packing */
145#define ACP_STATIC_ASSERT(cond, msg) typedef char acp_static_assert_##__LINE__[(cond) ? 1 : -1]
146 ACP_STATIC_ASSERT(sizeof(acp_wire_header_base_t) == 6, "Base wire header must be exactly 6 bytes");
147
148#define ACP_STATIC_ASSERT2(cond, msg) typedef char acp_static_assert2_##__LINE__[(cond) ? 1 : -1]
149 ACP_STATIC_ASSERT2(sizeof(acp_wire_header_t) == 10, "Full wire header must be exactly 10 bytes");
150
156 static inline size_t acp_wire_header_size(uint8_t flags)
157 {
158 size_t size = sizeof(acp_wire_header_base_t);
159 if (flags & ACP_FLAG_AUTHENTICATED)
160 {
161 size += sizeof(uint32_t); /* Add sequence field */
162 }
163 return size;

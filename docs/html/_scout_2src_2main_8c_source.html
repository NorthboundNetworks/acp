47static void task_cli(void *arg)
48{
49 (void)arg;
50 ESP_LOGI(TAG, "CLI task started");
51 while (1)
52 {
53 char line[128];
54 if (cli_transport_poll_line(line, sizeof(line)))
55 {
56 ESP_LOGI(TAG, "CLI received: %s", line);
57 cli_core_process_line(line);
58 }
59 else
60 {
61 vTaskDelay(pdMS_TO_TICKS(25));
62 }
63 }
64}

66static void task_telemetry(void *arg)

82static void task_control(void *arg)
83{
84 (void)arg;
85 ESP_LOGI(TAG, "Control task started");
86 const TickType_t period = pdMS_TO_TICKS(10); // 100 Hz control loop (minimum 1 tick)
87
88 // Safety check: ensure period is never 0
89 if (period == 0)
90 {
91 ESP_LOGE(TAG, "Control task period is 0! Using minimum 1 tick.");
92 const TickType_t safe_period = 1;
93 TickType_t last = xTaskGetTickCount();
94 while (1)
95 {
96 rt_metrics_loop_start();
97
98 // Update CPU usage (periodic, internal rate limiting)
99 rt_metrics_update_cpu_usage();
100
101 // Update GPS data regularly
102 neo6m_t *gps = neo6m_get_instance();
103 if (gps && gps->initialized)
104 {
105 neo6m_update(gps);
106 }
107
108 rt_metrics_loop_end(1000); // 1000 Hz effective with 1 tick period
109 vTaskDelayUntil(&last, safe_period);
110 }
111 }
112
113 TickType_t last = xTaskGetTickCount();
114 TickType_t last_sensor_summary = last;
115 const TickType_t sensor_summary_interval = pdMS_TO_TICKS(SENSOR_SUMMARY_INTERVAL_MS);
116
117 while (1)
118 {
119 rt_metrics_loop_start();
120
121 // Update CPU usage (periodic, internal rate limiting)
122 rt_metrics_update_cpu_usage();
123
124 // Update performance guardrails
125 uint32_t timestamp_ms = xTaskGetTickCount() * portTICK_PERIOD_MS;
126 rt_guardrails_update(timestamp_ms);
127
128 // TODO: run safety gates and control updates
129
130 // Update GPS data regularly
131 neo6m_t *gps = neo6m_get_instance();
132 if (gps && gps->initialized)
133 {
134 neo6m_update(gps);
135 }
136
137#ifdef FEATURE_GIMBAL
138 // Update gimbal control (uses GPS heading if available)
139 float vehicle_heading = 0.0f;
140 if (gps && gps->initialized && gps->fix_quality > 0)
141 {
142 vehicle_heading = 0; // gps->course; // TODO: Fix GPS course field
143 }
144 // gimbal_update(vehicle_heading); // TODO: Implement gimbal update
145#endif
146
147#ifdef FEATURE_BLACKBOX
148 // Periodic sensor summary logging (every 2 seconds)
149 TickType_t now = xTaskGetTickCount();
150 if (now - last_sensor_summary >= sensor_summary_interval)
151 {
152 // blackbox_log_sensor_summary(); // TODO: Implement blackbox logging
153 last_sensor_summary = now;
154 }
155#endif
156
157 rt_metrics_loop_end(100); // 100 Hz target
158 vTaskDelayUntil(&last, period);
159 }
160}

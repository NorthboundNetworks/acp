87 void update(bool pressed) {
88 _new = false;
89
90 // Optimization for the most common case: nothing happening.
91 if (!pressed && _state == StateIdle) {
92 return;
93 }
94
95 // Only take the low word, works as long as update() is called
96 // at least every ~32s, and typical use-case is <50ms.
97 // Also note that wrapping during StateIdle doesn't really matter,
98 // as the time isn't used then.
99 unsigned int now = millis();
100 int diff = now - _lastTransition;
101
102 State next = StateIdle;
103 switch (_state) {
104 case StateIdle: next = _checkIdle(pressed, diff); break;
105 case StateDebounce: next = _checkDebounce(pressed, diff); break;
106 case StatePressed: next = _checkPressed(pressed, diff); break;
107 case StateClickUp: next = _checkClickUp(pressed, diff); break;
108 case StateClickIdle: next = _checkClickIdle(pressed, diff); break;
109 case StateSingleClick: next = _checkSingleClick(pressed, diff); break;
110 case StateDoubleClickDebounce: next = _checkDoubleClickDebounce(pressed, diff); break;
111 case StateDoubleClick: next = _checkDoubleClick(pressed, diff); break;
112 case StateLongClick: next = _checkLongClick(pressed, diff); break;
113 case StateOtherUp: next = _checkOtherUp(pressed, diff); break;
114 }
115
116 if (next != _state) {
117 // Mark current state transition to make 'diff' meaningful
118 _lastTransition = now;
119 // Enter next state
120 _state = next;
121 // Only mark this state 'new' for one iteration, to cause
122 // e.g. isClick() to return true until the next call to update()
123 _new = true;
124 }
125 }

245 State _checkIdle(bool pressed, int diff) {

251 State _checkDebounce(bool pressed, int diff) {
252 // If released in this state: it was a glitch
253 if (!pressed) {
254 return StateIdle;
255 }
256 if (diff >= _configPtr->debounceDecay) {
257 // Still pressed after debounce delay: real 'press'
258 return StatePressed;
259 }
260 return StateDebounce;
261 }

263 State _checkPressed(bool pressed, int diff) {
264 // If released, go wait for either a double-click, or
265 // to generate the actual SingleClick event,
266 // but first mark that the button is released.
267 if (!pressed) {
268 return StateClickUp;
269 }
270 // If pressed, keep waiting to see if it's a long click
271 if (diff >= _configPtr->longClickDelay) {
272 return StateLongClick;
273 }
274 return StatePressed;
275 }

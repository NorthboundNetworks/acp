139{
140
141 // Get session for authentication key
142 scp_session_t *session = scp_session_get_by_id(session_id);
143 if (!session)
144 {
145 ESP_LOGE(TAG, "Invalid session ID %lu for response", session_id);
146 return false;
147 }
148
149 // Build SCP frame
150 scp_frame_header_t header = {
151 .message_type = message_type,
152 .sequence_num = sequence_num,
153 .session_id = session_id,
154 .payload_size = payload_size,
155 .priority = SCP_PRIORITY_NORMAL,
156 .timestamp = gcs_gateway_get_timestamp_ms()};
157
158 // Encode message with authentication
159 size_t frame_size;
160 scp_result_t result = scp_framer_encode(&header, payload,
161 session->auth_key,
162 gateway_state.tx_buffer,
163 sizeof(gateway_state.tx_buffer),
164 &frame_size);
165
166 if (result != SCP_RESULT_OK)
167 {
168 ESP_LOGE(TAG, "Failed to encode response: %d", result);
169 gateway_state.stats.encoding_errors++;
170 return false;
171 }
172
173 // Send via transport (implementation depends on active transport)
174 // For now, we'll use UART as the primary transport
175 // TODO: Implement transport abstraction in later phases
176
177 ESP_LOGI(TAG, "Sending %s response (seq=%d, session=%lu, size=%d)",
178 scp_message_type_get_name(message_type), sequence_num, session_id, frame_size);
179
180 gateway_state.stats.responses_sent++;
181 return true;
182}

404bool gcs_gateway_validate_sequence(uint32_t session_id, uint8_t sequence_num)
405{
406 scp_session_t *session = scp_session_get_by_id(session_id);
407 if (!session)
408 {
409 // Invalid session - potential security threat
410 gcs_gateway_audit_security_violation(session_id, 2, 3); // Type 2 = invalid session, Severity 3 = high
411 gateway_state.stats.security_violations++;
412 return false;
413 }
414
415 // Check for replay attack (sequence number reuse or going backwards)
416 if (sequence_num <= session->last_seq_num)
417 {
418 ESP_LOGW(TAG, "Replay attack detected: session=%lu, recv_seq=%d, last_seq=%d",
419 session_id, sequence_num, session->last_seq_num);
420
421 // Audit replay attack attempt
422 gcs_gateway_audit_auth_failure(session_id, 2, sequence_num); // 2 = replay attack
423 gcs_gateway_audit_security_violation(session_id, 3, 3); // Type 3 = replay attack, Severity 3 = high
424 gateway_state.stats.security_violations++;
425 gateway_state.stats.auth_failure_audits++;
426
427 // Log replay attack to persistent storage for forensic analysis (T032)
428 char replay_details[128];
429 snprintf(replay_details, sizeof(replay_details),
430 "Sequence number %u <= last processed %u", sequence_num, session->last_sequence_num);
431 gcs_gateway_log_security_violation("REPLAY_ATTACK", session_id, replay_details, "critical");
432
433 return false;
434 }
435
436 return true;

646 ESP_LOGW(TAG, "ARM command rejected: insufficient role permissions");

650
651 // Check safety manager arming permissions

666 {
667 ESP_LOGW(TAG, "ARM command rejected: safety gates not satisfied");
668 gateway_state.stats.safety_rejections++;
669 return GCS_CMD_ERR_SAFETY_BLOCK;
670 }
671
672 // Transition to FLIGHT mode
673 scout_mode_t current_mode = mode_manager_get_mode();
674 if (current_mode != SCOUT_MODE_FLIGHT)
675 {
676 mode_manager_set_mode(SCOUT_MODE_FLIGHT);
677 ESP_LOGI(TAG, "Transitioned to FLIGHT mode for arming");
678 }
679
680 // Set flight mode if specified
681 if (cmd_control->target_flight_mode != FLIGHT_MODE_HOVER)

684 if (mode_result != MODE_CHANGE_OK)
685 {
686 ESP_LOGW(TAG, "Flight mode change failed: %s",
687 mode_manager_get_mode_change_error(mode_result));
688 gateway_state.stats.mode_change_failures++;
689 return GCS_CMD_ERR_MODE_UNAVAILABLE;
690 }
691 }
692
693 ESP_LOGI(TAG, "ARM command successful - drone is now armed");
694 gateway_state.stats.arm_commands++;
695
696 // Log to existing audit system (T027 - User Story 4)
697 char command_summary[128];
698 snprintf(command_summary, sizeof(command_summary),
699 "SCP_ARM: mode=%d, session=%lu, role=%s",
700 cmd_control->target_flight_mode, session->session_id,
701 scp_role_get_name(session->operator_role));
702 gcs_gateway_log_to_audit_system(session->session_id, SCP_MSG_CMD_CONTROL,
703 GCS_CMD_OK, command_summary);
704
705 return GCS_CMD_OK;
706}

771{

803 return GCS_CMD_ERR_SAFETY_BLOCK;
804 case MODE_CHANGE_REJECTED_SENSORS:
805 return GCS_CMD_ERR_SENSORS;
806 case MODE_CHANGE_REJECTED_CPU:
807 return GCS_CMD_ERR_SYSTEM;
808 default:
809 return GCS_CMD_ERR_INVALID_PARAM;
810 }
811 }
812
813 ESP_LOGI(TAG, "Mode change successful to flight mode %d", cmd_control->target_flight_mode);
814 gateway_state.stats.mode_changes++;
815
816 // Log to existing audit system (T027 - User Story 4)
817 char command_summary[128];
818 snprintf(command_summary, sizeof(command_summary),
819 "SCP_MODE_CHANGE: from=%d to=%d, session=%lu, role=%s",
820 mode_manager_get_flight_mode(), cmd_control->target_flight_mode,
821 session->session_id, scp_role_get_name(session->operator_role));
822 gcs_gateway_log_to_audit_system(session->session_id, SCP_MSG_CMD_CONTROL,
823 GCS_CMD_OK, command_summary);
824
825 return GCS_CMD_OK;
826}

836static gcs_command_result_t gcs_gateway_process_cmd_control(uint32_t session_id,
837 uint8_t sequence_num,
838 const uint8_t *payload,
839 size_t payload_size)
840{
841
842 uint32_t start_time = gcs_gateway_get_timestamp_ms();
843
844 // Validate payload size
845 if (payload_size < sizeof(scp_cmd_control_t))
846 {
847 ESP_LOGE(TAG, "CMD_CONTROL payload too small: %d < %d",
848 payload_size, sizeof(scp_cmd_control_t));
849 gateway_state.stats.protocol_errors++;
850 return GCS_CMD_ERR_INVALID_PARAM;
851 }
852
853 // Parse command
854 const scp_cmd_control_t *cmd_control = (const scp_cmd_control_t *)payload;
855
856 // Get session for authorization
857 scp_session_t *session = scp_session_get_by_id(session_id);
858 if (!session)
859 {
860 ESP_LOGE(TAG, "Invalid session ID %lu for CMD_CONTROL", session_id);
861 gateway_state.stats.session_errors++;
862 return GCS_CMD_ERR_INVALID_SESSION;
863 }
864
865 // Check session authorization for command
866 char error_reason[64] = {0};
867 if (!scp_session_authorize_command(session_id, SCP_CMD_CONTROL, error_reason))
868 {
869 ESP_LOGW(TAG, "Command authorization failed: %s", error_reason);
870 gateway_state.stats.authorization_failures++;
871
872 // T028 - Security event detection and reporting
873 gcs_gateway_audit_auth_failure(session_id, 1, sequence_num); // 1 = authorization failure
874 gcs_gateway_audit_security_violation(session_id, 1, 2); // Type 1 = unauthorized access, Severity 2 = medium
875 gateway_state.stats.security_violations++;
876 gateway_state.stats.auth_failure_audits++;
877
878 // Log authorization failure to persistent storage for forensic analysis (T032)
879 gcs_gateway_log_security_violation("AUTH_FAILURE", session_id, error_reason, "error");
880
881 return GCS_CMD_ERR_UNAUTHORIZED;
882 }
883
884 // Process command based on control action
885 gcs_command_result_t result;
886
887 switch (cmd_control->control_action)
888 {
889 case SCP_CONTROL_ARM:
890 result = gcs_gateway_process_arm_command(session, cmd_control, sequence_num);
891 break;
892
893 case SCP_CONTROL_DISARM:
894 result = gcs_gateway_process_disarm_command(session, cmd_control, sequence_num);
895 break;
896
897 case SCP_CONTROL_MODE_CHANGE:
898 result = gcs_gateway_process_mode_change(session, cmd_control, sequence_num);
899 break;
900
901 default:
902 ESP_LOGW(TAG, "Unknown control action: %d", cmd_control->control_action);
903 gateway_state.stats.protocol_errors++;
904 result = GCS_CMD_ERR_INVALID_PARAM;
905 break;
906 }
907
908 // Record processing time
909 uint32_t processing_time = gcs_gateway_get_timestamp_ms() - start_time;
910 gateway_state.stats.total_processing_time_ms += processing_time;
911 if (processing_time > gateway_state.stats.max_processing_time_ms)
912 {
913 gateway_state.stats.max_processing_time_ms = processing_time;
914 }
915
916 // Warn if processing exceeded target
917 if (processing_time > 100)
918 {
919 ESP_LOGW(TAG, "Command processing took %lu ms (target <100ms)", processing_time);
920 }
921
922 gateway_state.stats.commands_processed++;
923
924 ESP_LOGD(TAG, "CMD_CONTROL processed in %lu ms, result: %d", processing_time, result);
925
926 return result;
927}

937static void gcs_gateway_process_message(const gateway_message_t *message)
938{
939
940 uint32_t processing_start = gcs_gateway_get_timestamp_ms();
941
942 ESP_LOGD(TAG, "Processing %s message (session=%lu, seq=%d, size=%d)",
943 scp_message_type_get_name(message->message_type),
944 message->session_id, message->sequence_num, message->payload_size);
945
946 // Process message based on type
947 gcs_command_result_t cmd_result = GCS_CMD_OK;
948
949 switch (message->message_type)
950 {
951 case SCP_MSG_CMD_CONTROL:
952 cmd_result = gcs_gateway_process_cmd_control(message->session_id,
953 message->sequence_num,
954 message->payload,
955 message->payload_size);
956 break;
957
958 case SCP_MSG_CMD_GIMBAL:
959 // TODO: Implement in User Story 3
960 ESP_LOGW(TAG, "CMD_GIMBAL not yet implemented");
961 cmd_result = GCS_CMD_ERR_NOT_IMPLEMENTED;
962 break;
963
964 case SCP_MSG_CMD_CONFIG:
965 // TODO: Implement configuration commands
966 ESP_LOGW(TAG, "CMD_CONFIG not yet implemented");
967 cmd_result = GCS_CMD_ERR_NOT_IMPLEMENTED;
968 break;
969
970 default:
971 ESP_LOGW(TAG, "Unsupported message type: %d", message->message_type);
972 cmd_result = GCS_CMD_ERR_INVALID_PARAM;
973 break;
974 }
975
976 // Generate response based on command result
977 if (cmd_result == GCS_CMD_OK)
978 {
979 // Send ACK response
980 scp_ack_response_t ack = {
981 .command_type = message->message_type,
982 .result = SCP_ACK_SUCCESS,
983 .processing_time_ms = gcs_gateway_get_timestamp_ms() - processing_start};
984
985 gcs_gateway_send_response(message->session_id, message->sequence_num,
986 SCP_MSG_ACK_RESPONSE, &ack, sizeof(ack));
987
988 gateway_state.stats.successful_commands++;
989 }
990 else
991 {
992 // Send ERR response
993 scp_err_response_t error = {
994 .command_type = message->message_type,
995 .error_code = cmd_result,
996 .processing_time_ms = gcs_gateway_get_timestamp_ms() - processing_start};
997
998 // Add error description based on result
999 switch (cmd_result)
1000 {
1001 case GCS_CMD_ERR_UNAUTHORIZED:
1002 strncpy(error.error_description, "Insufficient role permissions",
1003 sizeof(error.error_description) - 1);
1004 break;
1005 case GCS_CMD_ERR_SAFETY_BLOCK:
1006 strncpy(error.error_description, safety_manager_get_arming_block_reason() ?: "Safety block",
1007 sizeof(error.error_description) - 1);
1008 break;
1009 case GCS_CMD_ERR_MODE_UNAVAILABLE:
1010 strncpy(error.error_description, "Requested mode not available",
1011 sizeof(error.error_description) - 1);
1012 break;
1013 case GCS_CMD_ERR_RATE_LIMITED:
1014 strncpy(error.error_description, "Command rate limited - too many requests",
1015 sizeof(error.error_description) - 1);
1016 break;
1017 default:
1018 snprintf(error.error_description, sizeof(error.error_description),
1019 "Command failed: %d", cmd_result);
1020 break;
1021 }
1022
1023 gcs_gateway_send_response(message->session_id, message->sequence_num,
1024 SCP_MSG_ERR_RESPONSE, &error, sizeof(error));
1025
1026 gateway_state.stats.failed_commands++;
1027 }
1028
1029 // Audit command attempt (T027 - User Story 4)
1030 uint32_t total_time = gcs_gateway_get_timestamp_ms() - processing_start;
1031 gcs_gateway_audit_command_attempt(message->session_id, message->message_type,
1032 cmd_result, total_time);
1033 gateway_state.stats.command_audits++;
1034
1035 ESP_LOGD(TAG, "Message processing complete in %lu ms", total_time);
1036}

1476static bool gcs_gateway_queue_telemetry(const scp_tlm_status_t *status, uint32_t session_id, uint8_t priority)
1477{
1478 if (!status || telemetry_state.queue_size >= 16)
1479 {
1480 telemetry_state.priority_drops++;
1481 return false;
1482 }
1483
1484 // T040: Insert message maintaining priority order (high priority first)
1485 uint8_t insert_idx = telemetry_state.queue_size;
1486
1487 // Find insertion point to maintain priority order
1488 for (uint8_t i = 0; i < telemetry_state.queue_size; i++)
1489 {
1490 if (priority > telemetry_state.priority_queue[i].priority)
1491 {
1492 insert_idx = i;
1493 break;
1494 }
1495 }
1496
1497 // Shift existing messages if needed
1498 if (insert_idx < telemetry_state.queue_size)
1499 {
1500 memmove(&telemetry_state.priority_queue[insert_idx + 1],
1501 &telemetry_state.priority_queue[insert_idx],
1502 (telemetry_state.queue_size - insert_idx) * sizeof(telemetry_state.priority_queue[0]));
1503 }
1504
1505 // Insert new message
1506 telemetry_state.priority_queue[insert_idx].message = *status;
1507 telemetry_state.priority_queue[insert_idx].session_id = session_id;
1508 telemetry_state.priority_queue[insert_idx].priority = priority;

1533static uint8_t gcs_gateway_process_priority_queue(void)
1534{
1535 uint8_t processed = 0;
1536 uint32_t current_time = gcs_gateway_get_timestamp_ms();
1537
1538 // Process messages from front of queue (highest priority first)
1539 while (telemetry_state.queue_size > 0 && processed < 3) // Limit per cycle
1540 {
1541 struct
1542 {
1543 scp_tlm_status_t message;
1544 uint32_t session_id;
1545 uint8_t priority;
1546 uint32_t timestamp_ms;
1547 uint8_t retry_count;
1548 } *queued = &telemetry_state.priority_queue[0];
1549
1550 // Check if message has aged out (older than 5 seconds)
1551 if (current_time - queued->timestamp_ms > 5000)
1552 {
1553 ESP_LOGV(TAG, "Dropping aged telemetry message (age: %lu ms)",
1554 current_time - queued->timestamp_ms);
1555 goto remove_message;
1556 }
1557
1558 // Try to send the message
1559 uint8_t sent_count = 0;
1560 if (queued->session_id == 0)
1561 {
1562 // Broadcast message
1563 sent_count = gcs_gateway_send_tlm_status_targeted(0);
1564 }
1565 else
1566 {
1567 // Targeted message
1568 sent_count = gcs_gateway_send_tlm_status_targeted(queued->session_id);
1569 }
1570
1571 if (sent_count > 0)
1572 {
1573 // Successfully sent
1574 processed++;
1575 goto remove_message;
1576 }
1577 else
1578 {
1579 // Failed to send - increment retry count
1580 queued->retry_count++;
1581 if (queued->retry_count >= 3)
1582 {
1583 ESP_LOGW(TAG, "Dropping telemetry after %d retries", queued->retry_count);
1584 goto remove_message;
1585 }
1586 else
1587 {
1588 // Keep in queue for retry, but don't block other messages
1589 break;
1590 }
1591 }
1592
1593 remove_message:
1594 // Remove message from front of queue
1595 if (telemetry_state.priority_queue[0].priority == SCP_PRIORITY_HIGH)
1596 {
1597 telemetry_state.queue_high_priority_count--;
1598 }
1599
1600 memmove(&telemetry_state.priority_queue[0],
1601 &telemetry_state.priority_queue[1],
1602 (telemetry_state.queue_size - 1) * sizeof(telemetry_state.priority_queue[0]));
1603
1604 telemetry_state.queue_size--;
1605 }
1606
1607 return processed;
1608}

1680 }

1702 session_list[i].auth_key,
1703 gateway_state.tx_buffer,
1704 sizeof(gateway_state.tx_buffer),
1705 &frame_size);
1706
1707 if (encode_result == SCP_RESULT_OK)
1708 {
1709 // TODO: Send via actual transport layer
1710 // For now, we simulate successful transmission
1711
1712 // T039: Update per-session statistics
1713 telemetry_state.session_config[session_idx].last_send_ms = current_time;
1714 telemetry_state.session_config[session_idx].messages_sent++;
1715
1716 ESP_LOGD(TAG, "Sent TLM_STATUS to session %lu role=%s (armed=%d, mode=%d)",
1717 session_list[i].session_id, scp_role_get_name(session_list[i].operator_role),
1718 status.armed, status.mode);
1719 sent_count++;

1722 {

1724 session_list[i].session_id, encode_result);

1917static void gcs_gateway_update_telemetry(void)
1918{
1919 if (!telemetry_state.enabled)
1920 {
1921 return;
1922 }
1923
1924 uint32_t current_time = gcs_gateway_get_timestamp_ms();
1925
1926 // First transmission - initialize timestamp
1927 if (telemetry_state.last_status_send_ms == 0)
1928 {
1929 telemetry_state.last_status_send_ms = current_time;
1930 return;
1931 }
1932
1933 uint32_t elapsed_ms = current_time - telemetry_state.last_status_send_ms;
1934
1935 // Check if it's time to send next status update
1936 if (elapsed_ms >= telemetry_state.target_interval_ms)
1937 {
1938 // T038: Calculate timing jitter for validation
1939 int32_t jitter_ms = (int32_t)elapsed_ms - (int32_t)telemetry_state.target_interval_ms;
1940
1941 // Check for rate limiting (excessive early transmission attempts)
1942 if (jitter_ms < -50) // Don't send more than 50ms early
1943 {
1944 telemetry_state.rate_limit_drops++;
1945 telemetry_state.last_drop_timestamp_ms = current_time;
1946 return;
1947 }
1948
1949 // T040: Generate status and determine priority
1950 scp_tlm_status_t status;
1951 esp_err_t gen_result = gcs_gateway_generate_tlm_status(&status);
1952 if (gen_result != ESP_OK)
1953 {
1954 return;
1955 }
1956
1957 uint8_t priority = gcs_gateway_get_telemetry_priority(&status);
1958
1959 // T040: For high priority or if queue is empty, try direct send
1960 uint8_t sent_count = 0;
1961 if (priority == SCP_PRIORITY_HIGH || telemetry_state.queue_size == 0)
1962 {
1963 sent_count = gcs_gateway_send_tlm_status();
1964 }
1965
1966 // T040: If direct send failed or for normal/low priority, queue message
1967 if (sent_count == 0 || priority != SCP_PRIORITY_HIGH)
1968 {
1969 if (priority == SCP_PRIORITY_HIGH && sent_count == 0)
1970 {
1971 // High priority failed direct send - queue immediately
1972 gcs_gateway_queue_telemetry(&status, 0, priority);
1973 }
1974 else if (priority != SCP_PRIORITY_HIGH)
1975 {
1976 // Normal/low priority - queue for batched sending
1977 gcs_gateway_queue_telemetry(&status, 0, priority);
1978 }
1979 }
1980
1981 if (sent_count > 0)
1982 {
1983 telemetry_state.last_status_send_ms = current_time;
1984 telemetry_state.total_transmissions++;
1985
1986 // T038: Update timing statistics
1987 if (abs(jitter_ms) <= 50) // Within 50ms tolerance
1988 {
1989 telemetry_state.on_time_transmissions++;
1990 }
1991 else if (jitter_ms > 0)
1992 {
1993 telemetry_state.late_transmissions++;
1994 }
1995
1996 // Update jitter statistics
1997 if (jitter_ms > telemetry_state.max_jitter_ms)
1998 telemetry_state.max_jitter_ms = jitter_ms;
1999 if (jitter_ms < telemetry_state.min_jitter_ms)
2000 telemetry_state.min_jitter_ms = jitter_ms;
2001
2002 // Update rolling average jitter
2003 telemetry_state.jitter_samples[telemetry_state.jitter_sample_idx] =
2004 (uint32_t)abs(jitter_ms);
2005 telemetry_state.jitter_sample_idx =
2006 (telemetry_state.jitter_sample_idx + 1) % 16;
2007
2008 // Calculate average jitter from samples
2009 uint32_t jitter_sum = 0;
2010 for (int i = 0; i < 16; i++)
2011 {
2012 jitter_sum += telemetry_state.jitter_samples[i];
2013 }
2014 telemetry_state.avg_jitter_ms = (int32_t)(jitter_sum / 16);
2015
2016 // Update gateway statistics
2017 gateway_state.stats.telemetry_sent += sent_count;
2018
2019 ESP_LOGV(TAG, "TLM_STATUS sent to %d sessions (jitter: %ld ms, rate: %d Hz)",
2020 sent_count, jitter_ms, telemetry_state.rate_hz);
2021 }
2022 }
2023
2024 // T040: Process priority queue regardless of periodic timing
2025 gcs_gateway_process_priority_queue();
2026}

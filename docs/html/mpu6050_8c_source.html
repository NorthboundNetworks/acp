69esp_err_t mpu6050_init(mpu6050_t *dev)
70{
71 if (dev == NULL) {
72 return ESP_ERR_INVALID_ARG;
73 }
74
75 // Initialize I2C bus if not already done
76 esp_err_t ret = bus_io_init_i2c();
77 if (ret != ESP_OK) {
78 ESP_LOGE(TAG, "Failed to initialize I2C bus: %s", esp_err_to_name(ret));
79 return ret;
80 }
81
82 // Clear device structure
83 memset(dev, 0, sizeof(mpu6050_t));
84 dev->i2c_addr = MPU6050_I2C_ADDR;
85
86 // Check WHO_AM_I register
87 uint8_t who_am_i;
88 ret = mpu6050_read_register(MPU6050_REG_WHO_AM_I, &who_am_i);
89 if (ret != ESP_OK) {
90 ESP_LOGE(TAG, "Failed to read WHO_AM_I: %s", esp_err_to_name(ret));
91 // Try alternate address
92 dev->i2c_addr = MPU6050_I2C_ADDR_ALT;
93 ret = mpu6050_read_register(MPU6050_REG_WHO_AM_I, &who_am_i);
94 if (ret != ESP_OK) {
95 ESP_LOGE(TAG, "Failed to communicate with MPU-6050");
96 return ESP_ERR_NOT_FOUND;
97 }
98 }
99
100 if (who_am_i != MPU6050_WHO_AM_I_VAL) {
101 ESP_LOGE(TAG, "Invalid WHO_AM_I: 0x%02X (expected 0x%02X)", who_am_i, MPU6050_WHO_AM_I_VAL);
102 return ESP_ERR_NOT_FOUND;
103 }
104
105 ESP_LOGI(TAG, "MPU-6050 found at address 0x%02X", dev->i2c_addr);
106
107 // Reset device
108 ret = mpu6050_write_register(MPU6050_REG_PWR_MGMT_1, 0x80);
109 if (ret != ESP_OK) {
110 ESP_LOGE(TAG, "Failed to reset device");
111 return ret;
112 }
113
114 vTaskDelay(pdMS_TO_TICKS(100)); // Wait for reset
115
116 // Wake up device and set clock source to PLL with X-axis gyro reference
117 ret = mpu6050_write_register(MPU6050_REG_PWR_MGMT_1, 0x01);
118 if (ret != ESP_OK) {
119 ESP_LOGE(TAG, "Failed to wake up device");
120 return ret;
121 }
122
123 // Configure sample rate divider (1kHz / (1 + 7) = 125Hz)
124 ret = mpu6050_write_register(MPU6050_REG_SMPRT_DIV, 0x07);
125 if (ret != ESP_OK) return ret;
126
127 // Configure DLPF (94Hz bandwidth)
128 ret = mpu6050_write_register(MPU6050_REG_CONFIG, MPU6050_DLPF_CFG_94HZ);
129 if (ret != ESP_OK) return ret;
130
131 // Configure gyroscope range (±250 deg/s)
132 ret = mpu6050_write_register(MPU6050_REG_GYRO_CONFIG, MPU6050_GYRO_FS_250);
133 if (ret != ESP_OK) return ret;
134
135 // Configure accelerometer range (±2g)
136 ret = mpu6050_write_register(MPU6050_REG_ACCEL_CONFIG, MPU6050_ACCEL_FS_2G);
137 if (ret != ESP_OK) return ret;
138
139 // Disable power management 2 (enable all axes)
140 ret = mpu6050_write_register(MPU6050_REG_PWR_MGMT_2, 0x00);
141 if (ret != ESP_OK) return ret;
142
143 // Perform self-test
144 ret = mpu6050_self_test(dev);
145 if (ret != ESP_OK) {
146 ESP_LOGW(TAG, "Self-test failed, but continuing initialization");
147 dev->health = SENSOR_HEALTH_FAULT;
148 strcpy(dev->last_error, "Self-test failed");
149 } else {
150 dev->health = SENSOR_HEALTH_CAL_REQ; // Needs calibration
151 strcpy(dev->last_error, "Calibration required");
152 }
153
154 dev->initialized = true;
155 g_initialized = true;
156 memcpy(&g_mpu6050, dev, sizeof(mpu6050_t));
157
158 ESP_LOGI(TAG, "MPU-6050 initialized successfully");
159 return ESP_OK;
160}

162esp_err_t mpu6050_read_raw(mpu6050_t *dev, mpu6050_raw_data_t *raw_data)
163{
164 if (dev == NULL || raw_data == NULL) {
165 return ESP_ERR_INVALID_ARG;
166 }
167
168 if (!dev->initialized) {
169 return ESP_ERR_INVALID_STATE;
170 }
171
172 uint8_t buffer[14];
173 esp_err_t ret = mpu6050_read_registers(MPU6050_REG_ACCEL_XOUT_H, buffer, 14);
174 if (ret != ESP_OK) {
175 dev->health = SENSOR_HEALTH_FAULT;
176 strcpy(dev->last_error, "Read failure");
177 return ret;
178 }
179
180 // Parse accelerometer data (big-endian)
181 raw_data->accel_x = (int16_t)((buffer[0] << 8) | buffer[1]);
182 raw_data->accel_y = (int16_t)((buffer[2] << 8) | buffer[3]);
183 raw_data->accel_z = (int16_t)((buffer[4] << 8) | buffer[5]);
184
185 // Parse temperature data
186 raw_data->temp = (int16_t)((buffer[6] << 8) | buffer[7]);
187
188 // Parse gyroscope data
189 raw_data->gyro_x = (int16_t)((buffer[8] << 8) | buffer[9]);
190 raw_data->gyro_y = (int16_t)((buffer[10] << 8) | buffer[11]);
191 raw_data->gyro_z = (int16_t)((buffer[12] << 8) | buffer[13]);
192
193 raw_data->timestamp = xTaskGetTickCount();
194
195 // Update health status periodically
196 if ((raw_data->timestamp - g_last_health_check) > pdMS_TO_TICKS(MPU6050_HEALTH_CHECK_INTERVAL_MS)) {
197 mpu6050_update_health(dev);
198 g_last_health_check = raw_data->timestamp;
199 }
200
201 return ESP_OK;
202}

204esp_err_t mpu6050_read_scaled(mpu6050_t *dev, mpu6050_data_t *data)
205{
206 if (dev == NULL || data == NULL) {
207 return ESP_ERR_INVALID_ARG;
208 }
209
210 mpu6050_raw_data_t raw;
211 esp_err_t ret = mpu6050_read_raw(dev, &raw);
212 if (ret != ESP_OK) {
213 return ret;
214 }
215
216 // Apply calibration offsets and scaling
217 data->accel_x = ((float)raw.accel_x - dev->cal.accel_offset_x) / MPU6050_ACCEL_SCALE;
218 data->accel_y = ((float)raw.accel_y - dev->cal.accel_offset_y) / MPU6050_ACCEL_SCALE;
219 data->accel_z = ((float)raw.accel_z - dev->cal.accel_offset_z) / MPU6050_ACCEL_SCALE;
220
221 data->gyro_x = ((float)raw.gyro_x - dev->cal.gyro_offset_x) / MPU6050_GYRO_SCALE;
222 data->gyro_y = ((float)raw.gyro_y - dev->cal.gyro_offset_y) / MPU6050_GYRO_SCALE;
223 data->gyro_z = ((float)raw.gyro_z - dev->cal.gyro_offset_z) / MPU6050_GYRO_SCALE;
224
225 data->temperature = (float)raw.temp / MPU6050_TEMP_SCALE + MPU6050_TEMP_OFFSET;
226 data->timestamp = raw.timestamp;
227
228 return ESP_OK;
229}

231esp_err_t mpu6050_calibrate_level(mpu6050_t *dev, mpu6050_cal_callback_t callback, void *user_data)
232{
233 if (dev == NULL) {
234 return ESP_ERR_INVALID_ARG;
235 }
236
237 if (!dev->initialized) {
238 return ESP_ERR_INVALID_STATE;
239 }
240
241 ESP_LOGI(TAG, "Starting level calibration");
242
243 if (callback) {
244 callback(MPU6050_CAL_STEP_START, "Place airframe stationary on a level surface. Starting in 3 seconds...", user_data);
245 }
246
247 vTaskDelay(pdMS_TO_TICKS(3000));
248
249 // Collect calibration samples
250 float accel_sum_x = 0, accel_sum_y = 0, accel_sum_z = 0;
251 float gyro_sum_x = 0, gyro_sum_y = 0, gyro_sum_z = 0;
252 float gyro_variance_x = 0, gyro_variance_y = 0, gyro_variance_z = 0;
253
254 if (callback) {
255 callback(MPU6050_CAL_STEP_SAMPLING, "Collecting samples...", user_data);
256 }
257
258 // First pass: collect sums
259 for (int i = 0; i < MPU6050_CAL_SAMPLES; i++) {
260 mpu6050_raw_data_t raw;
261 esp_err_t ret = mpu6050_read_raw(dev, &raw);
262 if (ret != ESP_OK) {
263 if (callback) {
264 callback(MPU6050_CAL_STEP_ERROR, "Failed to read sensor during calibration", user_data);
265 }
266 return ret;
267 }
268
269 accel_sum_x += raw.accel_x;
270 accel_sum_y += raw.accel_y;
271 accel_sum_z += raw.accel_z;
272 gyro_sum_x += raw.gyro_x;
273 gyro_sum_y += raw.gyro_y;
274 gyro_sum_z += raw.gyro_z;
275
276 vTaskDelay(pdMS_TO_TICKS(10)); // 100Hz sampling
277 }
278
279 // Calculate means
280 float gyro_mean_x = gyro_sum_x / MPU6050_CAL_SAMPLES;
281 float gyro_mean_y = gyro_sum_y / MPU6050_CAL_SAMPLES;
282 float gyro_mean_z = gyro_sum_z / MPU6050_CAL_SAMPLES;
283
284 // Second pass: calculate variance
285 for (int i = 0; i < MPU6050_CAL_SAMPLES; i++) {
286 mpu6050_raw_data_t raw;
287 esp_err_t ret = mpu6050_read_raw(dev, &raw);
288 if (ret != ESP_OK) {
289 return ret;
290 }
291
292 float diff_x = raw.gyro_x - gyro_mean_x;
293 float diff_y = raw.gyro_y - gyro_mean_y;
294 float diff_z = raw.gyro_z - gyro_mean_z;
295
296 gyro_variance_x += diff_x * diff_x;
297 gyro_variance_y += diff_y * diff_y;
298 gyro_variance_z += diff_z * diff_z;
299
300 vTaskDelay(pdMS_TO_TICKS(10));
301 }
302
303 gyro_variance_x /= MPU6050_CAL_SAMPLES;
304 gyro_variance_y /= MPU6050_CAL_SAMPLES;
305 gyro_variance_z /= MPU6050_CAL_SAMPLES;
306
307 // Check if variance is acceptable (device was stationary)
308 float max_variance = fmaxf(fmaxf(gyro_variance_x, gyro_variance_y), gyro_variance_z);
309 if (max_variance > (MPU6050_CAL_VARIANCE_THRESH * MPU6050_GYRO_SCALE * MPU6050_GYRO_SCALE)) {
310 if (callback) {
311 callback(MPU6050_CAL_STEP_ERROR, "Motion detected during calibration. Keep device stationary.", user_data);
312 }
313 return ESP_ERR_INVALID_STATE;
314 }
315
316 // Calculate calibration offsets
317 dev->cal.accel_offset_x = accel_sum_x / MPU6050_CAL_SAMPLES;
318 dev->cal.accel_offset_y = accel_sum_y / MPU6050_CAL_SAMPLES;
319 dev->cal.accel_offset_z = (accel_sum_z / MPU6050_CAL_SAMPLES) - MPU6050_ACCEL_SCALE; // Subtract 1g for Z-axis
320
321 dev->cal.gyro_offset_x = gyro_mean_x;
322 dev->cal.gyro_offset_y = gyro_mean_y;
323 dev->cal.gyro_offset_z = gyro_mean_z;
324
325 dev->cal.calibrated = true;
326 dev->cal.timestamp = xTaskGetTickCount();
327 dev->health = SENSOR_HEALTH_OK;
328 strcpy(dev->last_error, "");
329
330 if (callback) {
331 char msg[128];
332 snprintf(msg, sizeof(msg), "Calibration complete. Offsets: G(%+.1f, %+.1f, %+.1f) A(%+.1f, %+.1f, %+.1f)",
333 dev->cal.gyro_offset_x, dev->cal.gyro_offset_y, dev->cal.gyro_offset_z,
334 dev->cal.accel_offset_x, dev->cal.accel_offset_y, dev->cal.accel_offset_z);
335 callback(MPU6050_CAL_STEP_COMPLETE, msg, user_data);
336 }
337
338 ESP_LOGI(TAG, "Calibration successful");
339 return ESP_OK;
340}

388static void mpu6050_update_health(mpu6050_t *dev)
389{
390 // Read temperature for health monitoring
391 mpu6050_raw_data_t raw;
392 if (mpu6050_read_raw(dev, &raw) != ESP_OK) {
393 dev->health = SENSOR_HEALTH_FAULT;
394 strcpy(dev->last_error, "Communication failure");
395 return;
396 }
397
398 float temp = (float)raw.temp / MPU6050_TEMP_SCALE + MPU6050_TEMP_OFFSET;
399
400 // Check temperature range
401 if (temp > MPU6050_MAX_TEMP_C || temp < MPU6050_MIN_TEMP_C) {
402 dev->health = SENSOR_HEALTH_FAULT;
403 snprintf(dev->last_error, sizeof(dev->last_error), "Temperature out of range: %.1f°C", temp);
404 return;
405 }
406
407 // If we get here and no calibration, mark as needing calibration
408 if (!dev->cal.calibrated) {
409 dev->health = SENSOR_HEALTH_CAL_REQ;
410 strcpy(dev->last_error, "Calibration required");
411 } else {
412 dev->health = SENSOR_HEALTH_OK;
413 strcpy(dev->last_error, "");
414 }
415}

417static esp_err_t mpu6050_self_test(mpu6050_t *dev)
418{
419 // Basic self-test: verify we can read all registers and data is reasonable
420 uint8_t reg_vals[7];
421 esp_err_t ret;
422
423 // Read configuration registers
424 ret = mpu6050_read_register(MPU6050_REG_PWR_MGMT_1, &reg_vals[0]);
425 if (ret != ESP_OK) return ret;
426
427 ret = mpu6050_read_register(MPU6050_REG_GYRO_CONFIG, &reg_vals[1]);
428 if (ret != ESP_OK) return ret;
429
430 ret = mpu6050_read_register(MPU6050_REG_ACCEL_CONFIG, &reg_vals[2]);
431 if (ret != ESP_OK) return ret;
432
433 // Verify configuration is as expected
434 if ((reg_vals[0] & 0x07) != 0x01) { // Clock source should be PLL with X gyro
435 ESP_LOGW(TAG, "Unexpected power management config: 0x%02X", reg_vals[0]);
436 return ESP_ERR_INVALID_RESPONSE;
437 }
438
439 // Read sensor data to verify it's not stuck
440 mpu6050_raw_data_t raw1, raw2;
441 ret = mpu6050_read_raw(dev, &raw1);
442 if (ret != ESP_OK) return ret;
443
444 vTaskDelay(pdMS_TO_TICKS(50));
445
446 ret = mpu6050_read_raw(dev, &raw2);
447 if (ret != ESP_OK) return ret;
448
449 // Check that data changes (indicating sensor is working)
450 if (raw1.accel_x == raw2.accel_x && raw1.accel_y == raw2.accel_y &&
451 raw1.accel_z == raw2.accel_z && raw1.gyro_x == raw2.gyro_x &&
452 raw1.gyro_y == raw2.gyro_y && raw1.gyro_z == raw2.gyro_z) {
453 ESP_LOGW(TAG, "Sensor data appears stuck");
454 return ESP_ERR_INVALID_RESPONSE;
455 }
456
457 ESP_LOGI(TAG, "Self-test passed");
458 return ESP_OK;
459}

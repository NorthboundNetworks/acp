86int util_safe_strcat(char *dest, const char *src, size_t dest_size)
87{
88 if (!dest || !src || dest_size == 0)
89 {
90 return UTIL_ERR_INVALID;
91 }
92
93 size_t dest_len = strlen(dest);
94 if (dest_len >= dest_size)
95 {
96 return UTIL_ERR_BUFFER_TOO_SMALL;
97 }
98
99 size_t remaining = dest_size - dest_len - 1;
100 size_t src_len = strlen(src);
101 size_t append_len = src_len < remaining ? src_len : remaining;
102
103 memcpy(dest + dest_len, src, append_len);
104 dest[dest_len + append_len] = '\0';
105
106 return (int)(dest_len + append_len);
107}

109char *util_trim_string(char *str)
110{
111 if (!str)
112 {
113 return NULL;
114 }
115
116 /* Trim leading whitespace */
117 while (isspace((unsigned char)*str))
118 {
119 str++;
120 }
121
122 /* Trim trailing whitespace */
123 char *end = str + strlen(str) - 1;
124 while (end > str && isspace((unsigned char)*end))
125 {
126 end--;
127 }
128 *(end + 1) = '\0';
129
130 return str;
131}

199int util_string_split(char *str, char delimiter, char **tokens, size_t max_tokens)
200{
201 if (!str || !tokens || max_tokens == 0)
202 {
203 return UTIL_ERR_INVALID;
204 }
205
206 size_t token_count = 0;
207 char *current = str;
208
209 while (*current && token_count < max_tokens)
210 {
211 /* Skip leading delimiters */
212 while (*current == delimiter)
213 {
214 current++;
215 }
216
217 if (*current == '\0')
218 {
219 break;
220 }
221
222 /* Mark start of token */
223 tokens[token_count] = current;
224 token_count++;
225
226 /* Find end of token */
227 while (*current && *current != delimiter)
228 {
229 current++;
230 }
231
232 /* Null-terminate token if delimiter found */
233 if (*current == delimiter)
234 {
235 *current = '\0';
236 current++;
237 }
238 }
239
240 return (int)token_count;
241}

265int util_create_directory(const char *dirname)
266{
267 if (!dirname)
268 {
269 return UTIL_ERR_INVALID;
270 }
271
272 /* Try to create directory */
273 if (mkdir(dirname, 0755) == 0)
274 {
275 return UTIL_SUCCESS;
276 }
277
278 /* If it already exists, that's fine */
279 if (errno == EEXIST && util_directory_exists(dirname))
280 {
281 return UTIL_SUCCESS;
282 }
283
284 /* Try to create parent directory first */
285 char *parent = strdup(dirname);
286 if (!parent)
287 {
288 return UTIL_ERR_MEMORY;
289 }
290
291 char *last_slash = strrchr(parent, '/');
292 if (last_slash && last_slash != parent)
293 {
294 *last_slash = '\0';
295 int result = util_create_directory(parent);
296 if (result != UTIL_SUCCESS)
297 {
298 free(parent);
299 return result;
300 }
301
302 /* Try again after creating parent */
303 if (mkdir(dirname, 0755) == 0 ||
304 (errno == EEXIST && util_directory_exists(dirname)))
305 {
306 free(parent);
307 return UTIL_SUCCESS;
308 }
309 }
310
311 free(parent);
312 return UTIL_ERR_FILE_ERROR;
313}

399int util_bytes_to_hex(const uint8_t *data, size_t data_len,
400 char *hex_str, size_t hex_str_size, bool uppercase)
401{
402 if (!data || !hex_str || hex_str_size < data_len * 2 + 1)
403 {
404 return UTIL_ERR_INVALID;
405 }
406
407 const char *hex_chars = uppercase ? "0123456789ABCDEF" : "0123456789abcdef";
408
409 for (size_t i = 0; i < data_len; i++)
410 {
411 hex_str[i * 2] = hex_chars[(data[i] >> 4) & 0xF];
412 hex_str[i * 2 + 1] = hex_chars[data[i] & 0xF];
413 }
414
415 hex_str[data_len * 2] = '\0';
416 return (int)(data_len * 2);
417}

419int util_hex_to_bytes(const char *hex_str, uint8_t *data, size_t data_size)
420{
421 if (!hex_str || !data)
422 {
423 return UTIL_ERR_INVALID;
424 }
425
426 size_t hex_len = strlen(hex_str);
427 if (hex_len % 2 != 0 || hex_len / 2 > data_size)
428 {
429 return UTIL_ERR_INVALID;
430 }
431
432 for (size_t i = 0; i < hex_len; i += 2)
433 {
434 char hex_byte[3] = {hex_str[i], hex_str[i + 1], '\0'};
435 char *endptr;
436 long value = strtol(hex_byte, &endptr, 16);
437
438 if (*endptr != '\0')
439 {
440 return UTIL_ERR_INVALID;
441 }
442
443 data[i / 2] = (uint8_t)value;
444 }
445
446 return (int)(hex_len / 2);
447}

449int util_format_bytes(uint64_t bytes, char *buffer, size_t buffer_size)
450{
451 if (!buffer || buffer_size < 16)
452 {
453 return UTIL_ERR_INVALID;
454 }
455
456 const char *units[] = {"B", "KB", "MB", "GB", "TB"};
457 const size_t num_units = sizeof(units) / sizeof(units[0]);
458
459 double size = (double)bytes;
460 size_t unit_index = 0;
461
462 while (size >= 1024.0 && unit_index < num_units - 1)
463 {
464 size /= 1024.0;
465 unit_index++;
466 }
467
468 if (unit_index == 0)
469 {
470 snprintf(buffer, buffer_size, "%llu %s", (unsigned long long)bytes, units[unit_index]);
471 }
472 else
473 {
474 snprintf(buffer, buffer_size, "%.1f %s", size, units[unit_index]);
475 }
476
477 return UTIL_SUCCESS;
478}

480uint32_t util_crc32(const uint8_t *data, size_t len)
481{
482 if (!data)
483 {

487 uint32_t crc = 0xFFFFFFFF;
488
489 for (size_t i = 0; i < len; i++)
490 {
491 crc = crc32_table[(crc ^ data[i]) & 0xFF] ^ (crc >> 8);
492 }
493
494 return crc ^ 0xFFFFFFFF;

660int util_backoff_next_delay(util_backoff_t *backoff)
661{
662 if (!backoff)
663 {
664 return -1;
665 }
666
667 /* Check if we've exceeded maximum attempts */
668 if (backoff->max_attempts > 0 && backoff->attempt >= backoff->max_attempts)
669 {
670 return -1;
671 }
672
673 /* Check if we've exceeded total timeout */
674 if (backoff->total_timeout_ms > 0)
675 {
676 time_t now = time(NULL);
677 int64_t elapsed_ms = (int64_t)(now - backoff->start_time) * 1000;
678 if (elapsed_ms >= backoff->total_timeout_ms)
679 {
680 return -1;
681 }
682 }
683
684 int delay_ms;
685
686 if (backoff->exponential)
687 {
688 /* Exponential backoff: base_delay * 2^attempt */
689 delay_ms = backoff->base_delay_ms;
690 for (int i = 0; i < backoff->attempt; i++)
691 {
692 delay_ms *= 2;
693 /* Check for overflow and cap at max_delay */
694 if (delay_ms > backoff->max_delay_ms || delay_ms < 0)
695 {
696 delay_ms = backoff->max_delay_ms;
697 break;
698 }
699 }
700 }
701 else
702 {
703 /* Linear backoff: base_delay * (attempt + 1) */
704 delay_ms = backoff->base_delay_ms * (backoff->attempt + 1);
705 }
706
707 /* Cap at maximum delay */
708 if (delay_ms > backoff->max_delay_ms)
709 {
710 delay_ms = backoff->max_delay_ms;
711 }
712
713 /* If we have a total timeout, make sure we don't exceed it */
714 if (backoff->total_timeout_ms > 0)
715 {
716 time_t now = time(NULL);
717 int64_t elapsed_ms = (int64_t)(now - backoff->start_time) * 1000;
718 int64_t remaining_ms = backoff->total_timeout_ms - elapsed_ms;
719
720 if (delay_ms > remaining_ms)
721 {
722 delay_ms = (int)remaining_ms;
723 if (delay_ms <= 0)
724 {
725 return -1; /* No time left */
726 }

728 }
729
730 backoff->attempt++;

775int util_backoff_get_stats(const util_backoff_t *backoff, int *total_attempts,
776 int64_t *total_time_ms, int *last_delay_ms)
777{
778 if (!backoff)
779 {
780 return UTIL_ERR_INVALID;
781 }
782
783 if (total_attempts)
784 {
785 *total_attempts = backoff->attempt;
786 }
787
788 if (total_time_ms)
789 {
790 time_t now = time(NULL);
791 *total_time_ms = (int64_t)(now - backoff->start_time) * 1000;
792 }
793
794 if (last_delay_ms)
795 {
796 /* Calculate what the last delay would have been */
797 if (backoff->attempt == 0)
798 {
799 *last_delay_ms = 0;
800 }
801 else
802 {
803 int delay;
804 if (backoff->exponential)
805 {
806 delay = backoff->base_delay_ms;
807 for (int i = 1; i < backoff->attempt; i++)
808 {
809 delay *= 2;
810 if (delay > backoff->max_delay_ms || delay < 0)
811 {
812 delay = backoff->max_delay_ms;
813 break;
814 }
815 }
816 }
817 else
818 {
819 delay = backoff->base_delay_ms * backoff->attempt;
820 }
821
822 *last_delay_ms = (delay > backoff->max_delay_ms) ? backoff->max_delay_ms : delay;
823 }
824 }
825
826 return UTIL_SUCCESS;
827}

41int serial_open(serial_session_t *session, const char *port, int baud)
42{
43 if (!session || !port)
44 {
45 return SERIAL_ERR_INVALID;
46 }
47
48 /* Initialize session structure */
49 memset(session, 0, sizeof(serial_session_t));
50 strncpy(session->port, port, sizeof(session->port) - 1);
51 session->port[sizeof(session->port) - 1] = '\0';
52 session->baud = baud;
53 session->fd = -1;
54 session->connected = false;
55
56 /* Open serial device */
57 session->fd = open(port, O_RDWR | O_NOCTTY | O_NONBLOCK);
58 if (session->fd < 0)
59 {
60 return SERIAL_ERR_OPEN;
61 }
62
63 /* Configure serial parameters */
64 int result = serial_config(session);
65 if (result != SERIAL_SUCCESS)
66 {
67 close(session->fd);
68 session->fd = -1;
69 return result;
70 }
71
72 session->connected = true;
73 return SERIAL_SUCCESS;
74}

76int serial_config(serial_session_t *session)
77{
78 if (!session || session->fd < 0)
79 {
80 return SERIAL_ERR_INVALID;
81 }
82
83 struct termios tty;
84
85 /* Get current attributes */
86 if (tcgetattr(session->fd, &tty) != 0)
87 {
88 return SERIAL_ERR_CONFIG;
89 }
90
91 /* Set baud rate */
92 speed_t baud_const = serial_get_baud_constant(session->baud);
93 cfsetispeed(&tty, baud_const);
94 cfsetospeed(&tty, baud_const);
95
96 /* 8 bits, no parity, 1 stop bit */
97 tty.c_cflag &= (tcflag_t)~PARENB; /* No parity */
98 tty.c_cflag &= (tcflag_t)~CSTOPB; /* 1 stop bit */
99 tty.c_cflag &= (tcflag_t)~CSIZE; /* Clear size bits */
100 tty.c_cflag |= (tcflag_t)CS8; /* 8 data bits */
101
102 /* Hardware flow control */
103 tty.c_cflag &= (tcflag_t)~CRTSCTS; /* Disable RTS/CTS */
104 tty.c_cflag |= (tcflag_t)(CREAD | CLOCAL); /* Enable receiver, ignore modem lines */
105
106 /* Input flags */
107 tty.c_iflag &= (tcflag_t) ~(IXON | IXOFF | IXANY); /* Disable software flow control */
108 tty.c_iflag &= (tcflag_t) ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR | ICRNL);
109
110 /* Output flags */
111 tty.c_oflag &= (tcflag_t)~OPOST; /* Raw output */
112
113 /* Local flags */
114 tty.c_lflag &= (tcflag_t) ~(ICANON | ECHO | ECHOE | ISIG); /* Raw mode */
115
116 /* Timeout settings for non-blocking reads */
117 tty.c_cc[VMIN] = 0; /* Return immediately */
118 tty.c_cc[VTIME] = 1; /* 100ms timeout */
119
120 /* Apply settings */
121 if (tcsetattr(session->fd, TCSANOW, &tty) != 0)
122 {
123 return SERIAL_ERR_CONFIG;
124 }
125
126 /* Flush buffers */
127 tcflush(session->fd, TCIOFLUSH);
128
129 return SERIAL_SUCCESS;
130}

149int serial_read(serial_session_t *session, int timeout_ms)
150{
151 if (!session || session->fd < 0)
152 {
153 return SERIAL_ERR_INVALID;
154 }
155
156 fd_set read_fds;
157 struct timeval timeout;
158
159 FD_ZERO(&read_fds);
160 FD_SET(session->fd, &read_fds);
161
162 timeout.tv_sec = timeout_ms / 1000;
163 timeout.tv_usec = (timeout_ms % 1000) * 1000;
164
165 int select_result = select(session->fd + 1, &read_fds, NULL, NULL, &timeout);
166
167 if (select_result < 0)
168 {
169 return SERIAL_ERR_IO;
170 }
171
172 if (select_result == 0)
173 {
174 return SERIAL_ERR_TIMEOUT;
175 }
176
177 /* Calculate available space in circular buffer */
178 size_t available_space = SERIAL_BUFFER_SIZE - 1 -
179 ((session->rx_head - session->rx_tail) % SERIAL_BUFFER_SIZE);
180
181 if (available_space == 0)
182 {
183 return 0; /* Buffer full */
184 }
185
186 /* Read directly into buffer (simplified for MVP) */
187 uint8_t temp_buffer[256];
188 ssize_t bytes_read = read(session->fd, temp_buffer, sizeof(temp_buffer));
189
190 if (bytes_read < 0)
191 {
192 if (errno == EAGAIN || errno == EWOULDBLOCK)
193 {
194 return 0; /* No data available */
195 }
196 session->connected = false;
197 return SERIAL_ERR_DISCONNECT;
198 }
199
200 if (bytes_read == 0)
201 {
202 session->connected = false;
203 return SERIAL_ERR_DISCONNECT;
204 }
205
206 /* Copy to circular buffer */
207 for (ssize_t i = 0; i < bytes_read && available_space > 0; i++)
208 {
209 session->rx_buffer[session->rx_head] = temp_buffer[i];
210 session->rx_head = (session->rx_head + 1) % SERIAL_BUFFER_SIZE;
211 available_space--;
212 }
213
214 session->bytes_received += (uint32_t)bytes_read;
215 return (int)bytes_read;
216}

218int serial_write(serial_session_t *session, const uint8_t *data, size_t len)
219{
220 if (!session || !data || session->fd < 0)
221 {
222 return SERIAL_ERR_INVALID;
223 }
224
225 ssize_t bytes_written = write(session->fd, data, len);
226
227 if (bytes_written < 0)
228 {
229 if (errno == EAGAIN || errno == EWOULDBLOCK)
230 {
231 return 0; /* Would block */
232 }
233 session->connected = false;
234 return SERIAL_ERR_DISCONNECT;
235 }
236
237 session->bytes_sent += (uint32_t)bytes_written;
238 return (int)bytes_written;

275size_t serial_read_buffer(serial_session_t *session, uint8_t *buffer, size_t max_len)
276{
277 if (!session || !buffer || max_len == 0)
278 {
279 return 0;
280 }
281
282 size_t bytes_read = 0;
283
284 while (bytes_read < max_len && session->rx_tail != session->rx_head)
285 {
286 buffer[bytes_read] = session->rx_buffer[session->rx_tail];
287 session->rx_tail = (session->rx_tail + 1) % SERIAL_BUFFER_SIZE;
288 bytes_read++;
289 }
290
291 return bytes_read;
292}

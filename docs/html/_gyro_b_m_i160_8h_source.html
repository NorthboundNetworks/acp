120 int begin(BusDevice * bus, uint8_t addr) override
121 {
122 setBus(bus, addr);
123
124 if(!testConnection()) return 0;
125
126 // reset device
127 _bus->writeByte(_addr, BMI160_RA_CMD, BMI160_CMD_SOFT_RESET);
128 delay(1);
129
130 if(_bus->isSPI())
131 {
132 /*
133 Issue a dummy-read to force the device into SPI comms mode
134 see https://www.bosch-sensortec.com/media/boschsensortec/downloads/datasheets/bst-bmi160-ds000.pdf section 3.2.1
135 */
136 uint8_t dummy = 0;
137 _bus->readByte(_addr, BMI160_CMD_SPI_MODE, &dummy);
138 delay(100);
139 }
140
141 // Start up accelerometer
142 _bus->writeByte(_addr, BMI160_RA_CMD, BMI160_CMD_ACC_MODE_NORMAL);
143 delay(100);
144
145 // Start up gyroscope
146 _bus->writeByte(_addr, BMI160_RA_CMD, BMI160_CMD_GYR_MODE_NORMAL);
147 delay(100);
148
149 // Set up full scale Accel range. +-16G
150 _bus->writeByte(_addr, BMI160_RA_ACCEL_RANGE, BMI160_ACCEL_RANGE_16G);
151 delay(1);
152
153 // Set up full scale Gyro range. +-2000dps
154 _bus->writeByte(_addr, BMI160_RA_GYRO_RANGE, BMI160_GYRO_RANGE_2000);
155 delay(1);
156
157 // Enable accel offset
158 //_bus->writeBitsLsb(_addr, BMI160_RA_OFFSET_6, BMI160_ACC_OFFSET_EN, BMI160_ACC_OFFSET_LEN, BMI160_RESULT_OK);
159 delay(1);
160
161 // Enable gyro offset
162 //_bus->writeBitsLsb(_addr, BMI160_RA_OFFSET_6, BMI160_GYR_OFFSET_EN, BMI160_GYR_OFFSET_LEN, BMI160_RESULT_OK);
163 delay(1);
164
165 // Enable data ready interrupt
166 _bus->writeByte(_addr, BMI160_REG_INT_EN1, BMI160_INT_EN1_DRDY);
167 delay(1);
168
169 // Enable INT1 pin
170 _bus->writeByte(_addr, BMI160_REG_INT_OUT_CTRL, BMI160_INT_OUT_CTRL_INT1_CONFIG);
171 delay(1);
172
173 // Map data ready interrupt to INT1 pin
174 _bus->writeByte(_addr, BMI160_REG_INT_MAP1, BMI160_REG_INT_MAP1_INT1_DRDY);
175 delay(1);
176
177 // Set Accel rate 1600HZ
178 _bus->writeByte(_addr, BMI160_RA_ACCEL_CONF, BMI160_ACCEL_RATE_800HZ);
179 delay(1);
180
181 // Set Gyro rate 3200HZ
182 _bus->writeByte(_addr, BMI160_RA_GYRO_CONF, BMI160_GYRO_RATE_3200HZ);
183 delay(1);
184
185 return 1;
186 }

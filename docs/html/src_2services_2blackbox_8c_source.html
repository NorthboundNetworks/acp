97esp_err_t blackbox_init(void)
98{
99 if (s_blackbox_initialized)
100 {
101 ESP_LOGW(TAG, "Blackbox already initialized");
102 return ESP_OK;
103 }
104
105 ESP_LOGI(TAG, "Initializing blackbox logging service...");
106
107 // Mount FATFS partition
108 esp_err_t ret = blackbox_mount_fatfs();
109 if (ret != ESP_OK)
110 {
111 ESP_LOGE(TAG, "Failed to mount FATFS partition: %s", esp_err_to_name(ret));
112 return ret;
113 }
114
115 // Ensure mission logs directory exists
116 ret = blackbox_ensure_mission_dir();
117 if (ret != ESP_OK)
118 {
119 ESP_LOGE(TAG, "Failed to create mission logs directory: %s", esp_err_to_name(ret));
120 return ret;
121 }
122
123 // Create event queue for background writing
124 s_event_queue = xQueueCreate(s_event_queue_size, sizeof(blackbox_event_t));
125 if (s_event_queue == NULL)
126 {
127 ESP_LOGE(TAG, "Failed to create blackbox event queue");
128 return ESP_FAIL;
129 }
130
131 // Create background writer task
132 ESP_LOGI(TAG, "Creating background writer task...");
133 xTaskCreatePinnedToCore(
134 blackbox_writer_task,
135 "blackbox_writer",
136 4096,
137 NULL,
138 2, // Lower priority than control loop
139 &s_writer_task_handle,
140 0 // Core 0 for I/O tasks
141 );
142
143 if (s_writer_task_handle == NULL)
144 {
145 ESP_LOGE(TAG, "Failed to create blackbox writer task");
146 return ESP_FAIL;
147 }
148
149 s_blackbox_initialized = true;
150 ESP_LOGI(TAG, "Blackbox logging service initialized successfully");
151 return ESP_OK;
152}

154static esp_err_t blackbox_mount_fatfs(void)
155{
156 ESP_LOGI(TAG, "Mounting FATFS partition...");
157
158 // Find the partition labeled 'storage' or similar
159 const esp_partition_t *partition = esp_partition_find_first(
160 ESP_PARTITION_TYPE_DATA,
161 ESP_PARTITION_SUBTYPE_DATA_FAT,
162 NULL);
163
164 if (partition == NULL)
165 {
166 ESP_LOGE(TAG, "No FAT partition found");
167 return ESP_FAIL;
168 }
169
170 ESP_LOGI(TAG, "Found partition '%s' at offset 0x%lx with size %ld bytes",
171 partition->label, partition->address, partition->size);
172
173 // Mount configuration
174 const esp_vfs_fat_mount_config_t mount_config = {
175 .max_files = 8,
176 .format_if_mount_failed = true, // Format if needed
177 .allocation_unit_size = CONFIG_WL_SECTOR_SIZE,
178 .use_one_fat = false};
179
180 esp_err_t ret = esp_vfs_fat_spiflash_mount_rw_wl(
181 "/fatfs",
182 partition->label,
183 &mount_config,
184 &s_wl_handle);
185
186 if (ret != ESP_OK)
187 {
188 ESP_LOGE(TAG, "Failed to mount FATFS: %s", esp_err_to_name(ret));
189 return ret;
190 }
191
192 s_fatfs_mounted = true;
193 ESP_LOGI(TAG, "FATFS mounted successfully at /fatfs");
194 return ESP_OK;
195}

227static void blackbox_writer_task(void *arg)
228{
229 (void)arg;
230 ESP_LOGI(TAG, "Blackbox writer task started");
231
232 blackbox_event_t event;
233 const TickType_t timeout = pdMS_TO_TICKS(BLACKBOX_WRITE_TIMEOUT_MS);
234
235 while (1)
236 {
237 // Wait for events in the queue
238 if (xQueueReceive(s_event_queue, &event, timeout) == pdTRUE)
239 {
240 // Process the event if we have an active mission
241 if (s_mission_active && s_mission_file != NULL)
242 {
243 // Write NDJSON line to file
244 if (fprintf(s_mission_file, "%s\n", event.json_line) < 0)
245 {
246 ESP_LOGE(TAG, "Failed to write event to mission file");
247 }
248 else
249 {
250 // Flush to ensure data is written
251 fflush(s_mission_file);
252 ESP_LOGD(TAG, "Wrote event: %s", event.json_line);
253 }
254 }
255 else
256 {
257 ESP_LOGD(TAG, "Discarding event (no active mission): %s", event.json_line);
258 }
259 }
260 // Timeout occurred - continue loop (allows for graceful shutdown)
261 }
262}

264void blackbox_deinit(void)
265{
266 if (!s_blackbox_initialized)
267 {
268 return;
269 }
270
271 ESP_LOGI(TAG, "Shutting down blackbox service...");
272
273 // Close any active mission
274 blackbox_close_mission_file();
275
276 // Stop writer task
277 if (s_writer_task_handle != NULL)
278 {
279 vTaskDelete(s_writer_task_handle);
280 s_writer_task_handle = NULL;
281 }
282
283 // Clean up event queue
284 if (s_event_queue != NULL)
285 {
286 vQueueDelete(s_event_queue);
287 s_event_queue = NULL;
288 }
289
290 // Unmount FATFS if mounted
291 if (s_fatfs_mounted)
292 {
293 esp_vfs_fat_spiflash_unmount_rw_wl("/fatfs", s_wl_handle);
294 s_fatfs_mounted = false;
295 s_wl_handle = WL_INVALID_HANDLE;
296 }
297
298 s_blackbox_initialized = false;
299 ESP_LOGI(TAG, "Blackbox service shut down");
300}

307esp_err_t blackbox_start_mission(const char *mission_id)
308{
309 if (!s_blackbox_initialized || mission_id == NULL)
310 {
311 return ESP_ERR_INVALID_ARG;
312 }
313
314 ESP_LOGI(TAG, "Starting mission: %s", mission_id);
315
316 // Close any existing mission
317 if (s_mission_active)
318 {
319 blackbox_close_mission_file();
320 }
321
322 // Rotate old missions (keep-last-1 policy)
323 esp_err_t ret = blackbox_rotate_missions();
324 if (ret != ESP_OK)
325 {
326 ESP_LOGW(TAG, "Mission rotation failed, continuing anyway");
327 }
328
329 // Create new mission file
330 ret = blackbox_create_mission_file(mission_id);
331 if (ret != ESP_OK)
332 {
333 ESP_LOGE(TAG, "Failed to create mission file: %s", esp_err_to_name(ret));
334 return ret;
335 }
336
337 // Update state
338 strncpy(s_current_mission_id, mission_id, sizeof(s_current_mission_id) - 1);
339 s_current_mission_id[sizeof(s_current_mission_id) - 1] = '\0';
340 s_mission_active = true;
341
342 // Log mission start event
343 char attrs[128];
344 snprintf(attrs, sizeof(attrs), "{\"mission_id\":\"%s\",\"start_time\":%lld}",
345 mission_id, blackbox_get_timestamp_ms());
346 blackbox_log_event("system", "info", attrs);
347
348 ESP_LOGI(TAG, "Mission started successfully: %s", mission_id);
349 return ESP_OK;
350}

380esp_err_t blackbox_log_event(const char *event_type, const char *severity, const char *attrs)
381{
382 if (!s_blackbox_initialized || event_type == NULL || attrs == NULL)
383 {
384 return ESP_ERR_INVALID_ARG;
385 }
386
387 // Create NDJSON event structure
388 blackbox_event_t event;
389 int64_t timestamp = blackbox_get_timestamp_ms();
390
391 // Format NDJSON line: {"ts":1234,"type":"event_type","severity":"info","attrs":{...},"schema_version":"1.0"}
392 if (severity != NULL)
393 {
394 snprintf(event.json_line, sizeof(event.json_line),
395 "{\"ts\":%lld,\"type\":\"%s\",\"severity\":\"%s\",\"attrs\":%s,\"schema_version\":\"%s\"}",
396 timestamp, event_type, severity, attrs, NDJSON_SCHEMA_VERSION);
397 }
398 else
399 {
400 snprintf(event.json_line, sizeof(event.json_line),
401 "{\"ts\":%lld,\"type\":\"%s\",\"attrs\":%s,\"schema_version\":\"%s\"}",
402 timestamp, event_type, attrs, NDJSON_SCHEMA_VERSION);
403 }
404
405 event.timestamp_ms = timestamp;
406
407 // Queue event for background writing (non-blocking)
408 if (xQueueSend(s_event_queue, &event, 0) != pdTRUE)
409 {
410 ESP_LOGW(TAG, "Event queue full, dropping event: %s", event_type);
411 return ESP_ERR_TIMEOUT;
412 }
413
414 return ESP_OK;
415}

429static esp_err_t blackbox_rotate_missions(void)
430{
431 ESP_LOGI(TAG, "Rotating missions (keep-last-1 policy)");
432
433 // Open missions directory
434 DIR *dir = opendir(MISSION_LOG_BASE_PATH);
435 if (dir == NULL)
436 {
437 ESP_LOGE(TAG, "Failed to open missions directory");
438 return ESP_FAIL;
439 }
440
441 // Find and remove old mission files
442 struct dirent *entry;
443 while ((entry = readdir(dir)) != NULL)
444 {
445 if (strstr(entry->d_name, ".ndjson") != NULL)
446 {
447 char old_path[256];
448 size_t base_len = strlen(MISSION_LOG_BASE_PATH);
449 size_t max_name_len = (sizeof(old_path) > (base_len + 2)) ? (sizeof(old_path) - base_len - 2) : 0; // account for '/' and NUL
450 snprintf(old_path, sizeof(old_path), "%s/%.*s", MISSION_LOG_BASE_PATH, (int)max_name_len, entry->d_name);
451
452 if (unlink(old_path) == 0)
453 {
454 ESP_LOGI(TAG, "Removed old mission file: %s", entry->d_name);
455 }
456 else
457 {
458 ESP_LOGW(TAG, "Failed to remove old mission file: %s", entry->d_name);
459 }
460 }
461 }
462
463 closedir(dir);
464 return ESP_OK;
465}

467static esp_err_t blackbox_create_mission_file(const char *mission_id)
468{
469 size_t base_len = strlen(MISSION_LOG_BASE_PATH);
470 const char *prefix = "/mission_";
471 const char *suffix = ".ndjson";
472 size_t overhead = base_len + strlen(prefix) + strlen(suffix) + 1; // +1 for NUL
473 size_t max_id_len = (sizeof(s_mission_file_path) > overhead) ? (sizeof(s_mission_file_path) - overhead) : 0;
474 snprintf(s_mission_file_path, sizeof(s_mission_file_path),
475 "%s%.*s%.*s%.*s",
476 MISSION_LOG_BASE_PATH,
477 (int)strlen(prefix), prefix,
478 (int)max_id_len, mission_id,
479 (int)strlen(suffix), suffix);
480
481 s_mission_file = fopen(s_mission_file_path, "w");
482 if (s_mission_file == NULL)
483 {
484 ESP_LOGE(TAG, "Failed to create mission file: %s", s_mission_file_path);
485 return ESP_FAIL;
486 }
487
488 ESP_LOGI(TAG, "Created mission file: %s", s_mission_file_path);
489 return ESP_OK;
490}

508esp_err_t blackbox_log_sensor_summary(void)
509{
510 if (!s_blackbox_initialized || !s_mission_active)
511 {
512 return ESP_ERR_INVALID_STATE;
513 }
514
515 // Collect sensor readings
516 char attrs[512];
517 int pos = 0;
518
519 // Start JSON object
520 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "{");
521
522 // IMU data (MPU6050)
523 mpu6050_t *imu = mpu6050_get_instance();
524 if (imu && imu->initialized)
525 {
526 mpu6050_data_t imu_data;
527 bool read_ok = (mpu6050_read_scaled(imu, &imu_data) == ESP_OK);
528 pos += snprintf(attrs + pos, sizeof(attrs) - pos,
529 "\"accel\":{\"x\":%.3f,\"y\":%.3f,\"z\":%.3f},"
530 "\"gyro\":{\"x\":%.3f,\"y\":%.3f,\"z\":%.3f},"
531 "\"temp\":%.1f,",
532 read_ok ? imu_data.accel_x : 0.0,
533 read_ok ? imu_data.accel_y : 0.0,
534 read_ok ? imu_data.accel_z : 0.0,
535 read_ok ? imu_data.gyro_x : 0.0,
536 read_ok ? imu_data.gyro_y : 0.0,
537 read_ok ? imu_data.gyro_z : 0.0,
538 read_ok ? imu_data.temperature : 0.0);
539 }
540 else
541 {
542 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "\"imu\":\"unavailable\",");
543 }
544
545 // Magnetometer data (HMC5883L)
546 hmc5883l_t *mag = hmc5883l_get_instance();
547 if (mag && mag->initialized)
548 {
549 hmc5883l_data_t mag_data;
550 bool read_ok = (hmc5883l_read_scaled(mag, &mag_data) == ESP_OK);
551 pos += snprintf(attrs + pos, sizeof(attrs) - pos,
552 "\"mag\":{\"x\":%.3f,\"y\":%.3f,\"z\":%.3f},",
553 read_ok ? mag_data.mag_x : 0.0,
554 read_ok ? mag_data.mag_y : 0.0,
555 read_ok ? mag_data.mag_z : 0.0);
556 }
557 else
558 {
559 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "\"mag\":\"unavailable\",");
560 }
561
562 // Barometer data (BMP388)
563 bmp388_t *baro = bmp388_get_instance();
564 if (baro && baro->initialized)
565 {
566 bmp388_data_t baro_data;
567 bool read_ok = (bmp388_read_scaled(baro, &baro_data) == ESP_OK);
568 pos += snprintf(attrs + pos, sizeof(attrs) - pos,
569 "\"pressure\":%.2f,\"altitude\":%.2f,",
570 read_ok ? baro_data.pressure : 0.0,
571 read_ok ? baro_data.altitude : 0.0);
572 }
573 else
574 {
575 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "\"baro\":\"unavailable\",");
576 }
577
578 // GPS data (NEO-6M)
579 neo6m_t *gps = neo6m_get_instance();
580 if (gps && gps->initialized && gps->fix_quality > 0)
581 {
582 pos += snprintf(attrs + pos, sizeof(attrs) - pos,
583 "\"gps\":{\"lat\":%.7f,\"lon\":%.7f,\"alt\":%.1f,\"sats\":%d,\"fix\":%d},",
584 gps->position.latitude, gps->position.longitude, gps->position.altitude,
585 gps->num_satellites, gps->fix_quality);
586 }
587 else
588 {
589 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "\"gps\":\"no_fix\",");
590 }
591
592 // System information
593 uint32_t heap_free = esp_get_free_heap_size();
594 uint32_t heap_min = esp_get_minimum_free_heap_size();
595
596 pos += snprintf(attrs + pos, sizeof(attrs) - pos,
597 "\"system\":{\"heap_free\":%lu,\"heap_min\":%lu,\"uptime_ms\":%lld}",
598 heap_free, heap_min, blackbox_get_timestamp_ms());
599
600 // Close JSON object
601 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "}");
602
603 // Ensure we don't overflow
604 if (pos >= sizeof(attrs))
605 {
606 ESP_LOGW(TAG, "Sensor summary JSON truncated");
607 }
608
609 // Log the sensor summary event
610 return blackbox_log_event("sensor_summary", "info", attrs);
611}

715 "\"gps\":{\"health\":\"%s\",\"fix\":%d,\"sats\":%d,\"lat\":%.7f,\"lon\":%.7f}",
716 (gps->fix_quality > 0) ? "ok" : "no_fix",
717 gps->fix_quality, gps->num_satellites,
718 gps->position.latitude, gps->position.longitude);
719 }
720 else
721 {
722 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "\"gps\":{\"health\":\"unavail\"}");
723 }
724
725 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "},"); // Close sensors
726
727 // System performance metrics
728 uint32_t heap_free = esp_get_free_heap_size();
729 uint32_t heap_min = esp_get_minimum_free_heap_size();
730 pos += snprintf(attrs + pos, sizeof(attrs) - pos,
731 "\"system\":{\"heap_free\":%lu,\"heap_min\":%lu,\"uptime_ms\":%lld},",
732 heap_free, heap_min, blackbox_get_timestamp_ms());
733
734 // Safety gate states (if available)
735 pos += snprintf(attrs + pos, sizeof(attrs) - pos,
736 "\"safety\":{\"collection_time\":%lld}",
737 blackbox_get_timestamp_ms());
738
739 // Close JSON structure
740 pos += snprintf(attrs + pos, sizeof(attrs) - pos, "}");
741
742 // Ensure no overflow
743 if (pos >= sizeof(attrs))
744 {
745 ESP_LOGW(TAG, "Evidence collection JSON truncated");
746 }
747
748 // Log the evidence collection event
749 esp_err_t ret = blackbox_log_event("evidence_collection", "critical", attrs);
750 if (ret == ESP_OK)
751 {
752 ESP_LOGI(TAG, "Evidence collected successfully for: %s", incident_type);
753 }
754 else
755 {
756 ESP_LOGE(TAG, "Failed to log evidence: %s", esp_err_to_name(ret));
757 }
758
759 return ret;
760}

762esp_err_t blackbox_log_critical_fault(const char *fault_type, const char *component, const char *details)
763{
764 if (fault_type == NULL || component == NULL)
765 {
766 ESP_LOGE(TAG, "Critical fault logging requires fault_type and component");
767 return ESP_ERR_INVALID_ARG;
768 }
769
770 ESP_LOGE(TAG, "CRITICAL FAULT: %s in %s - %s", fault_type, component, details ? details : "no details");
771
772 // First, log the critical fault event itself
773 char fault_attrs[512];
774 snprintf(fault_attrs, sizeof(fault_attrs),
775 "{\"fault_type\":\"%s\",\"component\":\"%s\",\"details\":\"%s\",\"timestamp\":%lld}",
776 fault_type, component, details ? details : "", blackbox_get_timestamp_ms());
777
778 esp_err_t ret = blackbox_log_event("critical_fault", "error", fault_attrs);
779 if (ret != ESP_OK)
780 {
781 ESP_LOGE(TAG, "Failed to log critical fault event: %s", esp_err_to_name(ret));
782 return ret;
783 }
784
785 // Immediately collect comprehensive evidence
786 char evidence_context[256];
787 snprintf(evidence_context, sizeof(evidence_context),
788 "critical_fault_%s_%s", fault_type, component);
789
790 ret = blackbox_collect_evidence("critical_fault", evidence_context);
791 if (ret != ESP_OK)
792 {
793 ESP_LOGW(TAG, "Failed to collect evidence after critical fault: %s", esp_err_to_name(ret));
794 // Don't return error - the fault was logged successfully
795 }
796
797 return ESP_OK;
798}

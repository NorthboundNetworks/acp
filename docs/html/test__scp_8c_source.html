31/* Helper function to create valid SCP frame */
32static void create_test_frame(scp_frame_t *frame, scp_frame_type_t type,
33 uint16_t sequence, const uint8_t *payload, uint16_t payload_len)
34{
35 frame->header.sync = SCP_SYNC_BYTE;
36 frame->header.type = (uint8_t)type;
37 frame->header.sequence = sequence;
38 frame->header.length = payload_len;
39
40 if (payload && payload_len > 0)
41 {
42 memcpy(frame->payload, payload, payload_len);
43 }
44
45 /* Calculate CRC over header + payload */
46 uint8_t crc_data[SCP_HEADER_SIZE + SCP_MAX_PAYLOAD_SIZE];
47 memcpy(crc_data, &frame->header, SCP_HEADER_SIZE);
48 if (payload_len > 0)
49 {
50 memcpy(crc_data + SCP_HEADER_SIZE, payload, payload_len);
51 }
52
53 frame->crc = scp_calculate_crc(crc_data, SCP_HEADER_SIZE + payload_len);
54 frame->total_length = SCP_HEADER_SIZE + payload_len + SCP_CRC_SIZE;
55}

95/* Test CRC calculation */
96static void test_scp_calculate_crc(void)
97{
98 uint8_t test_data[] = {0x01, 0x02, 0x03, 0x04};
99 uint16_t crc1 = scp_calculate_crc(test_data, sizeof(test_data));
100 uint16_t crc2 = scp_calculate_crc(test_data, sizeof(test_data));
101
102 /* CRC should be deterministic */
103 TEST_ASSERT_EQUAL_UINT16(crc1, crc2);
104
105 /* Different data should produce different CRC */
106 test_data[0] = 0xFF;
107 uint16_t crc3 = scp_calculate_crc(test_data, sizeof(test_data));
108 TEST_ASSERT_NOT_EQUAL(crc1, crc3);
109}

245/* Test telemetry extraction from valid SCP frame */
246static void test_extract_telemetry_valid(void)
247{
248 /* Create mock telemetry payload based on contract */
249 uint8_t mock_payload[] = {
250 0x01, /* armed status */
251 0x02, /* flight mode */
252 0x03, /* GPS fix (3 = 3D) */
253 100, /* battery percent */
254 0x10, 0x00, /* roll (16 deg*100) */
255 0x20, 0x00, /* pitch (32 deg*100) */
256 0x30, 0x00, /* yaw (48 deg*100) */
257 0x78, 0x56, 0x34, 0x12, /* uptime_ms (0x12345678) */
258 0xF0, 0xDE, 0xBC, 0x9A, /* timestamp_ms (0x9ABCDEF0) */
259 0xA5, 0xC3, 0x00 /* padding for 21 bytes */
260 };
261
262 /* Create DATA frame with mock payload */
263 create_test_frame(&test_frame, SCP_FRAME_DATA, 1, mock_payload, sizeof(mock_payload));
264
265 /* Extract telemetry */
266 telemetry_frame_t telemetry;
267 int result = scp_extract_telemetry(&test_frame, &telemetry);
268
269 TEST_ASSERT_EQUAL_INT(SCP_SUCCESS, result);
270
271 /* Verify extracted values according to contract */
272 TEST_ASSERT_EQUAL_STRING("3D", telemetry.gps_fix); /* GPS fix = 3 */
273 TEST_ASSERT_FLOAT_WITHIN(0.1f, 22.0f, telemetry.battery_v); /* 100 * 0.12 + 10.0 = 22.0V */
274 TEST_ASSERT_FLOAT_WITHIN(0.1f, 45.0f, telemetry.cpu_temp_c); /* Fixed mock value */
275
276 /* CPU load derived from uptime: (0x12345678 % 10000) / 100.0 */
277 float expected_cpu_load = (float)((0x12345678 % 10000) / 100.0);
278 TEST_ASSERT_FLOAT_WITHIN(0.1f, expected_cpu_load, telemetry.cpu_load_pct);
279
280 /* Yaw conversion: 48 * 0.01745329 / 100.0 (deg/100 to rad/s) */
281 float expected_gyro_z = (float)(48 * 0.01745329 / 100.0);
282 TEST_ASSERT_FLOAT_WITHIN(0.001f, expected_gyro_z, telemetry.imu_gyro_z);
283}

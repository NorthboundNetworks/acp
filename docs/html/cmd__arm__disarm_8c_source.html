55mavlink_command_result_t cmd_arm_disarm_handle_command_long(const mavlink_command_long_t *cmd)
56{
57 if (!cmd)
58 {
59 ESP_LOGE(TAG, "Null command received");
60 return MAVLINK_CMD_RESULT_FAILED;
61 }
62
63 g_command_sequence++;
64
65 ESP_LOGI(TAG, "Received COMMAND_LONG: cmd=%d, param1=%.1f, target_system=%d, target_component=%d",
66 (int)cmd->command, cmd->param1, cmd->target_system, cmd->target_component);
67
68 switch ((uint16_t)cmd->command)
69 {
70 case MAV_CMD_COMPONENT_ARM_DISARM:
71 return handle_arm_disarm_command(cmd);
72
73 case MAV_CMD_NAV_TAKEOFF:
74 ESP_LOGW(TAG, "TAKEOFF command received but not implemented");
75 return MAVLINK_CMD_RESULT_UNSUPPORTED;
76
77 case MAV_CMD_NAV_LAND:
78 ESP_LOGW(TAG, "LAND command received but not implemented");
79 return MAVLINK_CMD_RESULT_UNSUPPORTED;
80
81 default:
82 ESP_LOGW(TAG, "Unsupported command: %d", (int)cmd->command);
83 return MAVLINK_CMD_RESULT_UNSUPPORTED;
84 }
85}

87static mavlink_command_result_t handle_arm_disarm_command(const mavlink_command_long_t *cmd)
88{
89 // MAV_CMD_COMPONENT_ARM_DISARM parameters:
90 // param1: 1 to arm, 0 to disarm
91 // param2: force arming/disarming (override safety checks) - NOT SUPPORTED for safety
92 // param3-7: reserved
93
94 bool request_arm = (cmd->param1 > 0.5f);
95 bool force_override = (cmd->param2 > 0.5f);
96
97 ESP_LOGI(TAG, "ARM/DISARM command: %s%s",
98 request_arm ? "ARM" : "DISARM",
99 force_override ? " (FORCE)" : "");
100
101 // Never allow force override for safety reasons
102 if (force_override)
103 {
104 ESP_LOGW(TAG, "Force override requested but denied for safety");
105 telemetry_send_statustext("Force override not permitted", 4); // MAV_SEVERITY_ERROR
106 return MAVLINK_CMD_RESULT_DENIED;
107 }
108
109 // Check if we're in a valid mode for arming/disarming
110 scout_mode_t current_mode = mode_manager_get_mode();
111 if (request_arm && current_mode == SCOUT_MODE_CONFIG)
112 {
113 ESP_LOGW(TAG, "Cannot arm in CONFIG mode");
114 telemetry_send_statustext("Cannot arm in CONFIG mode", 4);
115 return MAVLINK_CMD_RESULT_DENIED;
116 }
117
118 // Get current safety status
119 safety_gates_t gates;
120 safety_manager_get_gates(&gates);
121
122 if (request_arm)
123 {
124 return handle_arm_request(&gates);
125 }
126 else
127 {
128 return handle_disarm_request(&gates);
129 }
130}

132static mavlink_command_result_t handle_arm_request(const safety_gates_t *gates)
133{
134 // Check all safety gates before allowing arming
135 if (!gates->arming_allowed)
136 {
137 const char *block_reason = safety_manager_get_arming_block_reason();
138
139 ESP_LOGW(TAG, "Arming denied: %s", block_reason);
140
141 char statustext[64];
142 snprintf(statustext, sizeof(statustext), "Arming denied: %s", block_reason);
143 telemetry_send_statustext(statustext, 4); // MAV_SEVERITY_ERROR
144
145 return MAVLINK_CMD_RESULT_DENIED;
146 }
147
148 // Additional checks for MAVLink arming
149 if (!gates->preflight_ok)
150 {
151 ESP_LOGW(TAG, "Arming denied: preflight check failed");
152 telemetry_send_statustext("Preflight check failed", 4);
153 return MAVLINK_CMD_RESULT_DENIED;
154 }
155
156 if (!gates->sensors_ok)
157 {
158 ESP_LOGW(TAG, "Arming denied: sensor health check failed");
159 telemetry_send_statustext("Sensor health check failed", 4);
160 return MAVLINK_CMD_RESULT_DENIED;
161 }
162
163 if (!gates->cpu_budget_ok)
164 {
165 ESP_LOGW(TAG, "Arming denied: CPU budget exceeded");
166 telemetry_send_statustext("CPU budget exceeded", 4);
167 return MAVLINK_CMD_RESULT_DENIED;
168 }
169
170 if (!gates->link_ok)
171 {
172 ESP_LOGW(TAG, "Arming denied: link quality insufficient");
173 telemetry_send_statustext("Link quality insufficient", 4);
174 return MAVLINK_CMD_RESULT_DENIED;
175 }
176
177 // All checks passed - proceed with arming
178 ESP_LOGI(TAG, "Arming approved - all safety checks passed");
179
180 // Call the existing telemetry handler to update armed state
181 telemetry_handle_arm_disarm_command(1, 0);
182
183 g_armed_via_mavlink = true;
184 g_last_arm_disarm_time = xTaskGetTickCount() * portTICK_PERIOD_MS;
185
186 telemetry_send_statustext("Armed via MAVLink", 6); // MAV_SEVERITY_INFO
187
188 return MAVLINK_CMD_RESULT_ACCEPTED;
189}

191static mavlink_command_result_t handle_disarm_request(const safety_gates_t *gates)

202
203 telemetry_send_statustext("Disarmed via MAVLink", 6); // MAV_SEVERITY_INFO
204
205 return MAVLINK_CMD_RESULT_ACCEPTED;
206}

72 _bus->read(_addr, BMP085_MEASUREMENT_REG, 2, buffer);
73
74 // calbrate temp

76 if(ut == 0) return NAN;

85 virtual float readPressure() override
86 {
87 uint8_t buffer[3];
88 _bus->read(_addr, BMP085_MEASUREMENT_REG, 3, buffer);
89
90 uint32_t up = ((uint32_t)buffer[0] << 16) | ((uint32_t)buffer[1] << 8) | buffer[2];
91
92 uint8_t oss = (_mode & 0xC0) >> 6;
93 if(_mode & 0x34) up = up >> (8 - oss);
94 if(up == 0) return NAN;
95
96 int32_t b6 = _t_fine - 4000;
97 int32_t x1 = ((int32_t)_cal.b2 * ((b6 * b6) >> 12)) >> 11;
98 int32_t x2 = ((int32_t)_cal.ac2 * b6) >> 11;
99 int32_t x3 = x1 + x2;
100 int32_t b3 = ((((int32_t)_cal.ac1 * 4 + x3) << oss) + 2) >> 2;
101 x1 = ((int32_t)_cal.ac3 * b6) >> 13;
102 x2 = ((int32_t)_cal.b1 * ((b6 * b6) >> 12)) >> 16;
103 x3 = ((x1 + x2) + 2) >> 2;
104 uint32_t b4 = ((uint32_t)_cal.ac4 * (uint32_t)(x3 + 32768)) >> 15;
105 uint32_t b7 = ((uint32_t)up - b3) * (uint32_t)(50000UL >> oss);
106
107 int32_t p = 0;
108 if (b7 < 0x80000000) {
109 p = (b7 << 1) / b4;
110 } else {
111 p = (b7 / b4) << 1;
112 }
113 x1 = (p >> 8) * (p >> 8);
114 x1 = (x1 * 3038) >> 16;
115 x2 = (-7357 * p) >> 16;
116
117 p = p + ((x1 + x2 + (int32_t)3791) >> 4);
118
119 return p;
120 }

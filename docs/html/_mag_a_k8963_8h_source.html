32
33 int begin(BusDevice * bus, uint8_t addr) override
34 {
35 setBus(bus, addr);
36
37 if(!testConnection()) return 0;
38
39 // set AK8963 to Power Down
40 _bus->writeByte(_addr, AK8963_CNTL1, AK8963_PWR_DOWN);
41 delay(AK8963_INIT_DELAY); // long wait between AK8963 mode changes
42
43 // set AK8963 to FUSE ROM access
44 _bus->writeByte(_addr, AK8963_CNTL1, AK8963_FUSE_ROM);
45 delay(AK8963_INIT_DELAY);
46
47 /* get the magnetometer calibration */
48 // read the AK8963 ASA registers and compute magnetometer scale factors
49 _bus->read(_addr, AK8963_ASA, 3, buffer);
50 // align CW90_FLIP (swap X Y)
51 scale.y = ((((float)buffer[0]) - 128.0f)/(256.0f) + 1.0f) * 4912.0f / 32760.0f; // micro Tesla
52 scale.x = ((((float)buffer[1]) - 128.0f)/(256.0f) + 1.0f) * 4912.0f / 32760.0f; // micro Tesla
53 scale.z = ((((float)buffer[2]) - 128.0f)/(256.0f) + 1.0f) * 4912.0f / 32760.0f; // micro Tesla
54 //scale *= 10.f; // convert to milli Gauss
55 scale *= 0.01f; // convert to Gauss
56
57 // set AK8963 to Power Down
58 _bus->writeByte(_addr, AK8963_CNTL1, AK8963_PWR_DOWN);
59 delay(AK8963_INIT_DELAY);
60
61 // set AK8963 to 16 bit resolution, 100 Hz update rate
62 _bus->writeByte(_addr, AK8963_CNTL1, AK8963_CNT_MEAS2);
63 delay(AK8963_INIT_DELAY);
64
65 // instruct master to get 7 bytes of data from the AK8963 at the sample rate
66 _bus->read(_addr, AK8963_HXL, 7, buffer);
67
68 return 1;
69 }
70
71 int readMag(VectorInt16& v) override
72 {
73 _bus->readFast(_addr, AK8963_HXL, 7, buffer);
74
75 // align CW90_FLIP (swap X Y, invert Z)
76 v.y = ((((int16_t)buffer[1]) << 8) | buffer[0]);
77 v.x = ((((int16_t)buffer[3]) << 8) | buffer[2]);
78 v.z = -((((int16_t)buffer[5]) << 8) | buffer[4]);
79 return 1;
80 }

36 int process(Device::SerialDevice& s) const
37 {
38 Rc::CrsfMessage f;
39 switch(_current)
40 {
41 case CRSF_TELEMETRY_STATE_ATTI:
42 attitude(f);
43 send(f, s);
44 _current = CRSF_TELEMETRY_STATE_BAT;
45 break;
46 case CRSF_TELEMETRY_STATE_BAT:
47 battery(f);
48 send(f, s);
49 _current = CRSF_TELEMETRY_STATE_FM;
50 break;
51 case CRSF_TELEMETRY_STATE_FM:
52 flightMode(f);
53 send(f, s);
54 _current = CRSF_TELEMETRY_STATE_GPS;
55 break;
56 case CRSF_TELEMETRY_STATE_GPS:
57 gps(f);
58 send(f, s);
59 _current = CRSF_TELEMETRY_STATE_BARO;
60 break;
61 case CRSF_TELEMETRY_STATE_BARO:
62 vario(f);
63 send(f, s);
64 _current = CRSF_TELEMETRY_STATE_HB;
65 break;
66 case CRSF_TELEMETRY_STATE_HB:
67 heartbeat(f);
68 send(f, s);
69 _current = CRSF_TELEMETRY_STATE_ATTI;
70 break;
71 }
72
73 return 1;
74 }

99 void attitude(Rc::CrsfMessage& msg) const
100 {
101 msg.prepare(Rc::CRSF_FRAMETYPE_ATTITUDE);
102
103 int16_t r = toAngle(_model.state.attitude.euler.x);
104 int16_t p = toAngle(_model.state.attitude.euler.y);
105 int16_t y = toAngle(_model.state.attitude.euler.z);
106
107 msg.writeU16(Utils::toBigEndian16(r));
108 msg.writeU16(Utils::toBigEndian16(p));
109 msg.writeU16(Utils::toBigEndian16(y));
110
111 msg.finalize();
112 }
113
114 void battery(Rc::CrsfMessage& msg) const
115 {
116 msg.prepare(Rc::CRSF_FRAMETYPE_BATTERY_SENSOR);
117
118 uint16_t voltage = Utils::clamp(lrintf(_model.state.battery.voltage * 10.0f), 0l, 32000l);
119 uint16_t current = Utils::clamp(lrintf(_model.state.battery.current * 10.0f), 0l, 32000l);
120 uint32_t mahDrawn = 0;
121 uint8_t remainPerc = lrintf(_model.state.battery.percentage);
122
123 msg.writeU16(Utils::toBigEndian16(voltage));
124 msg.writeU16(Utils::toBigEndian16(current));
125 msg.writeU8(mahDrawn >> 16);
126 msg.writeU8(mahDrawn >> 8);
127 msg.writeU8(mahDrawn);
128 msg.writeU8(remainPerc);
129
130 msg.finalize();
131 }

133 void flightMode(Rc::CrsfMessage& msg) const
134 {
135 msg.prepare(Rc::CRSF_FRAMETYPE_FLIGHT_MODE);
136
137 if(_model.armingDisabled()) msg.writeString("!DIS");
138 if(_model.isModeActive(MODE_FAILSAFE)) msg.writeString("!FS,");
139 if(_model.isModeActive(MODE_ARMED)) msg.writeString("ARM,");
140 if(_model.isModeActive(MODE_AIRMODE)) msg.writeString("AIR,");
141 if(_model.isModeActive(MODE_ANGLE)) msg.writeString("STAB,");
142 msg.writeU8(0);
143
144 msg.finalize();
145 }
146
147 void gps(Rc::CrsfMessage& msg) const
148 {
149 msg.prepare(Rc::CRSF_FRAMETYPE_GPS);
150
151 msg.writeU32(Utils::toBigEndian32(_model.state.gps.location.raw.lat)); // deg * 1e7
152 msg.writeU32(Utils::toBigEndian32(_model.state.gps.location.raw.lon)); // deg * 1e7
153 msg.writeU16(Utils::toBigEndian16((_model.state.gps.velocity.raw.groundSpeed * 36 + 500) / 1000)); // in km/h * 10
154 msg.writeU16(Utils::toBigEndian16((_model.state.gps.velocity.raw.heading + 500) / 1000)); // deg * 10
155 uint16_t altitude = std::clamp((_model.state.gps.location.raw.height + 500) / 1000, (int32_t)-900, (int32_t)5000) + 1000; // m
156 msg.writeU16(Utils::toBigEndian16(altitude));
157 msg.writeU8(_model.state.gps.numSats);
158
159 msg.finalize();
160 }
161
162 void vario(Rc::CrsfMessage& msg) const
163 {
164 // https://github.com/crsf-wg/crsf/wiki/CRSF_FRAMETYPE_BARO_ALTITUDE
165 msg.prepare(Rc::CRSF_FRAMETYPE_BARO_ALTITUDE);
166
167 msg.writeU16(Utils::toBigEndian16(0)); // (cm + 10000) or (m + 0 | 0x8000)
168 msg.writeU16(Utils::toBigEndian16(0)); // cm/s
169
170 msg.finalize();
171 }
172
173 void heartbeat(Rc::CrsfMessage& msg) const
174 {

177 msg.writeU16(Utils::toBigEndian16(Rc::CRSF_ADDRESS_FLIGHT_CONTROLLER));
178
179 msg.finalize();
180 }
